24 grids.clear();
25
26 const double angle_increment = split_params.fov_angle / split_params.n_angular_grids;
27 const int n_radius_grids =
28 static_cast<int>(split_params.max_range / split_params.radius_resolution) + 1;
29
30 // Matrix Initialization
31 grids.resize(n_radius_grids);
32 for (int radius = 0; radius < n_radius_grids; ++radius) {
33 grids[radius].resize(split_params.n_angular_grids);
34 for (int angle = 0; angle < split_params.n_angular_grids; ++angle) {
35 grids[radius][angle] = std::make_shared<pcl::PointCloud<pcl::PointXYZI>>();
36 }
37 }
38
39 // Matrix Population
40 for (const auto& point : *cloud) {
41 double radius = std::sqrt(point.x * point.x + point.y * point.y);
42 double angle = std::atan2(point.y, point.x) * 180.0 / M_PI;
43
44 if (angle < 0) {
45 angle += 360.0;
46 }
47
48 const int row = static_cast<int>(radius / split_params.radius_resolution);
49 const int column = static_cast<int>(angle / angle_increment) % split_params.n_angular_grids;
50
51 if (row < static_cast<int>(grids.size())) {
52 grids[row][column]->points.push_back(point);
53 }
54 }

67 } else {

73 split_point_cloud(point_cloud, grids, split_params);
74
75 int count = 0;
76
77 for (const auto& grid_row : grids) {

81 }

86
