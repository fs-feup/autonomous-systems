3double VelocityPlanning::find_curvature(const PathPoint &p1, const PathPoint &p2,
4 const PathPoint &p3) {
5 // lengths of the sides of the triangle formed by the three points
6 double a = std::hypot(p2.position.x - p1.position.x, p2.position.y - p1.position.y);
7 double b = std::hypot(p3.position.x - p2.position.x, p3.position.y - p2.position.y);
8 double c = std::hypot(p3.position.x - p1.position.x, p3.position.y - p1.position.y);
9
10 // area of the triangle using the determinant method
11 double area = 0.5 * std::abs(p1.position.x * (p2.position.y - p3.position.y) +
12 p2.position.x * (p3.position.y - p1.position.y) +
13 p3.position.x * (p1.position.y - p2.position.y));
14
15 // To avoid division by zero in case of near duplicate points
16 if (a * b * c < epsilon) {
17 return 0.0;
18 }
19
20 /* The Menger curvature is given by the formula: K = 4A / (abc),
21 where A is the triangle area and a,b,c are the side lengths*/
22 return 4 * area / (a * b * c);
23}

25void VelocityPlanning::point_speed(const std::vector<double> &curvatures,
26 std::vector<double> &velocities) {
27 for (const auto &k : curvatures) {
28 // This is a straight line, there is no curvature limit on the velocity
29 if (std::abs(k) < epsilon) {
30 velocities.push_back(config_.desired_velocity_);
31 continue;
32 }
33
34 double velocity = std::sqrt(config_.lateral_acceleration_ / std::abs(k));
35 velocities.push_back(std::min(velocity, config_.desired_velocity_));
36 }
37 // The last point is always the minimum velocity for safety
38 velocities.back() = config_.minimum_velocity_;
39}

41void VelocityPlanning::acceleration_limiter(const std::vector<PathPoint> &points,
42 std::vector<double> &velocities,
43 const std::vector<double> &curvatures) {
44 velocities[0] = config_.minimum_velocity_;
45 for (int i = 1; i < (int)points.size(); i++) {
46 double dx = points[i].position.x - points[i - 1].position.x;
47 double dy = points[i].position.y - points[i - 1].position.y;
48 double d = std::sqrt(dx * dx + dy * dy);
49
50 // lateral acceleration at previous point: v(i-1)^2 * curvature
51 double ay = velocities[i - 1] * velocities[i - 1] * std::abs(curvatures[i - 1]);
52 double ax_max = std::sqrt(
53 std::max(0.0, config_.lateral_acceleration_ * config_.lateral_acceleration_ - ay * ay));
54
55 // Cap by acceleration limit
56 ax_max = std::min(ax_max, config_.acceleration_);
57
58 // v_i^2 = v_(i-1)^2 + 2 * a_x_available * d
59 double max_velocity =
60 std::sqrt(std::max(0.0, velocities[i - 1] * velocities[i - 1] + 2 * ax_max * d));
61 velocities[i] = std::min(velocities[i], max_velocity);
62 }
63}

65void VelocityPlanning::braking_limiter(std::vector<PathPoint> &points,
66 std::vector<double> &velocities,
67 const std::vector<double> &curvatures) {
68
69 for (int i = static_cast<int>(points.size()) - 2; i >= 0; i--) {
70 // Calculate segment distance
71 int j = i + 1;
72 double distance = std::hypot(points[j].position.x - points[i].position.x,
73 points[j].position.y - points[i].position.y);
74
75 // Lateral acceleration at the next point: a = v(j)^2 * curvature
76 double ay = velocities[j] * velocities[j] * std::abs(curvatures[j]);
77 double ax_brake = std::sqrt(std::max(0.0, config_.lateral_acceleration_ * config_.lateral_acceleration_ - ay * ay));
78
79 // Cap by braking limit
80 ax_brake = std::min(ax_brake, config_.braking_acceleration_);
81
82 // Correct kinematic speed calculation
83 // v_f² = v_i² + 2ad
84 double max_speed =
85 std::sqrt(std::max(0.0, std::pow(velocities[j], 2) - 2 * ax_brake * distance));
86
87 max_speed = std::min(max_speed, this->config_.desired_velocity_);
88 velocities[i] = std::min(max_speed, velocities[i]);
89 }
90}

92void VelocityPlanning::set_velocity(std::vector<PathPoint> &final_path) {
93 if (!config_.use_velocity_planning_ || final_path.size() <= 2) {
94 for (auto &p : final_path){
95 p.ideal_velocity = config_.minimum_velocity_;
96 }
97 return;
98 }
99
100 const int n = static_cast<int>(final_path.size());
101
102 std::vector<double> curvatures(n, 0.0);
103 for (int i = 1; i < n - 1; ++i){
104 curvatures[i] = find_curvature(final_path[i - 1], final_path[i], final_path[i + 1]);
105 }
106
107 std::vector<double> velocities;
108 point_speed(curvatures, velocities);
109 acceleration_limiter(final_path, velocities, curvatures);
110 braking_limiter(final_path, velocities, curvatures);
111
112 for (int i = 0; i < n; ++i) {
113 velocities[i] = std::max(velocities[i], config_.minimum_velocity_);
114 velocities[i] = std::min(velocities[i], config_.desired_velocity_);
115 final_path[i].ideal_velocity = velocities[i];
116 }

119void VelocityPlanning::trackdrive_velocity(std::vector<PathPoint> &final_path) {
120 if (!config_.use_velocity_planning_ || final_path.size() <= 2) {
121 for (auto &p : final_path){
122 p.ideal_velocity = config_.minimum_velocity_;
123 }
124 return;
125 }
126
127 int path_size = static_cast<int>(final_path.size()) - 1;
128
129 // Triple the path
130 std::vector<PathPoint> triple_path;
131 triple_path.reserve(3 * path_size);
132 for (int lap = 0; lap < 3; ++lap) {
133 for (int i = 0; i < path_size; ++i) {
134 triple_path.push_back(final_path[i]);
135 }
136 }
137
138 set_velocity(triple_path);
139
140 // Extract middle path velocities ----
141 int offset = path_size; // middle lap start
142 for (int i = 0; i < path_size; ++i) {
143 final_path[i].ideal_velocity = triple_path[offset + i].ideal_velocity;
144 }
145
146 // Close loop explicitly
147 final_path.back().ideal_velocity = final_path.front().ideal_velocity;
148}
