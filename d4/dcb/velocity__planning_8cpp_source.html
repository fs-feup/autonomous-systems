3double VelocityPlanning::find_circle_center(const PathPoint &point1, const PathPoint &point2,
4 const PathPoint &point3) {
5 double x1 = point1.position.x;
6 double y1 = point1.position.y;
7 double x2 = point2.position.x;
8 double y2 = point2.position.y;
9 double x3 = point3.position.x;
10 double y3 = point3.position.y;
11
12 PathPoint mid1 = PathPoint((x1 + x2) / 2, (y1 + y2) / 2, 0);
13 PathPoint mid2 = PathPoint((x2 + x3) / 2, (y2 + y3) / 2, 0);
14
15 double slope1 = MAXFLOAT;
16 double slope2 = MAXFLOAT;
17 double slope1_perpendicular = 10'000;
18 double slope2_perpendicular = 10'000;
19 if (x2 != x1) {
20 slope1 = (y2 - y1) / (x2 - x1);
21 }
22
23 if (x3 != x2) {
24 slope2 = (y3 - y2) / (x3 - x2);
25 }
26
27 if (slope1 != 0) {
28 slope1_perpendicular = -1 / slope1;
29 }
30
31 if (slope2 != 0) {
32 slope2_perpendicular = -1 / slope2;
33 }
34
35 if (slope1_perpendicular == slope2_perpendicular) {
36 return 10'000;
37 }
38
39 double center_x = (slope1_perpendicular * mid1.position.x -
40 slope2_perpendicular * mid2.position.x + mid2.position.y - mid1.position.y) /
41 (slope1_perpendicular - slope2_perpendicular);
42 double center_y = slope1_perpendicular * (center_x - mid1.position.x) + mid1.position.y;
43 double radius = std::sqrt(std::pow(center_x - x2, 2) + std::pow(center_y - y2, 2));
44 return radius;
45}

58void VelocityPlanning::acceleration_limiter(const std::vector<PathPoint> &points,
59 std::vector<double> &velocities) {
60 velocities[0] = config_.minimum_velocity_;
61 for (int i = 1; i < (int)points.size(); i++) {
62 // distance to previous point
63 double dx = points[i].position.x - points[i - 1].position.x;
64 double dy = points[i].position.y - points[i - 1].position.y;
65 double d = std::sqrt(dx * dx + dy * dy);
66
67 // v_i^2 = v_(i-1)^2 + 2 * a_acc * d
68 double max_velocity = std::sqrt(
69 std::max(0.0, velocities[i - 1] * velocities[i - 1] + 2.0 * config_.acceleration_ * d));
70
71 // cap the curvature-based velocity by longitudinal acceleration
72 velocities[i] = std::min(velocities[i], max_velocity);
73 }
74}

76void VelocityPlanning::braking_limiter(std::vector<PathPoint> &points,
77 std::vector<double> &velocities) {
78 for (int i = static_cast<int>(points.size()) - 2; i >= 0; i--) {
79 double distance = 0;
80 double max_speed = velocities[i];
81
82 // Calculate segment distance
83 int j = i + 1;
84 distance = std::sqrt(std::pow(points[j].position.x - points[j - 1].position.x, 2) +
85 std::pow(points[j].position.y - points[j - 1].position.y, 2));
86
87 // Correct kinematic speed calculation
88 // v_f² = v_i² + 2ad
89 double max_terminal_speed = std::sqrt(
90 std::max(0.0, std::pow(velocities[j], 2) - 2 * config_.braking_acceleration_ * distance));
91
92 max_speed = std::min(max_speed, max_terminal_speed);
93 max_speed = std::min(max_speed, this->config_.desired_velocity_);
94
95 velocities[i] = max_speed;
96 }
