54Eigen::MatrixXd NoWSSBicycleModel::jacobian_cg_velocity_to_wheels(
55 const Eigen::Vector3d& cg_velocities) {
56 Eigen::MatrixXd jacobian = Eigen::MatrixXd::Zero(2, 3);
57 double vx = cg_velocities(0);
58 double omega = cg_velocities(2);
59 double common_term = (std::fabs(vx) <= 1e-2)
60 ? 0
61 : 1 / (1 + pow((this->car_parameters_.wheelbase * omega) / vx, 2));
62
63 jacobian(0, 0) =
64 common_term *
65 (std::fabs(vx) <= 1e-2 ? 0 : -this->car_parameters_.wheelbase * omega / (vx * vx));
66 jacobian(0, 1) = 0;
67 jacobian(0, 2) = common_term * (std::fabs(vx) <= 1e-2 ? 0 : this->car_parameters_.wheelbase / vx);
68 jacobian(1, 0) =
69 60 * this->car_parameters_.gear_ratio / (M_PI * this->car_parameters_.wheel_diameter);
70 jacobian(1, 1) = 0;
71 jacobian(1, 2) = 0;
