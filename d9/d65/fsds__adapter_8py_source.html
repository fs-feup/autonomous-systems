140 def quaternion_to_rotation_matrix(quaternion: Quaternion):
141 """!
142 Converts quaternion to a rotation matrix.
143
144 Args:
145 quaternion (Quaternion): Quaternion representing rotation.
146
147 Returns:
148 numpy.ndarray: Rotation matrix.
149 """
150 q0, q1, q2, q3 = quaternion.x, quaternion.y, quaternion.z, quaternion.w
151
152 # First row of the rotation matrix
153 r00 = 2 * (q0 * q0 + q1 * q1) - 1
154 r01 = 2 * (q1 * q2 - q0 * q3)
155 r02 = 2 * (q1 * q3 + q0 * q2)
156
157 # Second row of the rotation matrix
158 r10 = 2 * (q1 * q2 + q0 * q3)
159 r11 = 2 * (q0 * q0 + q2 * q2) - 1
160 r12 = 2 * (q2 * q3 - q0 * q1)
161
162 # Third row of the rotation matrix
163 r20 = 2 * (q1 * q3 - q0 * q2)
164 r21 = 2 * (q2 * q3 + q0 * q1)
165 r22 = 2 * (q0 * q0 + q3 * q3) - 1
166
167 rot_matrix = np.array([[r00, r01, r02], [r10, r11, r12], [r20, r21, r22]])
168 return rot_matrix
