77 custom_interfaces::msg::Point2d position = custom_interfaces::msg::Point2d();

79 position.y = detection.pose.pose.position.y;
80 auto cone_message = custom_interfaces::msg::Cone();
81 cone_message.position = position;

83 cone_message.color = common_lib::competition_logic::get_color_string(
84 common_lib::competition_logic::Color::UNKNOWN);
85 cone_array_msg.cones.cone_array.push_back(cone_message);
86 }

88 cone_array_msg.exec_time = 0.01;
