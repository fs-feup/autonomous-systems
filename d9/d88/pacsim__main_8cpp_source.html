519void updateStateEstimation(const pacsim::msg::PerceptionDetections msg)
520{
521 std::lock_guard<std::mutex> lock(mutexSimTime); // Ensure thread safety
522
523 // Get vehicle position and orientation from the model
524 Eigen::Vector3d vehicle_position = model->getPosition();
525 Eigen::Vector3d vehicle_orientation = model->getOrientation();
526
527 double yaw = vehicle_orientation.z(); // Extract yaw (rotation around Z-axis)
528
529 // Create 2D rotation matrix for transforming perception detections
530 Eigen::Matrix2d rotation_matrix;
531 rotation_matrix << cos(yaw), -sin(yaw),
532 sin(yaw), cos(yaw);
533
534 // Set duplicate removal threshold (meters)
535 const double epsilon = 2.5; // Ignore detections closer than 30cm to existing cones
536
537 for (const auto& detection : msg.detections)
538 {
539 // Get detected point in vehicle frame
540 Eigen::Vector2d local_point(detection.pose.pose.position.x, detection.pose.pose.position.y);
541
542 // Transform to map frame
543 Eigen::Vector2d global_point = rotation_matrix * local_point;
544 global_point += vehicle_position.head<2>(); // Apply translation
545
546 // Check if a cone already exists nearby
547 bool is_duplicate = false;
548 for (const auto& existing_cone : detected_cones_map)
549 {
550 double dist = std::hypot(global_point.x() - existing_cone.first.first,
551 global_point.y() - existing_cone.first.second);
552 if (dist < epsilon)
553 {
554 is_duplicate = true;
555 break;
556 }
557 }
558
559 if (!is_duplicate)
560 {
561 // Round positions to avoid tiny variations (grid-based key system)
562 double rounded_x = std::round(global_point.x() * 100.0) / 100.0; // Round to 2 decimal places
563 double rounded_y = std::round(global_point.y() * 100.0) / 100.0;
564
565 std::pair<double, double> key = {rounded_x, rounded_y};
566 geometry_msgs::msg::Point point;
567 point.x = global_point.x();
568 point.y = global_point.y();
569 point.z = detection.pose.pose.position.z; // Keep Z as detected
570
571 detected_cones_map[key] = point;
572 }
573 }
574}

683void handleTf2StaticTransforms()
684{
685 if (mainConfig.broadcast_sensors_tf2)
686 {
687
688 for (auto sensor : perceptionSensors)
689 {
690 geometry_msgs::msg::TransformStamped t;
691
692 t.header.stamp = rclcpp::Time(0, 0);
693 t.header.frame_id = mainConfig.cog_frame_id_pipeline;
694 t.child_frame_id = sensor->getFrameId();
695
696 auto translation = sensor->getPosition();
697 auto orientation = sensor->getOrientation();
698 t.transform.translation.x = translation.x();
699 t.transform.translation.y = translation.y();
700 t.transform.translation.z = translation.z();
701 tf2::Quaternion q;
702 q.setRPY(orientation.x(), orientation.y(), orientation.z());
703 t.transform.rotation.x = q.x();
704 t.transform.rotation.y = q.y();
705 t.transform.rotation.z = q.z();
706 t.transform.rotation.w = q.w();
707
708 tf_static_broadcaster_->sendTransform(t);
709 }
