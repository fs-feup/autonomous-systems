24bool customComparator(const std::pair<double, double>& a, const std::pair<double, double>& b) {
25 if (a.first != b.first) {
26 return a.first < b.first;
27 }
28 return a.second < b.second;

37std::vector<std::pair<double, double>> orderVectorOfPairs
38(const std::vector<std::pair<double, double>>& vec) {
39 std::vector<std::pair<double, double>> result = vec;
40 std::sort(result.begin(), result.end(), customComparator);

51float round_n(float num, int decimal_places) {
52 num *= pow(10, decimal_places);
53 int intermediate = round(num);
54 num = intermediate/pow(10, decimal_places);
55 return num;
56}

66std::vector<PathPoint> processTriangulations(std::string filename, std::string testname) {
67 Track *track = new Track();
68 track->fillTrack(filename); // fill track with file data
69
70 LocalPathPlanner *pathplanner = new LocalPathPlanner();
71 std::vector<PathPoint> path;
72 std::ofstream measuresPath = openWriteFile("src/performance/exec_time/planning.csv");
73 double total_time = 0;
74 int no_iters = 100;
75
76 // No_iters repetitions to get average
77 for (int i = 0; i < no_iters; i++) {
78 auto t0 = std::chrono::high_resolution_clock::now();
79
80 std::vector<PathPoint *> pathPointers = pathplanner->processNewArray(track);
81
82 auto t1 = std::chrono::high_resolution_clock::now();
83
84 double elapsed_time_ms = std::chrono::duration<double, std::milli>(t1 - t0).count();
85
86 if (i == 0) {
87 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Delaunay Triangulations processed in %f ms.\n",
88 elapsed_time_ms);
89 measuresPath << "planning, delaunay, " << testname << ", " <<
90 elapsed_time_ms;
91 }
92
93 total_time += elapsed_time_ms;
94
95 for (size_t i = 0; i < pathPointers.size(); i++) path.push_back(*pathPointers[i]);
96 }
97
98 measuresPath << ", " << total_time / no_iters << "\n";
99 measuresPath.close();
100
101 RCLCPP_INFO(rclcpp::get_logger("rclcpp"),
102 "Average Delaunay Triangulations processed in %f ms.", (total_time / no_iters));
103
104 return path;
105}

114void outlierCalculations(std::string filename, std::string testname) {
115 Track *track = new Track();
116 track->fillTrack(filename); // fill track with file data
117
118 // Remove outliers no_iters times to get average
119 int no_iters = 100;
120 double total_time = 0;
121 std::ofstream measuresPath = openWriteFile("src/performance/exec_time/planning.csv");
122
123 for (int i = 0; i < no_iters; i++) {
124 track->reset();
125 track->fillTrack(filename);
126 auto t0 = std::chrono::high_resolution_clock::now();
127 track->validateCones();
128 auto t1 = std::chrono::high_resolution_clock::now();
129 double elapsed_time_iter_ms = std::chrono::duration<double, std::milli>(t1 - t0).count();
130 if (i == 0) {
131 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"),
132 "First Iteration: Outliers removed in %f ms\n", (elapsed_time_iter_ms));
133 measuresPath << "planning, outliers, " << testname << ", " << elapsed_time_iter_ms;
134 }
135 total_time += elapsed_time_iter_ms;
136 }
137
138 measuresPath << ", " << total_time / no_iters << "\n";
139 measuresPath.close();
140
141 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "Outliers removed in average %f ms.",
142 (total_time / no_iters));
143}
