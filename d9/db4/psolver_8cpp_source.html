35std::tuple<Position, int, double> PointSolver::update_closest_point(
36 const std::vector<custom_interfaces::msg::PathPoint> &pathpoint_array) const {
37 double min_distance = 1e9;
38 double closest_point_velocity = 0;
39 Position closest_point = Position();
40 Position aux_point = Position();
41 int closest_point_id = -1;
42 for (size_t i = 0; i < pathpoint_array.size(); i++) {
43 aux_point = Position(pathpoint_array[i].x, pathpoint_array[i].y);

49 closest_point_velocity = pathpoint_array[i].v;

55std::tuple<Position, double, bool> PointSolver::update_lookahead_point(
56 const std::vector<custom_interfaces::msg::PathPoint> &pathpoint_array,
57 int closest_point_id) const {
58 Position rear_axis_point = this->vehicle_pose_.rear_axis_;
59 double ld = this->k_ * std::max(this->vehicle_pose_.velocity_, 3.0);
60 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Current ld: %f", ld);
61
62 for (size_t i = 0; i < pathpoint_array.size(); i++) {
63 size_t index_a = (closest_point_id + i) % pathpoint_array.size();
64 size_t index_b = (closest_point_id + i + 1) % pathpoint_array.size();
65 auto point_a = Position(pathpoint_array[index_a].x, pathpoint_array[index_a].y);
66 auto point_b = Position(pathpoint_array[index_b].x, pathpoint_array[index_b].y);
67
68 if (!(rear_axis_point.euclidean_distance(point_a) < ld &&
69 rear_axis_point.euclidean_distance(point_b) > ld)) {
70 continue;
71 }
72
73 double result_x;
74 double result_y;
75
76 // Slope of the line is infinite, don't need line equation
77 if (point_a.x == point_b.x) {
78 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Vertical line!!");
79 result_x = point_a.x;
80 double delta = ld * ld - std::pow(result_x - rear_axis_point.x, 2);
81 if (delta < 0) {
82 continue;
83 }
84 double y1 = rear_axis_point.y + std::sqrt(delta);
85 double y2 = rear_axis_point.y - std::sqrt(delta);
86
87 if (y1 >= std::min(point_a.y, point_b.y) && y1 <= std::max(point_a.y, point_b.y)) {
88 result_y = y1;
89 } else {
90 result_y = y2;
91 }
92
93 } else {
94 // y = mx + c
95 double m = (point_b.y - point_a.y) / (point_b.x - point_a.x);
96 double c = point_a.y - m * point_a.x;
97
98 // (x - x0)^2 + (y - y0)^2 = r^2
99 // with x0 = rear_axis_point.x, y0 = rear_axis_point.y, r = ld
100
101 // with this information we can find the intersection point between the circle and the line
102 // and obtain the lookahead point.
103 // Expanded form of the circle equation with y substituted by mx + c gives a quadratic
104 // equation:
105 double A = 1 + m * m;
106 double B = 2 * (m * c - rear_axis_point.x - m * rear_axis_point.y);
107 double C = rear_axis_point.x * rear_axis_point.x + c * c +
108 rear_axis_point.y * rear_axis_point.y - 2 * c * rear_axis_point.y - ld * ld;
109
110 double delta = B * B - 4 * A * C;
111
112 if (delta < 0) {
113 continue;
114 }
115
116 double x1 = (-B + std::sqrt(delta)) / (2 * A);
117 double x2 = (-B - std::sqrt(delta)) / (2 * A);
118
119 double y1 = m * x1 + c;
120 double y2 = m * x2 + c;
121
122 // solution (x,y) which is within bounds of point_a and point_b
123 if (x1 >= std::min(point_a.x, point_b.x) && x1 <= std::max(point_a.x, point_b.x) &&
124 y1 >= std::min(point_a.y, point_b.y) && y1 <= std::max(point_a.y, point_b.y)) {
125 result_x = x1;
126 result_y = y1;
127 } else {
128 result_x = x2;
129 result_y = y2;
130 }
131 }
132
133 return std::make_tuple(Position(result_x, result_y),
134 (pathpoint_array[index_a].v + pathpoint_array[index_b].v) / 2.0, false);
135 }
136 RCLCPP_DEBUG(rclcpp::get_logger("control"), "No lookahead point found");
137 return std::make_tuple(Position(), 0.0, true);
138}
