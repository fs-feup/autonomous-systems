50void Himmelsbach::process_slice(
51 const sensor_msgs::msg::PointCloud2::SharedPtr& trimmed_point_cloud,
52 sensor_msgs::msg::PointCloud2::SharedPtr& ground_removed_point_cloud, size_t slice_idx,
53 GroundGrid& ground_grid) const {
54 // The first ground point is assumed to be directly below the car
55 double previous_ground_point_x = 0.0;
56 double previous_ground_point_y = 0.0;
57 double previous_ground_point_z = -trim_params_.lidar_height;
58
59 auto& cloud_data = trimmed_point_cloud->data;
60 auto& output_data = ground_removed_point_cloud->data;
61
62 std::vector<int> ground_indices;
63 std::vector<double> ground_distances;
64
65 // For each ring in the slice, compare points to previous ground point
66 for (int ring_idx = NUM_RINGS - 1; ring_idx >= 0; --ring_idx) {
67 const auto& ring = slices_->at(slice_idx).rings[ring_idx];
68 if (ring.indices.empty()) {
69 continue;
70 }
71
72 int lowest_ground_idx_in_ring = -1;
73 float previous_ground_point_distance =
74 std::hypot(previous_ground_point_x, previous_ground_point_y);
75
76 for (int idx : ring.indices) {
77 float pt_x = *reinterpret_cast<float*>(&cloud_data[LidarPoint::PointX(idx)]);
78 float pt_y = *reinterpret_cast<float*>(&cloud_data[LidarPoint::PointY(idx)]);
79 float pt_z = *reinterpret_cast<float*>(&cloud_data[LidarPoint::PointZ(idx)]);
80
81 float current_ground_point_distance = std::hypot(pt_x, pt_y);
82
83 if (current_ground_point_distance - previous_ground_point_distance <
84 (-5 - (0.15 * ring_idx)) ||
85 current_ground_point_distance - previous_ground_point_distance >
86 (5 + (0.15 * ring_idx))) {
87 // It is noise, ignore
88 continue;
89 } else if (current_ground_point_distance <= previous_ground_point_distance) {
90 // Write to the output cloud
91 size_t write_idx = ground_removed_point_cloud->width;
92 std::memcpy(&output_data[write_idx * LidarPoint::POINT_STEP],
93 &cloud_data[idx * LidarPoint::POINT_STEP], LidarPoint::POINT_STEP);
94 ground_removed_point_cloud->width++;
95 } else {
96 // Compute slope to previous ground point
97 float dz = pt_z - previous_ground_point_z;
98 float dr = current_ground_point_distance - previous_ground_point_distance;
99 if (dr < 0.001) {
100 dr = 0.001;
101 }
102
103 float slope = std::abs(dz / dr);
104
105 if (slope < this->max_slope_) {
106 // Point is a ground candidate
107 ground_indices.push_back(idx);
108 ground_distances.push_back(current_ground_point_distance);
109 } else {
110 // Point is non-ground, write to output cloud
111 size_t write_idx = ground_removed_point_cloud->width;
112 std::memcpy(&output_data[write_idx * LidarPoint::POINT_STEP],
113 &cloud_data[idx * LidarPoint::POINT_STEP], LidarPoint::POINT_STEP);
114 ground_removed_point_cloud->width++;
115 }
116 }
117 }
118
119 // Evaluate ground candidates: compare each to the median distance of the farthest 10%.
120 // Points much closer than this reference are obstacles, not ground.
121 if (!ground_indices.empty()) {
122 int num_points = static_cast<int>(ground_distances.size());
123 int top_k = std::max<int>(1, num_points / 10);
124
125 // Partition so that the top 10% farthest values are at the end
126 std::nth_element(ground_distances.begin(), ground_distances.end() - top_k,
127 ground_distances.end());
128
129 // Median
130 std::vector<double> top_farthest(ground_distances.end() - top_k, ground_distances.end());
131 std::sort(top_farthest.begin(), top_farthest.end());
132 double r_ref = top_farthest[top_farthest.size() / 2];
133
134 // Evaluate each ground candidate compared to the reference distance
135 for (int idx : ground_indices) {
136 float gx = *reinterpret_cast<const float*>(&cloud_data[LidarPoint::PointX(idx)]);
137 float gy = *reinterpret_cast<const float*>(&cloud_data[LidarPoint::PointY(idx)]);
138 float gz = *reinterpret_cast<const float*>(&cloud_data[LidarPoint::PointZ(idx)]);
139 double r = std::hypot(gx, gy);
140
141 double threshold = this->initial_alpha_;
142 if (r > this->start_augmentation_) {
143 threshold += this->alpha_augmentation_m_ * (r - this->start_augmentation_);
144 }
145
146 if (r < r_ref - threshold) {
147 // too far from reference, non-ground
148 size_t write_idx = ground_removed_point_cloud->width;
149 std::memcpy(&output_data[write_idx * LidarPoint::POINT_STEP],
150 &cloud_data[idx * LidarPoint::POINT_STEP], LidarPoint::POINT_STEP);
151 ground_removed_point_cloud->width++;
152 } else {
153 // Ground, update the ground grid and lowest ground point
154 ground_grid.set_ground_height(gx, gy, gz);
155 if (lowest_ground_idx_in_ring == -1 ||
156 gz < *reinterpret_cast<const float*>(
157 &cloud_data[LidarPoint::PointZ(lowest_ground_idx_in_ring)])) {
158 lowest_ground_idx_in_ring = idx;
159 }
160 }
161 }
162
163 if (lowest_ground_idx_in_ring != -1) {
164 previous_ground_point_x =
165 *reinterpret_cast<float*>(&cloud_data[LidarPoint::PointX(lowest_ground_idx_in_ring)]);
166 previous_ground_point_y =
167 *reinterpret_cast<float*>(&cloud_data[LidarPoint::PointY(lowest_ground_idx_in_ring)]);
168 previous_ground_point_z =
169 *reinterpret_cast<float*>(&cloud_data[LidarPoint::PointZ(lowest_ground_idx_in_ring)]);
170 }
171
172 ground_indices.clear();
173 ground_distances.clear();
174 }
175 }

178void Himmelsbach::split_point_cloud(
179 const sensor_msgs::msg::PointCloud2::SharedPtr& input_cloud) const {
180 // Clear previous indices
181 for (auto& slice : *slices_) {
182 for (auto& ring : slice.rings) {
183 ring.indices.clear();
184 }
185 }
186
187 const size_t num_points = input_cloud->width * input_cloud->height;
188 if (num_points == 0) {
189 return;
190 }
191
192 const auto& cloud = input_cloud->data;
193 double slice_angle_rad = grid_angle_ * (M_PI / 180.0);
194
195 // Calculate the starting azimuth based on FOV
196 double fov_rad = trim_params_.fov * (M_PI / 180.0);
197 double min_azimuth = -(fov_rad / 2.0);
198
199 for (size_t i = 0; i < num_points; ++i) {
200 float x = *reinterpret_cast<const float*>(&cloud[LidarPoint::PointX(i)]);
201 float y = *reinterpret_cast<const float*>(&cloud[LidarPoint::PointY(i)]);
202 uint16_t ring = *reinterpret_cast<const uint16_t*>(&cloud[LidarPoint::PointRing(i)]);
203
204 double azimuth = std::atan2(y, x);
205 double relative_azimuth = azimuth - min_azimuth;
206
207 if (relative_azimuth < 0 || relative_azimuth >= fov_rad) {
208 continue;
209 }
210
211 int slice_idx = static_cast<int>(relative_azimuth / slice_angle_rad);
212
213 if (slice_idx >= 0 && slice_idx < static_cast<int>(slices_->size())) {
214 slices_->at(slice_idx).rings[ring].indices.push_back(i);
215 }
216 }
