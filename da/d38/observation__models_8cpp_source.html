23Eigen::Vector2f ObservationModel::observation_model(const Eigen::VectorXf &expected_state,
24 const unsigned int landmark_index) const {
25 Eigen::Matrix3f transformation_matrix = Eigen::Matrix3f::Identity();
26 // transformation matrix is the matrix that transforms the landmark position from the world
27 // coordinate system to the robot coordinate system for future comparison with the observed
28 // landmark position
29
30 transformation_matrix(0, 0) = cos(-expected_state(2));
31 transformation_matrix(0, 1) = -sin(-expected_state(2));
32 transformation_matrix(0, 2) =
33 -expected_state(0) * cos(-expected_state(2)) + expected_state(1) * sin(-expected_state(2));
34 transformation_matrix(1, 0) = sin(-expected_state(2));
35 transformation_matrix(1, 1) = cos(-expected_state(2));
36 transformation_matrix(1, 2) =
37 -expected_state(0) * sin(-expected_state(2)) - expected_state(1) * cos(-expected_state(2));
38 Eigen::Vector3f observation =
39 transformation_matrix *
40 Eigen::Vector3f(expected_state(landmark_index), expected_state(landmark_index + 1), 1);
41
42 return Eigen::Vector2f(observation(0), observation(1));
43}
