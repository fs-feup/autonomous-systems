16 double min_distance = 1e9;
17 double closest_point_velocity = 0;
18 Position closest_point = Position();

20 int closest_point_id = -1;
21 for (size_t i = 1; i < pathpoint_array.size(); i++) {

35std::tuple<Position, double, bool> get_lookahead_point(
36 const std::vector<custom_interfaces::msg::PathPoint> &pathpoint_array,
37 int closest_point_id, double lookahead_distance, Position rear_axis_position, double last_to_first_max_dist) {
38 Position rear_axis_point = rear_axis_position;
39
40 for (size_t i = 0; i < pathpoint_array.size(); i++) {
41 size_t index_a = (closest_point_id + i) % pathpoint_array.size();
42 size_t index_b = (closest_point_id + i + 1) % pathpoint_array.size();
43
44 // We reached the end of the path
45 if (index_b < index_a) {
46 Position first_point(pathpoint_array.front().x, pathpoint_array.front().y);
47 Position last_point(pathpoint_array.back().x, pathpoint_array.back().y);
48 double start_end_distance = first_point.euclidean_distance(last_point);
49
50 // If the path is not a closed track, the lookahead point is the last point
51 // If the path is a closed track, we continue normally, the first point is the continuation
52 if (start_end_distance > last_to_first_max_dist) {
53 RCLCPP_DEBUG(rclcpp::get_logger("control"),
54 "Lookahead extends beyond path end and it is not a closed track, using last "
55 "point of the path as lookahead point");
56 return std::make_tuple(Position(pathpoint_array.back().x, pathpoint_array.back().y),
57 pathpoint_array.back().v, false);
58 }
59 }
60
61 auto point_a = Position(pathpoint_array[index_a].x, pathpoint_array[index_a].y);
62 auto point_b = Position(pathpoint_array[index_b].x, pathpoint_array[index_b].y);
63
64 if (!(rear_axis_point.euclidean_distance(point_a) < lookahead_distance &&
65 rear_axis_point.euclidean_distance(point_b) > lookahead_distance)) {
66 continue;
67 }
68
69 double result_x;
70 double result_y;
71
72 // Slope of the line is infinite, don't need line equation
73 if (point_a.x == point_b.x) {
74 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Vertical line!!");
75 result_x = point_a.x;
76 double delta = lookahead_distance * lookahead_distance - std::pow(result_x - rear_axis_point.x, 2);
77 if (delta < 0) {
78 continue;
79 }
80 double y1 = rear_axis_point.y + std::sqrt(delta);
81 double y2 = rear_axis_point.y - std::sqrt(delta);
82
83 if (y1 >= std::min(point_a.y, point_b.y) && y1 <= std::max(point_a.y, point_b.y)) {
84 result_y = y1;
85 } else {
86 result_y = y2;
87 }
88
89 } else {
90 // y = mx + c
91 double m = (point_b.y - point_a.y) / (point_b.x - point_a.x);
92 double c = point_a.y - m * point_a.x;
93
94 // (x - x0)^2 + (y - y0)^2 = r^2
95 // with x0 = rear_axis_point.x, y0 = rear_axis_point.y, r = lookahead_distance
96
97 // with this information we can find the intersection point between the circle and the line
98 // and obtain the lookahead point.
99 // Expanded form of the circle equation with y substituted by mx + c gives a quadratic
100 // equation:
101 double A = 1 + m * m;
102 double B = 2 * (m * c - rear_axis_point.x - m * rear_axis_point.y);
103 double C = rear_axis_point.x * rear_axis_point.x + c * c +
104 rear_axis_point.y * rear_axis_point.y - 2 * c * rear_axis_point.y - lookahead_distance * lookahead_distance;
105
106 double delta = B * B - 4 * A * C;
107
108 if (delta < 0) {
109 continue;
110 }
111
112 double x1 = (-B + std::sqrt(delta)) / (2 * A);
113 double x2 = (-B - std::sqrt(delta)) / (2 * A);
114
115 double y1 = m * x1 + c;
116 double y2 = m * x2 + c;
117
118 // solution (x,y) which is within bounds of point_a and point_b
119 if (x1 >= std::min(point_a.x, point_b.x) && x1 <= std::max(point_a.x, point_b.x) &&
120 y1 >= std::min(point_a.y, point_b.y) && y1 <= std::max(point_a.y, point_b.y)) {
121 result_x = x1;
122 result_y = y1;
123 } else {
124 result_x = x2;
125 result_y = y2;
126 }
127 }
128
129 return std::make_tuple(Position(result_x, result_y),
130 (pathpoint_array[index_a].v + pathpoint_array[index_b].v) / 2.0, false);
131 }
132 RCLCPP_DEBUG(rclcpp::get_logger("control"), "No lookahead point found");
133 return std::make_tuple(Position(), 0.0, true);
134}
