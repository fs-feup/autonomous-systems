81void GraphSLAMSolver::add_observations(const std::vector<common_lib::structures::Cone>& cones) {
82 if (cones.empty()) {
83 this->_landmark_filter_->filter(Eigen::VectorXd::Zero(0), Eigen::VectorXd::Zero(0));
84 this->_observations_global_ = Eigen::VectorXd::Zero(0);
85 this->_associations_ = Eigen::VectorXi::Zero(0);
86 return;
87 }
88 rclcpp::Time start_time, initialization_time, covariance_time, association_time,
89 factor_graph_time, optimization_time;
90 start_time = rclcpp::Clock().now();
91 Eigen::VectorXd observations(cones.size() * 2);
92 Eigen::VectorXd observations_confidences(cones.size());
93 Eigen::VectorXd observations_global(cones.size() * 2);
94 Eigen::VectorXd state;
95 Eigen::MatrixXd covariance;
96 for (unsigned int i = 0; i < cones.size(); i++) {
97 // Iterate through the cones
98 observations(i * 2) = cones[i].position.x;
99 observations(i * 2 + 1) = cones[i].position.y;
100 observations_confidences(i) = cones[i].certainty;
101 }
102
103 {
104 const std::shared_lock lock(this->_mutex_);
105 // Reduce error associated to accumulated pose difference by creating new factor
106 if (this->_graph_slam_instance_.new_pose_factors() && !_is_stopped_at_beginning_) {
107 this->_graph_slam_instance_.process_pose_difference(
108 Eigen::Vector3d::Zero(), this->_pose_updater_.get_last_pose(), true);
109 }
110 state = this->_graph_slam_instance_.get_state_vector();
111 observations_global =
112 common_lib::maths::local_to_global_coordinates(state.head<3>(), observations);
113 initialization_time = rclcpp::Clock().now();
114 covariance = this->_graph_slam_instance_.get_covariance_matrix();
115 covariance_time = rclcpp::Clock().now();
116 }
117
118 // Data association
119 Eigen::VectorXi associations(cones.size());
120 associations = this->_data_association_->associate(
121 state.segment(3, state.size() - 3), observations_global, covariance,
122 observations_confidences); // TODO: implement different mahalanobis distance
123 this->_associations_ = associations;
124 this->_observations_global_ = observations_global;
125 this->_map_coordinates_ = state.segment(3, state.size() - 3);
126 association_time = rclcpp::Clock().now();
127 Eigen::VectorXd unfiltered_new_observations;
128 Eigen::VectorXd unfiltered_new_observations_confidences;
129 for (int i = 0; i < associations.size(); i++) {
130 if (associations(i) == -1) {
131 unfiltered_new_observations.conservativeResize(unfiltered_new_observations.size() + 2);
132 unfiltered_new_observations(unfiltered_new_observations.size() - 2) =
133 observations_global(i * 2);
134 unfiltered_new_observations(unfiltered_new_observations.size() - 1) =
135 observations_global(i * 2 + 1);
136 unfiltered_new_observations_confidences.conservativeResize(
137 unfiltered_new_observations_confidences.size() + 1);
138 unfiltered_new_observations_confidences(unfiltered_new_observations_confidences.size() - 1) =
139 observations_confidences(i);
140 associations(i) = -2; // Mark as not ready to be added to the graph
141 }
142 }
143 Eigen::VectorXd filtered_new_observations = this->_landmark_filter_->filter(
144 unfiltered_new_observations, unfiltered_new_observations_confidences);
145 if (!this->_graph_slam_instance_.new_pose_factors() && !this->_is_stopped_at_beginning_) {
146 return;
147 }
148
149 if (this->_is_stopped_at_beginning_) {
150 for (int j = 0; j < associations.size(); j++) {
151 if (associations(j) >= 0) associations(j) = -2;
152 }
153 }
154
155 this->_landmark_filter_->delete_landmarks(filtered_new_observations);
156 if (this->_mission_ != common_lib::competition_logic::Mission::NONE &&
157 !(this->_params_.using_preloaded_map_ &&
158 (this->_mission_ == common_lib::competition_logic::Mission::SKIDPAD ||
159 this->_mission_ == common_lib::competition_logic::Mission::ACCELERATION ||
160 this->_mission_ == common_lib::competition_logic::Mission::AUTOCROSS ||
161 this->_mission_ == common_lib::competition_logic::Mission::TRACKDRIVE)) &&
162 lap_counter_ == 0) {
163 // Set the associations to -1 for the filtered observations
164 for (int i = 0; i < filtered_new_observations.size() / 2; i++) {
165 for (int j = 0; j < associations.size(); j++) {
166 if (std::hypot(filtered_new_observations(i * 2) - observations_global(j * 2),
167 filtered_new_observations(i * 2 + 1) - observations_global(j * 2 + 1)) <
168 this->_params_.data_association_gate_) {
169 associations(j) = -1;
170 break;
171 }
172 }
173 }
174 }
175
176 Eigen::Vector3d pose;
177 pose << state(0), state(1), state(2);
178
179 Eigen::VectorXi landmarks(state.size() - 3);
180 landmarks = state.segment(3, state.size() - 3).cast<int>();
181 LoopClosure::Result result = _loop_closure_->detect(pose, landmarks, associations, observations);
182 if (result.detected) {
183 lap_counter_++;
184 // Create a (hard-ish) lock on the landmarks' positions after the first lap.
185 if (lap_counter_ == 1) {
186 this->_graph_slam_instance_.lock_landmarks(0);
187 }
188 }
189
190 {
191 std::unique_lock uniq_lock(this->_mutex_);
192 ObservationData observation_data(std::make_shared<Eigen::VectorXd>(observations),
193 std::make_shared<Eigen::VectorXi>(associations),
194 std::make_shared<Eigen::VectorXd>(observations_global),
195 std::make_shared<Eigen::VectorXd>(observations_confidences),
196 cones.at(0).timestamp);
197 if (this->_optimization_under_way_) {
198 this->_observation_data_queue_.push(observation_data);
199 }
200 this->_graph_slam_instance_.process_observations(observation_data);
201 }
202 factor_graph_time = rclcpp::Clock().now();
203
204 // Optimize the graph
205 if (this->_params_.slam_optimization_mode_ == "sync" &&
206 !_is_stopped_at_beginning_) { // If optimization is synchronous
207 std::unique_lock uniq_lock(this->_mutex_);
208 this->_graph_slam_instance_.optimize();
209 optimization_time = rclcpp::Clock().now();
210 this->_pose_updater_.set_last_pose(gtsam_pose_to_eigen(this->_graph_slam_instance_.get_pose()));
211 }
212
213 // Timekeeping
214 if (this->_execution_times_ == nullptr) {
215 return;
216 }
217 this->_execution_times_->at(3) = (covariance_time - initialization_time).seconds() * 1000.0;
218 this->_execution_times_->at(2) = (association_time - covariance_time).seconds() * 1000.0;
219 this->_execution_times_->at(4) = (factor_graph_time - association_time).seconds() * 1000.0;
220 if (this->_params_.slam_optimization_mode_ == "sync") {
221 this->_execution_times_->at(5) = (optimization_time - factor_graph_time).seconds() * 1000.0;
222 }
223}

229void GraphSLAMSolver::_asynchronous_optimization_routine() {
230 rclcpp::Time start_time = rclcpp::Clock().now();
231 GraphSLAMInstance graph_slam_instance_copy;
232 PoseUpdater pose_updater_copy;
233 {
234 const std::shared_lock lock(this->_mutex_);
235 if (!this->_graph_slam_instance_.new_observation_factors()) {
236 return;
237 }
238 graph_slam_instance_copy = this->_graph_slam_instance_;
239 pose_updater_copy = this->_pose_updater_;
240 }
241 {
242 std::unique_lock lock(this->_mutex_);
243 this->_optimization_under_way_ = true;
244 }
245 rclcpp::Time initialization_time = rclcpp::Clock().now();
246 // Optimize the graph
247 graph_slam_instance_copy.optimize();
248 rclcpp::Time optimization_time = rclcpp::Clock().now();
249 {
250 std::unique_lock lock(this->_mutex_);
251 this->_optimization_under_way_ = false;
252 while (this->_motion_data_queue_.size() > 0 || this->_observation_data_queue_.size() > 0) {
253 bool process_pose = this->_observation_data_queue_.size() <= 0 ||
254 (this->_motion_data_queue_.size() > 0 &&
255 this->_motion_data_queue_.front().timestamp_ <
256 this->_observation_data_queue_.front().timestamp_);
257 if (process_pose) {
258 auto [pose_difference, new_pose] =
259 pose_updater_copy.update_pose(this->_motion_data_queue_.front(), this->_motion_model_);
260 graph_slam_instance_copy.process_pose_difference(pose_difference,
261 new_pose); // Process the pose difference
262 this->_motion_data_queue_.pop();
263 } else {
264 graph_slam_instance_copy.process_observations(this->_observation_data_queue_.front());
265 this->_observation_data_queue_.pop();
266 }
267 }
268 this->_graph_slam_instance_ = graph_slam_instance_copy;
269 this->_pose_updater_ = pose_updater_copy;
270 }
271 rclcpp::Time end_time = rclcpp::Clock().now();
272 if (this->_execution_times_ != nullptr) {
273 this->_execution_times_->at(5) =
274 (optimization_time - initialization_time).seconds() * 1000.0; // Optimization time
275 this->_execution_times_->at(6) =
276 (initialization_time - start_time).seconds() * 1000.0; // Initialization time
277 this->_execution_times_->at(7) =
278 (end_time - optimization_time).seconds() * 1000.0; // Graph update time
279 this->_execution_times_->at(8) =
280 (end_time - start_time).seconds() * 1000.0; // Optimization time
281 }
