126void EufsAdapter::pose_callback(const eufs_msgs::msg::CarState& msg) {
127 RCLCPP_DEBUG(this->get_logger(), "Received pose from EUFS");
128 custom_interfaces::msg::VehicleState pose;
129 // only gets the x, y, and theta since those are the only ones necessary for planning
130 pose.position.x = msg.pose.pose.position.x;
131 pose.position.y = msg.pose.pose.position.y;
132 pose.theta = atan2(2.0f * (msg.pose.pose.orientation.w * msg.pose.pose.orientation.z +
133 msg.pose.pose.orientation.x * msg.pose.pose.orientation.y),
134 msg.pose.pose.orientation.w * msg.pose.pose.orientation.w +
135 msg.pose.pose.orientation.x * msg.pose.pose.orientation.x -
136 msg.pose.pose.orientation.y * msg.pose.pose.orientation.y -
137 msg.pose.pose.orientation.z * msg.pose.pose.orientation.z);
138 this->vehicle_localization_callback(pose);
139}
