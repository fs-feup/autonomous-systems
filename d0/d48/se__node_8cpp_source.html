58void SENode::_perception_subscription_callback(const custom_interfaces::msg::ConeArray &msg) {
59 auto const &cone_array = msg.cone_array;
60 if (this->_perception_map_ == nullptr) {
61 RCLCPP_WARN(this->get_logger(), "SUB - Perception map is null");
62 return;
63 }
64 this->_perception_map_->clear();
65 RCLCPP_DEBUG(this->get_logger(), "SUB - cones from perception:");
66 RCLCPP_DEBUG(this->get_logger(), "--------------------------------------");
67
68 for (auto &cone : cone_array) {
69 RCLCPP_DEBUG(this->get_logger(), "(%f, %f)\t%s", cone.position.x, cone.position.y,
70 cone.color.c_str());
71
72 this->_perception_map_->push_back(common_lib::structures::Cone(cone.position.x, cone.position.y,
73 cone.color, cone.confidence));
74 }
75 RCLCPP_DEBUG(this->get_logger(), "--------------------------------------");
76
77 if (this->_ekf_ == nullptr) {
78 RCLCPP_WARN(this->get_logger(), "ATTR - EKF object is null");
79 return;
80 }
81 this->_ekf_->correction_step(*(this->_perception_map_));
82 this->_ekf_->update(this->_vehicle_state_, this->_track_map_);
83 RCLCPP_DEBUG(this->get_logger(), "EKF - EFK correction Step");
84 this->_publish_vehicle_state();
85 this->_publish_map();
86}

122void SENode::_wheel_speeds_subscription_callback(double rl_speed, double fl_speed, double rr_speed,
123 double fr_speed, double steering_angle,
124 const rclcpp::Time &timestamp) {
125 if (!this->_use_odometry_) {
126 return;
127 }
128 RCLCPP_DEBUG(this->get_logger(),
129 "SUB - Raw from wheel speeds: lb:%f - rb:%f - lf:%f - rf:%f - "
130 "steering: %f",
131 rl_speed, rr_speed, fl_speed, fr_speed, steering_angle);
132 auto [linear_velocity, angular_velocity] =
133 common_lib::vehicle_dynamics::odometry_to_velocities_transform(rl_speed, fl_speed, rr_speed,
134 fr_speed, steering_angle);
135 MotionUpdate motion_prediction_data;
136 motion_prediction_data.translational_velocity = linear_velocity;
137 motion_prediction_data.rotational_velocity = angular_velocity;
138 this->_motion_update_->translational_velocity = motion_prediction_data.translational_velocity;
139 this->_motion_update_->rotational_velocity = motion_prediction_data.rotational_velocity;
140 this->_motion_update_->steering_angle = steering_angle;
141 this->_motion_update_->last_update = timestamp;
142 RCLCPP_DEBUG(this->get_logger(), "SUB - translated from wheel speeds: v:%f - w:%f",
143 this->_motion_update_->translational_velocity,
144 this->_motion_update_->rotational_velocity);
145
146 if (this->_ekf_ == nullptr) {
147 RCLCPP_ERROR(this->get_logger(), "PUB - EKF object is null");
148 return;
149 }
150 MotionUpdate temp_update = *(this->_motion_update_);
151 this->_ekf_->prediction_step(temp_update);
152 this->_ekf_->update(this->_vehicle_state_, this->_track_map_);
153 RCLCPP_DEBUG(this->get_logger(), "EKF - EFK prediction Step");
154 this->_publish_vehicle_state();
155 this->_publish_map();
156}
