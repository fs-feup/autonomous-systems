83void SENode::_perception_subscription_callback(const custom_interfaces::msg::ConeArray &msg) {
84 auto const &cone_array = msg.cone_array;
85 if (this->_perception_map_ == nullptr) {
86 RCLCPP_WARN(this->get_logger(), "SUB - Perception map is null");
87 return;
88 }
89
90 rclcpp::Time start_time = this->get_clock()->now();
91
92 this->_perception_map_->clear();
93
94 for (auto &cone : cone_array) {
95 this->_perception_map_->push_back(common_lib::structures::Cone(cone.position.x, cone.position.y,
96 cone.color, cone.confidence));
97 }
98
99 if (this->_ekf_ == nullptr) {
100 RCLCPP_WARN(this->get_logger(), "ATTR - EKF object is null");
101 return;
102 }
103 // RCLCPP_DEBUG(this->get_logger(), "CORRECTION STEP END\n---------------------------\n");
104 this->_ekf_->correction_step(*(this->_perception_map_));
105 this->_ekf_->update(this->_vehicle_state_, this->_track_map_);
106
107 rclcpp::Time end_time = this->get_clock()->now();
108
109 // Execution Time calculation
110 std_msgs::msg::Float64 correction_execution_time;
111 correction_execution_time.data = (end_time - start_time).seconds() * 1000.0;
112 this->_correction_execution_time_publisher_->publish(correction_execution_time);
113 this->_publish_vehicle_state();
114 this->_publish_map();
115}

117void SENode::_imu_subscription_callback(const sensor_msgs::msg::Imu &imu_msg) {
118 if (this->_use_odometry_) {
119 return;
120 }
121
122 rclcpp::Time start_time = this->get_clock()->now();
123
124 double ax = imu_msg.linear_acceleration.x;
125 // double ay = imu_msg.linear_acceleration.y;
126
127 double v_rot = imu_msg.angular_velocity.z;
128
129 double angle = this->_ekf_->get_state()(2);
130
131 double ax_map = ax * cos(angle) /* - ay * sin(angle)*/;
132 double ay_map = ax * sin(angle) /* + ay * cos(angle) */;
133
134 MotionUpdate motion_prediction_data;
135 motion_prediction_data.acceleration_x = ax_map;
136 motion_prediction_data.acceleration_y = ay_map;
137 motion_prediction_data.rotational_velocity = v_rot;
138 this->_motion_update_->acceleration_x = motion_prediction_data.acceleration_x;
139 this->_motion_update_->acceleration_y = motion_prediction_data.acceleration_y;
140 this->_motion_update_->rotational_velocity = motion_prediction_data.rotational_velocity;
141 this->_motion_update_->last_update = imu_msg.header.stamp;
142 if (this->_ekf_ == nullptr) {
143 RCLCPP_ERROR(this->get_logger(), "ATTR - EKF object is null");
144 return;
145 }
146 MotionUpdate temp_update = *(this->_motion_update_);
147 this->_ekf_->prediction_step(temp_update, "imu");
148 this->_ekf_->update(this->_vehicle_state_, this->_track_map_);
149
150 // Execution Time calculation
151 rclcpp::Time end_time = this->get_clock()->now();
152 std_msgs::msg::Float64 prediction_execution_time;
153 prediction_execution_time.data = (end_time - start_time).seconds() * 1000.0;
154 this->_prediction_execution_time_publisher_->publish(prediction_execution_time);
155
156 this->_publish_vehicle_state_imu();
157 this->_publish_map();
158}

196void SENode::_publish_vehicle_state() {
197 if (this->_vehicle_state_ == nullptr) {
198 RCLCPP_WARN(this->get_logger(), "PUB - Vehicle state object is null");
199 return;
200 }
201 auto message = custom_interfaces::msg::VehicleState();
202 message.position.x = this->_vehicle_state_->pose.position.x;
203 message.position.y = this->_vehicle_state_->pose.position.y;
204 message.theta = this->_vehicle_state_->pose.orientation;
205 double velocity_x = this->_vehicle_state_->velocity_x;
206 double velocity_y = this->_vehicle_state_->velocity_y;
207 message.linear_velocity = std::sqrt(velocity_x * velocity_x + velocity_y * velocity_y);
208 message.angular_velocity = this->_vehicle_state_->rotational_velocity;
209 message.header.stamp = this->get_clock()->now();
210
211 RCLCPP_DEBUG(this->get_logger(), "PUB - Pose: (%f, %f, %f); Velocities: (%f, %f)",
212 message.position.x, message.position.y, message.theta, message.linear_velocity,
213 message.angular_velocity);
214 this->_vehicle_state_publisher_->publish(message);
215}
