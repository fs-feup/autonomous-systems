18AngleNorm ConeColoring::angle_and_norm(const Cone *possible_next_cone, TrackSide side) const {
19 const std::vector<Cone *> &current_cones = (bool)side ? current_left_cones : current_right_cones;
20 auto n = (int)current_cones.size();
21 double x_last = current_cones[n - 1]->getX() - current_cones[n - 2]->getX();
22 double y_last = current_cones[n - 1]->getY() - current_cones[n - 2]->getY();
23 double norm2 = sqrt(x_last * x_last + y_last * y_last);
24 double x_next = possible_next_cone->getX() - current_cones[n - 1]->getX();
25 double y_next = possible_next_cone->getY() - current_cones[n - 1]->getY();
26 double new_norm = sqrt(x_next * x_next + y_next * y_next);
27 double dot_product = x_last * x_next + y_last * y_next;
28 return {acos(dot_product / (new_norm * norm2)), new_norm};
29}

40 double y) const {
41 if ((bool)side)

51Cone ConeColoring::get_initial_cone(const std::vector<Cone *> &candidates, Pose initial_car_pose,
52 TrackSide side) const {
53 double minimum_cost = MAXFLOAT;
54 Cone initial_cone(1 - (int)side, 0, 0); // id must be 0 for left side and 1 for right side
55 for (const Cone *c : candidates) {
56 double distance_to_car = sqrt(pow(initial_car_pose.position.x - c->getX(), 2) +
57 pow(initial_car_pose.position.y - c->getY(), 2));
58 double cost = distance_to_car * distance_to_side(initial_car_pose, side, c->getX(), c->getY());
59 if (cost < minimum_cost) {
60 minimum_cost = cost;
61 initial_cone.setX(c->getX());
62 initial_cone.setY(c->getY());
63 }
64 }
65 return initial_cone;
66}

68std::vector<Cone *> ConeColoring::filter_cones_by_distance(const std::vector<Cone *> &input_cones,
69 Position position, double radius) const {
70 std::vector<Cone *> filtered_cones;
71 double radius_squared = radius * radius;
72 for (Cone *c : input_cones) {
73 double dx = position.x - c->getX();
74 double dy = position.y - c->getY();
75 double squared_distance = dx * dx + dy * dy;
76 if (squared_distance <= radius_squared) filtered_cones.push_back(c);
77 }
78 return filtered_cones;
79}

81void ConeColoring::place_initial_cones(const std::vector<Cone *> &input_cones,
82 Pose initial_car_pose) {
83 // filter cones that are farther away than 5 meters
84 std::vector<Cone *> candidates =
85 filter_cones_by_distance(input_cones, initial_car_pose.position, 5);
86
87 // get the initial cone for the left side
88 this->initial_cone_left =
89 std::make_shared<Cone>(get_initial_cone(candidates, initial_car_pose, TrackSide::LEFT));
90
91 // get the initial cone for the right side
92 this->initial_cone_right =
93 std::make_shared<Cone>(get_initial_cone(candidates, initial_car_pose, TrackSide::RIGHT));
94
95 // Create virtual cones to be added at the beginning of the vector. They must be dismissed later.
96 virtual_left_cone = std::make_shared<Cone>(
97 -2, initial_cone_left->getX() - 2 * (float)cos(initial_car_pose.orientation),
98 initial_cone_left->getY() - 2 * (float)sin(initial_car_pose.orientation));
99 virtual_right_cone = std::make_shared<Cone>(
100 -1, initial_cone_right->getX() - 2 * (float)cos(initial_car_pose.orientation),
101 initial_cone_right->getY() - 2 * (float)sin(initial_car_pose.orientation));
102
103 current_left_cones.push_back(virtual_left_cone.get());
104 current_right_cones.push_back(virtual_right_cone.get());
105 current_left_cones.push_back(initial_cone_left.get());
106 current_right_cones.push_back(initial_cone_right.get());
107}

109double ConeColoring::cost(const Cone *possible_next_cone, int n, TrackSide side) const {
110 const std::vector<Cone *> &current_cones = (bool)side ? current_left_cones : current_right_cones;
111 if (current_cones.size() < 2) {
112 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"),
113 "Not enough cones to calculate cost when coloring cones");
114 throw std::runtime_error("Not enough cones to calculate cost in Cone Coloring");
115 }
116 AngleNorm angle_norm;
117 angle_norm = angle_and_norm(possible_next_cone, side);
118 double cost = pow(distance_gain * angle_norm.norm, exponent1) +
119 pow(angle_gain * angle_norm.angle, exponent2) +
120 ncones_gain * (double)current_cones.size() / (float)n;
121 return cost;
122}

135bool ConeColoring::place_next_cone(
136 std::shared_ptr<std::vector<std::pair<Cone *, bool>>> visited_cones, double distance_threshold,
137 int n, TrackSide side) {
138 // std::cout << " Recieved cones are : " << std::endl;
139 // for (auto p : *visited_cones) {
140 // std::cout << p.second << " (" << p.first->getX() << ',' << p.first->getY() << ")" <<
141 // std::endl;
142 // }
143 double minimum_cost = MAXFLOAT;
144
145 std::vector<Cone *> &current_cones = (bool)side ? current_left_cones : current_right_cones;
146
147 const Cone *last_cone = current_cones.back();
148
149 const Cone *next_cone = (*visited_cones)[0].first;
150
151 std::pair<Cone *, bool> *best_pair = &(*visited_cones)[0];
152
153 double squared_distance_threshold = distance_threshold * distance_threshold;
154
155 for (std::pair<Cone *, bool> &p : *visited_cones) {
156 if (p.second) {
157 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Skipping visited cone at : %f , %f",
158 p.first->getX(), p.first->getY());
159 continue;
160 } // skip visited cones
161 double cost = ConeColoring::cost(p.first, n, side);
162 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Cost is %f for cone at (%f,%f)", cost,
163 p.first->getX(), p.first->getY());
164 if (cost < minimum_cost &&
165 last_cone->squared_distance_to(p.first) < squared_distance_threshold && cost < max_cost) {
166 minimum_cost = cost;
167 next_cone = p.first;
168 best_pair = &p;
169 }
170 }
171 if (minimum_cost == MAXFLOAT) return false;
172 auto selected_cone =
173 new Cone(2 * (int)current_cones.size() - 1 - int(side), next_cone->getX(), next_cone->getY());
174 current_cones.push_back(selected_cone);
175 best_pair->second = true;
176 return true;
177}
