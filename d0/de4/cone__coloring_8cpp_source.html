22AngleNorm ConeColoring::angle_and_norm(const Cone *possible_next_cone, TrackSide side) const {
23 const std::vector<Cone *> &current_cones = (bool)side ? current_left_cones : current_right_cones;
24 auto n = (int)current_cones.size();
25 double x_last = current_cones[n - 1]->getX() - current_cones[n - 2]->getX();
26 double y_last = current_cones[n - 1]->getY() - current_cones[n - 2]->getY();
27 double norm2 = sqrt(x_last * x_last + y_last * y_last);
28 double x_next = possible_next_cone->getX() - current_cones[n - 1]->getX();
29 double y_next = possible_next_cone->getY() - current_cones[n - 1]->getY();
30 double new_norm = sqrt(x_next * x_next + y_next * y_next);
31 double dot_product = x_last * x_next + y_last * y_next;
32 return {acos(dot_product / (new_norm * norm2)), new_norm};
33}

43double ConeColoring::distance_to_side(Pose initial_car_pose, TrackSide side, double x,
44 double y) const {
45 if ((bool)side)
46 return fabs(y - (initial_car_pose.position.y + 2 * cos(initial_car_pose.orientation)) -
47 tan(initial_car_pose.orientation) *

49 else

55Cone ConeColoring::get_initial_cone(const std::vector<Cone *> &candidates, Pose initial_car_pose,
56 TrackSide side) const {
57 double minimum_cost = MAXFLOAT;
58 Cone initial_cone(1 - (int)side, 0, 0); // id must be 0 for left side and 1 for right side
59 for (const Cone *c : candidates) {
60 double distance_to_car = sqrt(pow(initial_car_pose.position.x - c->getX(), 2) +
61 pow(initial_car_pose.position.y - c->getY(), 2));
62 double cost = distance_to_car * distance_to_side(initial_car_pose, side, c->getX(), c->getY());
63 if (cost < minimum_cost) {
64 minimum_cost = cost;
65 initial_cone.setX(c->getX());
66 initial_cone.setY(c->getY());
67 }
68 }
69 return initial_cone;
70}

72std::vector<Cone *> ConeColoring::filter_cones_by_distance(const std::vector<Cone *> &input_cones,
73 Position position, double radius) const {
74 std::vector<Cone *> filtered_cones;
75 double radius_squared = radius * radius;
76 for (Cone *c : input_cones) {
77 double dx = position.x - c->getX();
78 double dy = position.y - c->getY();
79 double squared_distance = dx * dx + dy * dy;
80 if (squared_distance <= radius_squared) filtered_cones.push_back(c);
81 }
82 return filtered_cones;
83}

85void ConeColoring::place_initial_cones(const std::vector<Cone *> &input_cones,
86 Pose initial_car_pose) {
87 // filter cones that are farther away than 5 meters
88 std::vector<Cone *> candidates =
89 filter_cones_by_distance(input_cones, initial_car_pose.position, 5);
90
91 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Car position is : (%f,%f)",
92 initial_car_pose.position.x, initial_car_pose.position.y);
93
94 // get the initial cone for the left side
95 this->initial_cone_left =
96 std::make_shared<Cone>(get_initial_cone(candidates, initial_car_pose, TrackSide::LEFT));
97
98 // get the initial cone for the right side
99 this->initial_cone_right =
100 std::make_shared<Cone>(get_initial_cone(candidates, initial_car_pose, TrackSide::RIGHT));
101
102 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Initial cones are at (%f, %f) and (%f, %f)",
103 initial_cone_left->getX(), initial_cone_left->getY(), initial_cone_right->getX(),
104 initial_cone_right->getY());
105
106 // Create virtual cones to be added at the beginning of the vector. They must be dismissed later.
107 virtual_left_cone = std::make_shared<Cone>(
108 -2, initial_cone_left->getX() - 2 * (float)cos(initial_car_pose.orientation),
109 initial_cone_left->getY() - 2 * (float)sin(initial_car_pose.orientation));
110 virtual_right_cone = std::make_shared<Cone>(
111 -1, initial_cone_right->getX() - 2 * (float)cos(initial_car_pose.orientation),
112 initial_cone_right->getY() - 2 * (float)sin(initial_car_pose.orientation));
113
114 current_left_cones.push_back(virtual_left_cone.get());
115 current_right_cones.push_back(virtual_right_cone.get());
116 current_left_cones.push_back(initial_cone_left.get());
117 current_right_cones.push_back(initial_cone_right.get());
118}

120double ConeColoring::cost(const Cone *possible_next_cone, int n, TrackSide side) const {
121 const std::vector<Cone *> &current_cones = (bool)side ? current_left_cones : current_right_cones;
122 if (current_cones.size() < 2) {
123 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"),
124 "Not enough cones to calculate cost when coloring cones");
125 throw std::runtime_error("Not enough cones to calculate cost in Cone Coloring");
126 }
127 AngleNorm angle_norm;
128 angle_norm = angle_and_norm(possible_next_cone, side);
129 double cost = pow(distance_gain * angle_norm.norm, exponent1) +
130 pow(angle_gain * angle_norm.angle, exponent2) +
131 ncones_gain * (double)current_cones.size() / (float)n;
132 return cost;
133}

146bool ConeColoring::place_next_cone(
147 std::shared_ptr<std::vector<std::pair<Cone *, bool>>> visited_cones, double distance_threshold,
148 int n, TrackSide side) {
149 double minimum_cost = MAXFLOAT;
150
151 std::vector<Cone *> &current_cones = (bool)side ? current_left_cones : current_right_cones;
152
153 const Cone *last_cone = current_cones.back();
154
155 const Cone *next_cone = (*visited_cones)[0].first;
156
157 std::pair<Cone *, bool> *best_pair = &(*visited_cones)[0];
158
159 double squared_distance_threshold = distance_threshold * distance_threshold;
160
161 for (std::pair<Cone *, bool> &p : *visited_cones) {
162 if (p.second) {
163 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Skipping visited cone at : %f , %f",
164 p.first->getX(), p.first->getY());
165 continue;
166 } // skip visited cones
167 double cost = ConeColoring::cost(p.first, n, side);
168 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Cost is %f for cone at (%f,%f)", cost,
169 p.first->getX(), p.first->getY());
170 if (cost < minimum_cost &&
171 last_cone->squared_distance_to(p.first) < squared_distance_threshold && cost < max_cost) {
172 minimum_cost = cost;
173 next_cone = p.first;
174 best_pair = &p;
175 }
176 }
177 if (minimum_cost == MAXFLOAT) return false;
178 auto selected_cone =
179 new Cone(2 * (int)current_cones.size() - 1 - int(side), next_cone->getX(), next_cone->getY());
180 current_cones.push_back(selected_cone);
181 best_pair->second = true;
182 return true;
183}
