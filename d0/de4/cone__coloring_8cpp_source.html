3void ConeColoring::remove_duplicates(std::vector<Cone>& cones) const {
4 std::vector<Cone> clustered_cones;
5 clustered_cones.reserve(cones.size());
6
7 for (const auto& cone : cones) {
8 bool is_duplicate = false;
9
10 // Check against each cone in the clustered_cones list
11 for (const auto& clustered_cone : clustered_cones) {
12 if (cone.position.euclidean_distance(clustered_cone.position) <
13 this->config_.same_cone_distance_threshold_) {
14 is_duplicate = true;
15 break;
16 }
17 }
18
19 // If not a duplicate, add to the clustered list
20 if (!is_duplicate) {
21 clustered_cones.push_back(cone);
22 }
23 }
24

26 cones = clustered_cones;

158void ConeColoring::place_second_cones(std::unordered_set<Cone, std::hash<Cone>>& uncolored_cones,
159 const Pose& car_pose, int& n_colored_cones) {
160 Cone first_left_cone = colored_blue_cones_.front();
161 Cone first_right_cone = colored_yellow_cones_.front();
162
163 double car_orientation = car_pose.orientation;
164 double min_distance = std::numeric_limits<double>::max();
165
166 Cone second_left_cone;
167 for (const auto& cone : uncolored_cones) {
168 double segment_orientation = atan2(cone.position.y - first_left_cone.position.y,
169 cone.position.x - first_left_cone.position.x);
170 segment_orientation = std::fmod(segment_orientation + 2.0 * M_PI, 2.0 * M_PI);
171 double orientation_difference = std::fabs(car_orientation - segment_orientation);
172 if (orientation_difference > 2.0 * M_PI / 5.0 && orientation_difference < 8.0 * M_PI / 5.0) {
173 continue;
174 }
175 double distance = cone.position.euclidean_distance(first_left_cone.position);
176 if (distance < min_distance) {
177 min_distance = distance;
178 second_left_cone = cone;
179 }
180 }
181 if (min_distance < std::numeric_limits<double>::max()) {
182 colored_blue_cones_.push_back(second_left_cone);
183 uncolored_cones.erase(second_left_cone);
184 n_colored_cones++;
185 } else {
186 RCLCPP_ERROR(rclcpp::get_logger("ConeColoring"), "Could not find second left cone");
187 }
188
189 min_distance = std::numeric_limits<double>::max();
190 Cone second_right_cone;
191 for (const auto& cone : uncolored_cones) {
192 double segment_orientation = atan2(cone.position.y - first_right_cone.position.y,
193 cone.position.x - first_right_cone.position.x);
194 segment_orientation = std::fmod(segment_orientation + 2.0 * M_PI, 2.0 * M_PI);

197 continue;
198 }

201 min_distance = distance;

214double ConeColoring::calculate_cost(const Cone& next_cone, const Cone& last_cone,
215 const TwoDVector& previous_to_last_vector,
216 const double& colored_to_input_cones_ratio) const {
217 AngleAndNorms angle_and_norms = common_lib::maths::angle_and_norms(
218 previous_to_last_vector, Position{next_cone.position.x - last_cone.position.x,
219 next_cone.position.y - last_cone.position.y});
220 double distance = angle_and_norms.norm2_;
221 double angle = angle_and_norms.angle_;
222 double cost = this->config_.distance_weight_ * pow(distance, this->config_.distance_exponent_) +
223 this->config_.angle_weight_ * pow(angle, this->config_.angle_exponent_) +
224 this->config_.ncones_weight_ * colored_to_input_cones_ratio;
225 return cost;
226}

246 this->colored_yellow_cones_[j].position);
247 if (distance < this->config_.same_cone_distance_threshold_) {
248 Cone last_cone = this->colored_blue_cones_[i - 1];
249 Cone previous_cone = this->colored_blue_cones_[i - 2];
250 TwoDVector last_vector = {last_cone.position.x - previous_cone.position.x,
251 last_cone.position.y - previous_cone.position.y};
252 double blue_cost = calculate_cost(this->colored_blue_cones_[i], last_cone, last_vector, 0);
253 last_cone = this->colored_yellow_cones_[j - 1];
254 previous_cone = this->colored_yellow_cones_[j - 2];
255 last_vector = {last_cone.position.x - previous_cone.position.x,
256 last_cone.position.y - previous_cone.position.y};
257 double yellow_cost =
258 calculate_cost(this->colored_yellow_cones_[j], last_cone, last_vector, 0);
259
260 if (blue_cost > yellow_cost) {
261 this->colored_blue_cones_.erase(this->colored_blue_cones_.begin() + i);
262 break; // Exit inner loop as this blue cone has been removed
263 } else {
