75
76 double car_orientation = car_pose.orientation;

81 double segment_orientation = atan2(cone.position.y - first_left_cone.position.y,

100 }

127double ConeColoring::calculate_cost(const Cone& next_cone, const Cone& last_cone,
128 const TwoDVector& previous_to_last_vector,
129 const double& colored_to_input_cones_ratio) const {
130 AngleAndNorms angle_and_norms = common_lib::maths::angle_and_norms(
131 previous_to_last_vector, Position{next_cone.position.x - last_cone.position.x,
132 next_cone.position.y - last_cone.position.y});
133 double distance = angle_and_norms.norm2_;
134 double angle = angle_and_norms.angle_;
135 double cost = this->config_.distance_weight_ * pow(distance, this->config_.distance_exponent_) +
136 this->config_.angle_weight_ * pow(angle, this->config_.angle_exponent_) +
137 this->config_.ncones_weight_ * colored_to_input_cones_ratio;
138 return cost;
139}
