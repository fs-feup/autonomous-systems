16Position ConeColoring::expected_initial_cone_position(const Pose& car_pose,
17 const TrackSide& track_side) const {
18 constexpr double distance_to_car = 2.0;
19 if (track_side == TrackSide::LEFT) {
20 return Position{car_pose.position.x - distance_to_car * sin(car_pose.orientation),
21 car_pose.position.y + distance_to_car * cos(car_pose.orientation)};

32 const Pose& car_pose, const TrackSide track_side) const {
33 Position expected_cone_position = expected_initial_cone_position(car_pose, track_side);
34 Cone initial_cone = *std::min_element(
35 cones.begin(), cones.end(), [&expected_cone_position](const Cone& cone1, const Cone& cone2) {
36 return cone1.position.euclidean_distance(expected_cone_position) <
37 cone2.position.euclidean_distance(expected_cone_position);

71double ConeColoring::calculate_cost(const Cone& next_cone, const Cone& last_cone,
72 const TwoDVector& previous_to_last_vector,
73 const double& colored_to_input_cones_ratio) const {
74 AngleAndNorms angle_and_norms = common_lib::maths::angle_and_norms(
75 previous_to_last_vector, Position{next_cone.position.x - last_cone.position.x,
76 next_cone.position.y - last_cone.position.y});
77 double distance = angle_and_norms.norm2_;
78 double angle = angle_and_norms.angle_;
79 double cost = this->config_.distance_weight_ * pow(distance, this->config_.distance_exponent_) +
80 this->config_.angle_weight_ * pow(angle, this->config_.angle_exponent_) +
81 this->config_.ncones_weight_ * colored_to_input_cones_ratio;
82 return cost;
83}

85bool ConeColoring::try_to_color_next_cone(
86 std::unordered_set<Cone, std::hash<Cone>>& uncolored_cones, std::vector<Cone>& colored_cones,
87 int& n_colored_cones, const int n_input_cones) const {
88 double min_cost = std::numeric_limits<double>::max();
89 Cone cheapest_cone;
90 const Cone last_cone = colored_cones.back();
91 const Cone second_last_cone = colored_cones[colored_cones.size() - 2];
92 const TwoDVector last_vector = {last_cone.position.x - second_last_cone.position.x,
93 last_cone.position.y - second_last_cone.position.y};
94 for (const auto& cone : uncolored_cones) {
95 double cost =
96 calculate_cost(cone, last_cone, last_vector,
97 static_cast<double>(n_colored_cones) / static_cast<double>(n_input_cones));
98 if (cost < min_cost && cone.position.euclidean_distance(last_cone.position) < 5 &&
99 cost < this->config_.max_cost_) {
100 min_cost = cost;
101 cheapest_cone = cone;
102 }
103 }
104 if (min_cost < std::numeric_limits<double>::max()) {
105 uncolored_cones.erase(cheapest_cone);
106 colored_cones.push_back(cheapest_cone);
107 n_colored_cones++;
108 return true;
109 } else {
110 return false;
111 }
