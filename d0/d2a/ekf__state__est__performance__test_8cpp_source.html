175 void run_execution(const std::vector<common_lib::structures::Cone>& cone_map) {
176 prediction_step_duration = std::chrono::microseconds::zero();
177 correction_step_duration = std::chrono::microseconds::zero();
178
179 for (int i = 0; i < 10; i++) {
180 auto ekf_for_predict = *ekf_test;
181 auto ekf_for_correction = *ekf_test;
182
183 start_time = std::chrono::high_resolution_clock::now();
184 ekf_for_predict.prediction_step(*motion_update_test);
185 prediction_step_time = std::chrono::high_resolution_clock::now();
186 ekf_for_correction.correction_step(cone_map);
187 end_time = std::chrono::high_resolution_clock::now();
188
189 prediction_step_duration +=
190 std::chrono::duration_cast<std::chrono::microseconds>(prediction_step_time - start_time);
191 correction_step_duration +=
192 std::chrono::duration_cast<std::chrono::microseconds>(end_time - prediction_step_time);
193 }
194
195 prediction_step_duration = prediction_step_duration / 10;
196 correction_step_duration = correction_step_duration / 10;
