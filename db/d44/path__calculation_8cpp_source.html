77vector<PathPoint> PathCalculation::calculate_trackdrive(const vector<Cone>& cone_array) {
78 vector<PathPoint> result = calculate_path(cone_array);
79
80 // Check if we have enough points to form a loop
81 if (result.size() < 3) {
82 RCLCPP_WARN(rclcpp::get_logger("planning"), "Not enough points to create trackdrive loop");
83 return result;
84 }
85
86 // Find the best point to close the loop
87 int best_cutoff_index = find_best_loop_closure(result);
88
89 // Trim the path to the best cutoff point
90 (void)result.erase(result.begin() + best_cutoff_index + 1, result.end());
91
92 // Add interpolated points between the last point and the first point
93 if (!result.empty()) {
94 const PathPoint& last_point = result.back();
95 const PathPoint& first_point = result.front();
96
97 vector<PathPoint> interpolated = add_interpolated_points(last_point, first_point, 4);
98 (void)result.insert(result.end(), interpolated.begin(), interpolated.end());
99 }
100
101 // Close the loop by adding the first point again
102 result.push_back(result[0]);
103
104 // Add overlap points (10 points or as many as available)
105 int overlap_count = min(10, static_cast<int>(result.size()) - 1);
106 for (int i = 1; i <= overlap_count; ++i) {
107 result.push_back(result[i]);
108 }
109
110 return result;

166 Point last_added_point;
167 bool first_point_added = false;

210void PathCalculation::extend_path(int max_points) {
211 int n_points = 0;
212 // Define cost threshold for discarding poor path options
213 const double worst_cost = config_.max_cost_ * config_.search_depth_;
214
215 while (true) {
216 if (current_path_.size() < 2) {
217 break;
218 }
219
220 const Point& prev = current_path_[current_path_.size() - 2];
221 const Point& last = current_path_.back();
222
223 shared_ptr<Midpoint> prev_mp = find_nearest_midpoint(prev);
224 shared_ptr<Midpoint> last_mp = find_nearest_midpoint(last);
225
226 // Abort if midpoints can't be matched
227 if (!prev_mp || !last_mp) {
228 RCLCPP_WARN(rclcpp::get_logger("planning"),
229 "No valid midpoints found for path extension.");
230 break;
231 }
232
233 // Search for the best continuation from the current midpoint
234 auto [best_cost, best_point] = find_best_next_midpoint(
235 config_.search_depth_, prev_mp, last_mp, config_.max_cost_);
236
237 // Stop if no good extension is found or point was already visited
238 if (best_cost > worst_cost || !best_point || visited_midpoints_.count(best_point) > 0) {
239 break;
240 }
241
242 current_path_.push_back(best_point->point);
243 (void)visited_midpoints_.insert(best_point);
244 n_points++;
245
246 if (n_points >= max_points) {
247 break;
248 }
249
250 // Update midpoints validity and discard cones if needed
251 if (current_path_.size() > 2) {
252 discard_cones_along_path();
253 }
254 }

259pair<shared_ptr<Midpoint>, shared_ptr<Midpoint>> PathCalculation::select_starting_midpoints() {
260 pair<shared_ptr<Midpoint>, shared_ptr<Midpoint>> result{nullptr, nullptr};
261
262 Midpoint anchor_pose_midpoint{
263 Point(initial_pose_.position.x, initial_pose_.position.y), nullptr, nullptr};
264
265 // Get candidate midpoints
266 anchor_pose_midpoint.close_points = select_candidate_midpoints(anchor_pose_midpoint, 8);
267
268 double best_cost = numeric_limits<double>::max();
269
270 for (const auto& first : anchor_pose_midpoint.close_points) {
271 shared_ptr<Midpoint> anchor_mp = make_shared<Midpoint>(anchor_pose_midpoint);
272
273 auto [cost, midpoint] = find_best_next_midpoint(
274 config_.search_depth_, anchor_mp, first, numeric_limits<double>::max());
275
276 double dx = first->point.x() - anchor_pose_midpoint.point.x();
277 double dy = first->point.y() - anchor_pose_midpoint.point.y();
278 double distance = std::sqrt(dx * dx + dy * dy);
279
280 //Calculate the deviation angle relative to the cars heading
281 double angle_to_point = std::atan2(dy, dx);
282 double angle_deviation = std::abs(std::atan2(
283 std::sin(angle_to_point - initial_pose_.orientation),
284 std::cos(angle_to_point - initial_pose_.orientation)
285 ));
286
287 double initial_cost =
288 std::pow(distance, config_.distance_exponent_) * config_.distance_gain_ +
289 std::pow(angle_deviation, config_.angle_exponent_) * config_.angle_gain_;
290
291
292 cost += initial_cost;
293
294 if (cost < best_cost) {
295 result.first = first;
296 result.second = midpoint;
297 best_cost = cost;
298 }
299 }
300
301 return result;
302}

304vector<shared_ptr<Midpoint>> PathCalculation::select_candidate_midpoints(
305 const Midpoint& anchor_pose, int num_candidates) const {
306 auto cmp = [](const pair<double, shared_ptr<Midpoint>>& cost1,
307 const pair<double, shared_ptr<Midpoint>>& cost2) {
308 return cost1.first > cost2.first;
309 };
310
311 priority_queue<pair<double, shared_ptr<Midpoint>>,
312 vector<pair<double, shared_ptr<Midpoint>>>, decltype(cmp)>
313 pq(cmp);
314
315 double car_direction_x = std::cos(initial_pose_.orientation);
316 double car_direction_y = std::sin(initial_pose_.orientation);
317
318 // Find midpoints that are in front of the car
319 for (const auto& mp : midpoints_) {
320 double dx = mp->point.x() - anchor_pose.point.x();
321 double dy = mp->point.y() - anchor_pose.point.y();
322
323 // Skip points behind the car
324 if ((dx * car_direction_x + dy * car_direction_y) <= 0.0) {
325 continue;
326 }
327
328 double dist = std::sqrt(dx * dx + dy * dy);
329
330 //Calculate the deviation angle relative to the cars heading
331 double angle_to_point = std::atan2(dy, dx);
332 double angle_deviation = std::abs(std::atan2(
333 std::sin(angle_to_point - initial_pose_.orientation),
334 std::cos(angle_to_point - initial_pose_.orientation)
335 ));
336
337 double cost = std::pow(angle_deviation, config_.angle_exponent_) * config_.angle_gain_ +
338 std::pow(dist, config_.distance_exponent_) * config_.distance_gain_;
339
340 pq.push({cost, mp});
341 }
342
343 // Get the closest midpoints in front of the car
344 vector<shared_ptr<Midpoint>> candidate_points;
345 int count = 0;
346 while (count < num_candidates && !pq.empty()) {
347 candidate_points.push_back(pq.top().second);
348 pq.pop();
349 count++;
350 }
351
352 return candidate_points;
353}

355pair<double, shared_ptr<Midpoint>> PathCalculation::find_best_next_midpoint(
356 int depth, const shared_ptr<Midpoint>& previous, const shared_ptr<Midpoint>& current,
357 double max_cost) const {
358 if (depth == 0) {
359 return {0.0, current};
360 }
361
362 double min_cost = config_.max_cost_ * config_.search_depth_;
363 shared_ptr<Midpoint> min_point = current;
364
365 for (const auto& next : current->close_points) {
366 // Avoid revisiting the previous point
367 if (next == previous) {
368 continue;
369 }
370
371 double local_cost = calculate_midpoint_cost(previous, current, next);
372
373 // Skip if local cost exceeds maximum allowed cost
374 if (local_cost > max_cost) {
375 continue;
376 }
377
378 // Recursive cost calculation
379 auto [best_cost, best_point] =
380 find_best_next_midpoint(depth - 1, current, next, max_cost);
381
382 double total_cost = local_cost + best_cost;
383
384 // Update minimum cost and corresponding point
385 if (total_cost < min_cost) {
386 min_cost = total_cost;
387 min_point = next;
388 }
389 }
390
391 return {min_cost, min_point};
392}

394double PathCalculation::calculate_midpoint_cost(const shared_ptr<Midpoint>& previous,
395 const shared_ptr<Midpoint>& current,
396 const shared_ptr<Midpoint>& next) const {
397 double dx_dist = current->point.x() - next->point.x();
398 double dy_dist = current->point.y() - next->point.y();
399 double distance = std::sqrt(dx_dist * dx_dist + dy_dist * dy_dist);
400
401 // Angle calculation
402 double angle_with_previous =
403 std::atan2(current->point.y() - previous->point.y(),
404 current->point.x() - previous->point.x());
405 double angle_with_next = std::atan2(next->point.y() - current->point.y(),
406 next->point.x() - current->point.x());
407
408 // Normalize angle to be between 0 and
409 double angle = abs(angle_with_next - angle_with_previous);
410 if (angle > M_PI) {
411 angle = 2 * M_PI - angle;
412 }
413
414 // Local cost calculation
415 return std::pow(angle, config_.angle_exponent_) * config_.angle_gain_ +
416 std::pow(distance, config_.distance_exponent_) * config_.distance_gain_;
417}

506int PathCalculation::find_best_loop_closure(const vector<PathPoint>& path) const {
507 if (path.size() < 3) {
508 return path.size() - 1;
509 }
510
511 const PathPoint& first_point = path[0];
512 double min_cost = numeric_limits<double>::max();
513 int best_cutoff_index = static_cast<int>(path.size() - 1);
514
515 // Check each point in the path to find the best one to close the loop
516 for (int i = 2; i < static_cast<int>(path.size()); ++i) {
517 const PathPoint& current = path[i];
518 const PathPoint& previous = path[i - 1];
519
520 double dx = current.position.x - first_point.position.x;
521 double dy = current.position.y - first_point.position.y;
522 double distance = std::sqrt(dx * dx + dy * dy);
523
524 double angle_with_previous = std::atan2(current.position.y - previous.position.y,
525 current.position.x - previous.position.x);
526 double angle_with_first = std::atan2(first_point.position.y - current.position.y,
527 first_point.position.x - current.position.x);
528
529 // Normalize angle to be between 0 and
530 double angle = abs(angle_with_first - angle_with_previous);
531 if (angle > M_PI) {
532 angle = 2 * M_PI - angle;
533 }
534
535 // Local cost calculation
536 double cost = std::pow(angle, config_.angle_exponent_) * config_.angle_gain_ +
537 std::pow(distance, config_.distance_exponent_) * config_.distance_gain_;
538
539 if (cost < min_cost) {
540 min_cost = cost;
541 best_cutoff_index = i;
542 }
543 }
544
545 return best_cutoff_index;

548vector<PathPoint> PathCalculation::add_interpolated_points(const PathPoint& start,
549 const PathPoint& end,
550 int num_points) const {
551 vector<PathPoint> interpolated;
552 interpolated.reserve(num_points);
553
554 if (num_points <= 0) {
555 return interpolated;
556 }
557
558 float dx = end.position.x - start.position.x;
559 float dy = end.position.y - start.position.y;
560
561 for (int i = 1; i <= num_points; ++i) {
562 float t = static_cast<float>(i) / (num_points + 1);
563 PathPoint intermediate;
564 intermediate.position.x = start.position.x + t * dx;
565 intermediate.position.y = start.position.y + t * dy;
566 interpolated.push_back(intermediate);
567 }
568
569 return interpolated;
570}
