14std::pair<double, PathCalculation::MidPoint*> PathCalculation::dfs_cost(int depth,
15 const MidPoint* previous,
16 MidPoint* current,
17 double maxcost) {
18 if (depth == 0) {
19 return {0, current}; // Return current point if depth is 0
20 }
21
22 double min_cost = this->config_.max_cost_ * this->config_.search_depth_;
23 MidPoint* min_point = current; // Default to current point
24
25 for (const auto& next : current->close_points) {
26 // Avoid revisiting the previous point
27 if (next.get() == previous) {
28 continue;
29 }
30
31 // Distance calculation
32 double distance = std::sqrt(std::pow(current->point.x() - next->point.x(), 2) +
33 std::pow(current->point.y() - next->point.y(), 2));
34
35 // Angle calculation
36 double angle_with_previous = std::atan2(current->point.y() - previous->point.y(),
37 current->point.x() - previous->point.x());
38 double angle_with_next =
39 std::atan2(next->point.y() - current->point.y(), next->point.x() - current->point.x());
40
41 // Normalize angle to be between 0 and Ï€
42 double angle = std::abs(angle_with_next - angle_with_previous);
43 if (angle > M_PI) {
44 angle = 2 * M_PI - angle;
45 }
46
47 // Local cost calculation
48 double local_cost =
49 std::pow(angle, this->config_.angle_exponent_) * this->config_.angle_gain_ +
50 std::pow(distance, this->config_.distance_exponent_) * this->config_.distance_gain_;
51
52 // Skip if local cost exceeds maximum allowed cost
53 if (local_cost > maxcost) {
54 continue;
55 }
56
57 // Recursive cost calculation
58 auto [cost, selected_point] = dfs_cost(depth - 1, current, next.get(), maxcost);
59
60 // Total cost calculation
61 double total_cost = local_cost + cost;
62
63 // Update minimum cost and corresponding point
64 if (total_cost < min_cost) {
65 min_cost = total_cost;
66 min_point = next.get();
67 }
68 }
69
70 return {min_cost, min_point};
71}

77

101 for (size_t i = 0; i < global_path_.size(); ++i) {

103 if (dist < min_dist) {
104 min_dist = dist;
105 cutoff_index = static_cast<int>(i);

107 }

115 if (cutoff_index != -1 && cutoff_index > config_.lookback_points_) {
116 (void)path_to_car.insert(path_to_car.end(), global_path_.begin(),
117 global_path_.begin() + cutoff_index - config_.lookback_points_);
118 }
119
120 // Reset the path periodically to avoid long-term drift or degradation
121 if (path_update_counter_ >= config_.reset_global_path_) {

123 path_to_car.clear();

135 max_points);
136

138 for (const auto& point : global_path) {

143 }

151void PathCalculation::create_mid_points(std::vector<Cone>& cone_array,
152 std::vector<std::shared_ptr<MidPoint>>& midPoints) {
153 this->midPoints.clear();
154 DT dt;
155
156 // Insert all cone positions into the Delaunay triangulation
157 for (const auto& cone : cone_array) {
158 (void)dt.insert(Point(cone.position.x, cone.position.y));
159 }
160
161 // Avoid duplicate midpoints for the same cone pair
162 std::map<std::pair<int, int>, std::shared_ptr<MidPoint>> segment_to_midpoint;
163
164 for (auto fit = dt.finite_faces_begin(); fit != dt.finite_faces_end(); ++fit) {
165 std::array<std::shared_ptr<MidPoint>, 3> mids;
166
167 // Iterate over the 3 edges of the triangle
168 for (int i = 0; i < 3; ++i) {
169 Vertex_handle va = fit->vertex((i + 1) % 3);
170 Vertex_handle vb = fit->vertex((i + 2) % 3);
171
172 Point p1 = va->point();
173 Point p2 = vb->point();
174
175 int id1 = ::find_cone(cone_array, p1.x(), p1.y());
176 int id2 = ::find_cone(cone_array, p2.x(), p2.y());
177
178 if (id1 == -1 || id2 == -1) {
179 continue;
180 }
181
182 double sq_dist = CGAL::squared_distance(p1, p2);
183 if ((sq_dist <= config_.minimum_cone_distance_ * config_.minimum_cone_distance_) ||
184 (sq_dist >= config_.maximum_cone_distance_ * config_.maximum_cone_distance_)) {
185 continue;
186 }
187
188 // Use ordered cone IDs to uniquely identify the segment
189 auto key = std::minmax(id1, id2);
190 auto it = segment_to_midpoint.find(key);
191
192 if (it != segment_to_midpoint.end()) {
193 mids[i] = it->second;
194 } else {
195 auto midpoint = std::make_shared<MidPoint>(
196 MidPoint{CGAL::midpoint(p1, p2), {}, &cone_array[id1], &cone_array[id2]});
197 segment_to_midpoint[key] = midpoint;
198 midPoints.push_back(midpoint);
199 mids[i] = midpoint;
200 }
201 }
202
203 // Connect midpoints if they share the same triangle
204 for (int i = 0; i < 3; ++i) {
205 if (!mids[i]) {
206 continue;
207 }
208 for (int j = 0; j < 3; ++j) {
209 if (i == j || !mids[j]) {
210 continue;
211 }
212 mids[i]->close_points.push_back(mids[j]);
213 }
214 }
215 }
216 for (const auto& p : midPoints) {
217 this->midPoints.push_back(*p);
218 }

286void PathCalculation::extend_path(
287 std::vector<Point>& path, const std::vector<std::shared_ptr<MidPoint>>& midPoints,
288 const std::unordered_map<Point, MidPoint*, PointHash>& point_to_midpoint,
289 std::unordered_set<MidPoint*>& visited_midpoints, std::unordered_set<Cone*>& discarded_cones,
290 int max_points) {
291 int n_points = 0;
292 // Define cost threshold for discarding poor path options
293 double worst_cost = config_.max_cost_ * config_.search_depth_;
294
295 while (true) {
296 const auto& prev = path[path.size() - 2];
297 const auto& last = path.back();
298
299 const MidPoint* prev_mp = find_nearest_point(prev, point_to_midpoint);
300 MidPoint* last_mp = find_nearest_point(last, point_to_midpoint);
301
302 // Abort if midpoints can't be matched
303 if (prev_mp == nullptr || last_mp == nullptr) {
304 RCLCPP_ERROR(rclcpp::get_logger("planning"), "No valid midpoints found for path extension.");
305 break;
306 }
307
308 // Search for the best continuation from the current midpoint
309 auto [best_cost, best_point] =
310 dfs_cost(config_.search_depth_, prev_mp, last_mp, config_.max_cost_);
311
312 // Stop if no good extension is found or point was already visited
313 if (best_cost > worst_cost || best_point == nullptr ||
314 visited_midpoints.count(best_point) > 0) {
315 break;
316 }
317
318 path.push_back(best_point->point);
319 (void)visited_midpoints.insert(best_point);
320 n_points++;
321
322 if (n_points > max_points) {
323 break;
324 }
325
326 // Update midpoints validity and discard cones if needed
327 if (path.size() > 2) {
328 discard_cones_along_path(path, midPoints, point_to_midpoint, discarded_cones);
329 }
330 }

333void PathCalculation::discard_cones_along_path(
334 const std::vector<Point>& path, const std::vector<std::shared_ptr<MidPoint>>& midPoints,
335 const std::unordered_map<Point, MidPoint*, PointHash>& point_to_midpoint,
336 std::unordered_set<Cone*>& discarded_cones) {
337 const auto& last = path[path.size() - 2];
338 const auto& current = path.back();
339
340 MidPoint* last_mp = find_nearest_point(last, point_to_midpoint);
341 const MidPoint* current_mp = find_nearest_point(current, point_to_midpoint);
342
343 if (last_mp == nullptr || current_mp == nullptr) {
344 RCLCPP_ERROR(rclcpp::get_logger("planning"), "No valid midpoints found for discarding cones.");
345 return;
346 }
347
348 // Identify a cone that was likely passed and should be discarded
349 Cone* discarded_cone = nullptr;
350
351 if (last_mp->cone1 == current_mp->cone1 || last_mp->cone1 == current_mp->cone2) {
352 if (last_mp->cone2 != current_mp->cone1 && last_mp->cone2 != current_mp->cone2) {
353 discarded_cone = last_mp->cone2;
354 }
355 } else if (last_mp->cone2 == current_mp->cone1 || last_mp->cone2 == current_mp->cone2) {
356 if (last_mp->cone1 != current_mp->cone1 && last_mp->cone1 != current_mp->cone2) {
357 discarded_cone = last_mp->cone1;
358 }
359 } else {
360 RCLCPP_DEBUG(rclcpp::get_logger("planning"), "No valid cones found for discarding.");
361 }
362
363 // Mark cone as discarded
364 (void)discarded_cones.insert(discarded_cone);
365
366 // Invalidate midpoints that rely on discarded cones
367 for (auto& mp : midPoints) {
368 if (!mp->valid) {
369 continue;
370 }
371 if (discarded_cones.count(mp->cone1) > 0 || discarded_cones.count(mp->cone2) > 0) {
372 mp->valid = false;
373 }
374 }
375
376 // Remove invalid neighbors from each midpoint's connections
377 for (auto& mp : midPoints) {
378 if (!mp->valid) {
379 continue;
380 }
381 (void)mp->close_points.erase(
382 std::remove_if(mp->close_points.begin(), mp->close_points.end(),
383 [](const std::shared_ptr<MidPoint>& neighbor) { return !neighbor->valid; }),
384 mp->close_points.end());
385 }

395std::pair<PathCalculation::MidPoint*, PathCalculation::MidPoint*>
396PathCalculation::find_path_start_points(const std::vector<std::shared_ptr<MidPoint>>& mid_points,
397 const common_lib::structures::Pose& anchor_pose) {
398 std::pair<MidPoint*, MidPoint*> result{nullptr, nullptr};
399
400 auto cmp = [](const std::pair<double, MidPoint*>& cost1,
401 const std::pair<double, MidPoint*>& cost2) {
402 return cost1.first > cost2.first;
403 };
404 std::priority_queue<std::pair<double, MidPoint*>, std::vector<std::pair<double, MidPoint*>>,
405 decltype(cmp)>
406 pq(cmp);
407
408 MidPoint anchor_midpoint{
409 Point(anchor_pose.position.x, anchor_pose.position.y), {}, nullptr, nullptr};
410
411 // Find midpoints that are in front of the car
412 for (const auto& p : mid_points) {
413 double dx = p->point.x() - anchor_midpoint.point.x();
414 double dy = p->point.y() - anchor_midpoint.point.y();
415 double car_direction_x = std::cos(anchor_pose.orientation);
416 double car_direction_y = std::sin(anchor_pose.orientation);
417 if ((dx * car_direction_x + dy * car_direction_y) <= 0.0) {
418 continue;
419 }
420 double dist = std::sqrt(std::pow(dx, 2) + std::pow(dy, 2));
421 double angle = std::atan2(p->point.y() - anchor_midpoint.point.y(),
422 p->point.x() - anchor_midpoint.point.x());
423 double cost =
424 std::pow(angle, this->config_.angle_exponent_) * this->config_.angle_gain_ +
425 std::pow(dist, this->config_.distance_exponent_) * this->config_.distance_gain_;
426 pq.push({cost, p.get()});
427 }
428
429 // Get the closest midpoints in front of the car
430 std::vector<MidPoint*> candidate_points;
431 int count = 0;
432 while (count < 8 && !pq.empty()) {
433 candidate_points.push_back(pq.top().second);
434 pq.pop();
435 count++;
436 }
437
438 // Set the anchor point's connections to these candidates
439 anchor_midpoint.close_points.clear();
440 anchor_midpoint.close_points.reserve(candidate_points.size());
441
442 for (auto* raw_ptr : candidate_points) {
443 for (const auto& mp : mid_points) {
444 if (mp.get() == raw_ptr) {
445 anchor_midpoint.close_points.push_back(mp);
446 break;
447 }
448 }
449 }
450
451 double best_cost = std::numeric_limits<double>::max();
452 for (const auto& first : anchor_midpoint.close_points) {
453 auto [cost, second] = dfs_cost(this->config_.search_depth_, &anchor_midpoint, first.get(),
454 std::numeric_limits<double>::max());
455 cost += std::pow(std::sqrt(std::pow(first->point.x() - anchor_midpoint.point.x(), 2) +
456 std::pow(first->point.y() - anchor_midpoint.point.y(), 2)),
457 this->config_.distance_exponent_) *
458 this->config_.distance_gain_;
459 if (cost < best_cost) {
460 result.first = first.get();
461 result.second = second;
462 best_cost = cost;
463 }
464 }
465
466 return result;
467}

469PathCalculation::MidPoint* PathCalculation::find_nearest_point(
470 const Point& target, const std::unordered_map<Point, MidPoint*, PointHash>& map) {
471 double min_dist_sq = config_.tolerance_ * config_.tolerance_;
472 MidPoint* nearest = nullptr;
473
474 for (const auto& [pt, mp] : map) {
475 double dx = pt.x() - target.x();
476 double dy = pt.y() - target.y();
477 double dist_sq = dx * dx + dy * dy;
478

480 min_dist_sq = dist_sq;
481 nearest = mp;
482 }

485 return nearest; // nullptr if none within tolerance

488std::vector<PathPoint> PathCalculation::calculate_trackdrive(std::vector<Cone>& cone_array,
489 common_lib::structures::Pose pose) {
490 vector<PathPoint> result = no_coloring_planning(cone_array, pose);
491
492 // Check if we have enough points to form a loop
493 if (result.size() < 3) {
494 RCLCPP_WARN(rclcpp::get_logger("planning"), "Not enough points to create trackdrive loop");
495 return result;
496 }
497
498 PathPoint first_point = result[0];
499 PathPoint last_point = result.back();
500
501 // Find the best point to close the loop by checking cost to connect back to first point
502 double min_cost = std::numeric_limits<double>::max();
503 int best_cutoff_index = result.size() - 1; // Default to last point
504
505 // Check each point in the path to find the best one to close the loop
506 for (int i = 2; i < static_cast<int>(result.size()); ++i) {
507 PathPoint current = result[i];
508 PathPoint previous = result[i - 1];
509
510 double dx = current.position.x - first_point.position.x;
511 double dy = current.position.y - first_point.position.y;
512 double distance = std::sqrt(dx * dx + dy * dy);
513
514 double angle_with_previous = std::atan2(current.position.y - previous.position.y,
515 current.position.x - previous.position.x);
516 double angle_with_first = std::atan2(first_point.position.y - current.position.y,
517 first_point.position.x - current.position.x);
518
519 // Normalize angle to be between 0 and Ï€
520 double angle = std::abs(angle_with_first - angle_with_previous);
521 if (angle > M_PI) {
522 angle = 2 * M_PI - angle;
523 }
524
525 // Local cost calculation
526 double cost =
527 std::pow(angle, this->config_.angle_exponent_) * this->config_.angle_gain_ +
528 std::pow(distance, this->config_.distance_exponent_) * this->config_.distance_gain_;
529
530 if (cost < min_cost) {
531 min_cost = cost;
532 best_cutoff_index = i;
533 }
534 }
535
536 // Trim the path to the best cutoff point
537 result.erase(result.begin() + best_cutoff_index + 1, result.end());
538
539 // Add X intermediate points between the last point and the first point
540 int X = 4; // You can adjust this value as needed
541 if (X > 0 && result.size() > 0) {
542 PathPoint last_point = result.back();
543 PathPoint first_point = result[0];
544
545 // Calculate the vector from last to first point
546 float dx = first_point.position.x - last_point.position.x;
547 float dy = first_point.position.y - last_point.position.y;
548
549 // Add X evenly spaced intermediate points
550 for (int i = 1; i <= X; ++i) {
551 float t = static_cast<float>(i) / (X + 1); // Interpolation parameter
552 PathPoint intermediate;
553 intermediate.position.x = last_point.position.x + t * dx;
554 intermediate.position.y = last_point.position.y + t * dy;
555 result.push_back(intermediate);
556 }
557 }
558
559 // Close the loop by adding the first point again
560 result.push_back(first_point);
561
562 // Add overlap points (10 points or as many as available)
563 int overlap_count = std::min(10, static_cast<int>(result.size()) -
564 1); // -1 to exclude the duplicate first point we just added
565 for (int i = 1; i <= overlap_count; ++i) {
566 result.push_back(result[i]);
567 }
568
569 return result;
570}

572std::vector<PathPoint> PathCalculation::process_delaunay_triangulations(
573 std::pair<std::vector<Cone>, std::vector<Cone>> refined_cones) const {
574 // merge left and right cones for next step
575 // RCLCPP_WARN(rclcpp::get_logger("planning"), "Refined cones: %d blue, %d yellow",
576 // static_cast<int>(refined_cones.first.size()),
577 // static_cast<int>(refined_cones.second.size()));
578 std::vector<Cone> cones;
579 cones.reserve(refined_cones.first.size() + refined_cones.second.size());
580 cones.insert(cones.end(), refined_cones.first.begin(), refined_cones.first.end());
581 cones.insert(cones.end(), refined_cones.second.begin(), refined_cones.second.end());
582
583 std::vector<PathPoint> unordered_path;
584 unordered_path.reserve(cones.size());
585
586 // Create a Delaunay triangulation
587 DT dt;
588 // Insert cones coordinates into the Delaunay triangulation
589 for (const Cone& cone : cones) {
590 dt.insert(Point(cone.position.x, cone.position.y));
591 }
592
593 // Process valid triangulations and add positions to unordered_path
594 for (DT::Finite_edges_iterator it = dt.finite_edges_begin(); it != dt.finite_edges_end(); ++it) {
595 // Extract vertices' coordinates from both edges
596 double x1 = it->first->vertex((it->second + 1) % 3)->point().x();
597 double y1 = it->first->vertex((it->second + 1) % 3)->point().y();
598 double x2 = it->first->vertex((it->second + 2) % 3)->point().x();
599 double y2 = it->first->vertex((it->second + 2) % 3)->point().y();
600
601 // Find corresponding cones for the vertices
602 int id_cone1 = find_cone(cones, x1, y1);
603 int id_cone2 = find_cone(cones, x2, y2);
604 // Check both cones have been found
605 if (id_cone1 == -1 || id_cone2 == -1) {
606 RCLCPP_INFO(rclcpp::get_logger("planning"), "Cone not found in triangulations");
607 continue;
608 }
609
610 Cone cone1 = cones[id_cone1];
611 Cone cone2 = cones[id_cone2];
612 // If cones are from different sides
613 if (cone1.color != cone2.color) {
614 // Calculate the midpoint between the two cones
615 double x_dist = cone2.position.x - cone1.position.x;
616 double y_dist = cone2.position.y - cone1.position.y;
617 double dist = std::sqrt(std::pow(x_dist, 2) + std::pow(y_dist, 2));
618
619 if (dist < config_.dist_threshold_) {
620 PathPoint pt = PathPoint(cone1.position.x + x_dist / 2, cone1.position.y + y_dist / 2);
621 unordered_path.push_back(pt);
622 }

642std::vector<PathPoint> PathCalculation::skidpad_path(const std::vector<Cone>& cone_array,
643 common_lib::structures::Pose pose) {
644 std::vector<PathPoint> result;
645
646 // Load data from files only on first iteration
647 if (!skidpad_data_loaded_) {
648 const std::string file = "./src/planning/src/utils/skidpad.txt";
649 const std::string conesfile = "./src/planning/src/utils/skidpadcones1.txt";
650
651 // 1. Read reference cones from file
652 std::ifstream cfile(conesfile);
653 reference_cones_.clear();
654 std::string line;
655 while (std::getline(cfile, line)) {
656 std::istringstream iss(line);
657 double x = 0.0, y = 0.0, z = 0.0;
658 if (iss >> x >> y >> z) {
659 reference_cones_.emplace_back(x, y);
660 } else {
661 break;
662 }
663 }
664
665 // 2. Read hardcoded path from file
666 hardcoded_path_.clear();
667 std::ifstream infile(file);
668 while (std::getline(infile, line)) {
669 std::istringstream iss(line);
670 double x = 0.0, y = 0.0, v = 0.0;
671 if (iss >> x >> y >> v) {
672 hardcoded_path_.emplace_back(x, y, v);
673 } else {
674 break;
675 }
676 }
677
678 skidpad_data_loaded_ = true;
679 }
680
681 // Check if we have enough cones for ICP
682 if (reference_cones_.size() < static_cast<std::size_t>(config_.skidpad_minimum_cones_) ||
683 cone_array.size() < static_cast<std::size_t>(config_.skidpad_minimum_cones_)) {
684 RCLCPP_ERROR(rclcpp::get_logger("planning"), "Not enough cones to perform ICP alignment.");
685 return result; // empty to indicate failure
686 }
687
688 // Perform ICP alignment on every iteration
689 // 3. Convert cone_array (LiDAR) to source cloud
690 pcl::PointCloud<pcl::PointXYZ> cloud_source;
691 cloud_source.reserve(cone_array.size());
692 for (const auto& cone : cone_array) {
693 cloud_source.emplace_back(cone.position.x, cone.position.y, 0.0);
694 }
695
696 // 4. Convert reference cones to target cloud
697 pcl::PointCloud<pcl::PointXYZ> cloud_target;
698 cloud_target.reserve(reference_cones_.size());
699 for (const auto& [x, y] : reference_cones_) {
700 cloud_target.emplace_back(x, y, 0.0);
701 }
702
703 // 5. Run ICP alignment
704 pcl::IterativeClosestPoint<pcl::PointXYZ, pcl::PointXYZ> icp;
705 icp.setInputSource(cloud_source.makeShared());
706 icp.setInputTarget(cloud_target.makeShared());
707 icp.setMaxCorrespondenceDistance(config_.skidpad_tolerance_);
708 icp.setMaximumIterations(std::numeric_limits<int>::max());
709 icp.setTransformationEpsilon(1e-6);
710 icp.setEuclideanFitnessEpsilon(1e-3);
711
712 pcl::PointCloud<pcl::PointXYZ> Final;
713 icp.align(Final);
714
715 if (!icp.hasConverged()) {
716 RCLCPP_ERROR(rclcpp::get_logger("planning"), "ICP did not converge.");
717 return result; // empty to indicate failure
718 }
719
720 // 6. Apply transformation to hardcoded path (create a copy)
721 result = hardcoded_path_; // Copy the hardcoded path
722 Eigen::Matrix4f icp_transform = icp.getFinalTransformation(); // LiDAR â†’ MAP
723 Eigen::Matrix4f map_to_lidar_transform = icp_transform.inverse();
724
725 for (auto& point : result) {
726 Eigen::Vector4f p(point.position.x, point.position.y, 0.0f, 1.0f);
727 Eigen::Vector4f transformed = map_to_lidar_transform * p;
728 point.position.x = transformed[0];
729 point.position.y = transformed[1];
730 }
731
732 // Update predefined_path_ with the transformed path
733 predefined_path_ = result;
734
735 // Get the closest points to the car pose
736 if (predefined_path_.empty()) {
737 RCLCPP_ERROR(rclcpp::get_logger("planning"), "Predefined path is empty.");
738 return result;
739 }
740
741 double min_dist = std::numeric_limits<double>::max();
742 size_t closest_index = 0;
743 for (size_t i = 0; i < predefined_path_.size(); ++i) {
744 double dx = predefined_path_[i].position.x - pose.position.x;
745 double dy = predefined_path_[i].position.y - pose.position.y;
746 double dist = std::sqrt(dx * dx + dy * dy);
747 if (dist < min_dist) {
748 min_dist = dist;
749 closest_index = i;
750 } else if (dist > min_dist) {
751 break; // Stop searching if distance starts increasing
752 }
753 }
754
755 RCLCPP_INFO(rclcpp::get_logger("planning"),
756 "predefined path size is %zu, closest index is %zu \n Hardcoded path size is %zu",
757 predefined_path_.size(), closest_index, hardcoded_path_.size());
758 // Remove all the points before the closest point not removing the closest point itself
759 predefined_path_.erase(predefined_path_.begin(), predefined_path_.begin() + closest_index);
760 hardcoded_path_.erase(hardcoded_path_.begin(), hardcoded_path_.begin() + closest_index);
761
762 size_t path_size = predefined_path_.size();
763 size_t count = 0;
764 if (path_size >= 70) {
765 count = 70;
766 } else {
767 count = path_size;
768 }
769
770 return std::vector<PathPoint>(predefined_path_.begin(), predefined_path_.begin() + count);
771}
