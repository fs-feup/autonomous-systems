15 // RCLCPP_WARN(rclcpp::get_logger("planning"), "Refined cones: %d blue, %d yellow",
16 // static_cast<int>(refined_cones.first.size()),
17 // static_cast<int>(refined_cones.second.size()));

21 cones.insert(cones.end(), refined_cones.second.begin(), refined_cones.second.end());

27 DT dt;
28 // Insert cones coordinates into the Delaunay triangulation
29 for (const Cone &cone : cones) {
30 dt.insert(Point(cone.position.x, cone.position.y));
31 }
32
33 // Process valid triangulations and add positions to unordered_path
34 for (DT::Finite_edges_iterator it = dt.finite_edges_begin(); it != dt.finite_edges_end(); ++it) {
35 // Extract vertices' coordinates from both edges
36 double x1 = it->first->vertex((it->second + 1) % 3)->point().x();
37 double y1 = it->first->vertex((it->second + 1) % 3)->point().y();
38 double x2 = it->first->vertex((it->second + 2) % 3)->point().x();
39 double y2 = it->first->vertex((it->second + 2) % 3)->point().y();
40
41 // Find corresponding cones for the vertices

44 // Check both cones have been found
45 if (id_cone1 == -1 || id_cone2 == -1) {
46 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "Cone not found in triangulations");
47 continue;
48 }
49
50 Cone cone1 = cones[id_cone1];
51 Cone cone2 = cones[id_cone2];
52 // If cones are from different sides
53 if (cone1.color != cone2.color) {
54 // Calculate the midpoint between the two cones
55 double x_dist = cone2.position.x - cone1.position.x;
56 double y_dist = cone2.position.y - cone1.position.y;
57 double dist = sqrt(pow(x_dist, 2) + pow(y_dist, 2));

61 unordered_path.push_back(pt);
62 }
63 }
64 }

96 pow((a.position.y - pose.position.y), 2)) <

110
111 // Find the slope of the line between the two middle points
112 double dx = middle_farthest.position.x - middle_closest.position.x; // --------------------
113 double dy = middle_farthest.position.y - middle_closest.position.y; // --------------------

121 double x = point.position.x;
122 double y = point.position.y;
123 point.position.x = x * cos(angle) - y * sin(angle) + middle_closest.position.x;
124 point.position.y = x * sin(angle) + y * cos(angle) + middle_closest.position.y;
125 }

129 }

133 predefined_path_.erase(predefined_path_.begin());
