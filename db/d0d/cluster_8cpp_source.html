55std::tuple<double, double, double, double> Cluster::calculate_mean_and_std_dev(
56 std::vector<Cluster>& clusters) {
57 double sum_x = 0.0;
58 double sum_y = 0.0;
59 for (auto& point : clusters) {
60 sum_x += point.get_centroid().x();
61 sum_y += point.get_centroid().y();
62 }
63
64 double mean_x = sum_x / static_cast<double>(clusters.size());
65 double mean_y = sum_y / static_cast<double>(clusters.size());
66
67 double sum_squared_diff_x = 0.0;
68 double sum_squared_diff_y = 0.0;
69 for (auto& point : clusters) {
70 sum_squared_diff_x += pow(point.get_centroid().x() - mean_x, 2);
71 sum_squared_diff_y += pow(point.get_centroid().y() - mean_y, 2);
72 }
73
74 double stddev_x = sqrt(sum_squared_diff_x / static_cast<double>(clusters.size()));
75 double stddev_y = sqrt(sum_squared_diff_y / static_cast<double>(clusters.size()));
76
77 return std::make_tuple(mean_x, mean_y, stddev_x, stddev_y);
78}
