3void PathSmoothing::order_path(std::vector<PathPoint>& unord_path, const Pose& car_pose) const {
4 std::unordered_set<PathPoint> unord_set(unord_path.begin(), unord_path.end());
5
6 PathPoint current_point = PathPoint(car_pose.position, 1);
7 int index = 0;
8
9 while (!unord_set.empty()) {
10 double min_distance = std::numeric_limits<double>::max();
11 PathPoint closest_point;
12
13 for (const auto& point : unord_set) {
14 // If the algorithm is trying to decide which point is the second in the path, it should
15 // guarantee that the orientation that the point gives to the path is aligned with the car's
16 // orientation.
17 // This is not done in the first point because the first point often occurs backwards
18 // relative
19 // to the car's pose.
20 if (index == 1) {
21 double path_angle = atan2(point.position.y - current_point.position.y,
22 point.position.x - current_point.position.x);
23
24 // Check if next point is in the same direction range or side as the vehicle orientation
25 double angle_diff = fmod(path_angle - car_pose.orientation + 2 * M_PI, 2 * M_PI);
26 if (abs(angle_diff) > M_PI / 2 && abs(angle_diff) < 3 * M_PI / 2) {
27 continue;
28 }
29 }
30
31 double distance = current_point.position.euclidean_distance(point.position);
32 if (distance < min_distance) {
33 min_distance = distance;
34 closest_point = point;
35 }
36 }
37
38 if (min_distance < std::numeric_limits<double>::max() &&
39 index < static_cast<int>(unord_path.size())) {
40 current_point = closest_point;
41 unord_set.erase(current_point);
42 unord_path[index] = current_point;
43 index++;
44 } else {
45 RCLCPP_ERROR(rclcpp::get_logger("planning"),
46 "Index out of bounds while ordering path OR no valid point found");
47 break;
48 }
49 }
50}
