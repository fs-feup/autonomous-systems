28void PathSmoothing::add_curvature_terms(
29 int num_path_points,
30 const std::function<int(int)>& circular_index,
31 const std::function<void(int, int, double)>& add_coefficient) const {
32 // -------- ADD CURVATURE PENALTY TERMS --------
33 // Penalize second-order differences to minimize curvature
34 // For each point, we penalize: (p[i-1] - 2*p[i] + p[i+1])^2
35 for (int point_idx = 0; point_idx < num_path_points; ++point_idx) {
36 int prev_point = circular_index(point_idx - 1);
37 int next_point = circular_index(point_idx + 1);
38
39 // X-coordinate curvature terms
40 int x_prev = 2 * prev_point;
41 int x_current = 2 * point_idx;
42 int x_next = 2 * next_point;
43
44 add_coefficient(x_prev, x_prev, config_.curvature_weight_);
45 add_coefficient(x_current, x_current, 4 * config_.curvature_weight_);
46 add_coefficient(x_next, x_next, config_.curvature_weight_);
47 add_coefficient(x_prev, x_current, -2 * config_.curvature_weight_);
48 add_coefficient(x_current, x_next, -2 * config_.curvature_weight_);
49 add_coefficient(x_prev, x_next, config_.curvature_weight_);
50
51 // Y-coordinate curvature terms
52 int y_prev = 2 * prev_point + 1;
53 int y_current = 2 * point_idx + 1;
54 int y_next = 2 * next_point + 1;
55
56 add_coefficient(y_prev, y_prev, config_.curvature_weight_);
57 add_coefficient(y_current, y_current, 4 * config_.curvature_weight_);
58 add_coefficient(y_next, y_next, config_.curvature_weight_);
59 add_coefficient(y_prev, y_current, -2 * config_.curvature_weight_);
60 add_coefficient(y_current, y_next, -2 * config_.curvature_weight_);
61 add_coefficient(y_prev, y_next, config_.curvature_weight_);
62 }
63}

65void PathSmoothing::add_smoothness_terms(
66 int num_path_points,
67 const std::function<int(int)>& circular_index,
68 const std::function<void(int, int, double)>& add_coefficient) const {
69 // -------- ADD SMOOTHNESS PENALTY TERMS --------
70 // Penalize first-order differences to minimize jerk
71 // For each consecutive pair, we penalize: (p[i+1] - p[i])^2
72 for (int point_idx = 0; point_idx < num_path_points; ++point_idx) {
73 int next_point = circular_index(point_idx + 1);
74
75 // X-coordinate smoothness terms
76 int x_current = 2 * point_idx;
77 int x_next = 2 * next_point;
78 add_coefficient(x_current, x_current, config_.smoothness_weight_);
79 add_coefficient(x_next, x_next, config_.smoothness_weight_);
80 add_coefficient(x_current, x_next, -config_.smoothness_weight_);
81
82 // Y-coordinate smoothness terms
83 int y_current = 2 * point_idx + 1;
84 int y_next = 2 * next_point + 1;
85 add_coefficient(y_current, y_current, config_.smoothness_weight_);
86 add_coefficient(y_next, y_next, config_.smoothness_weight_);
87 add_coefficient(y_current, y_next, -config_.smoothness_weight_);
88 }

103void PathSmoothing::add_boundary_constraints(
104 std::vector<OSQPFloat>& constraint_values, std::vector<OSQPInt>& constraint_row_indices,
105 std::vector<OSQPInt>& constraint_col_indices, std::vector<OSQPFloat>& constraint_lower_bounds,
106 std::vector<OSQPFloat>& constraint_upper_bounds, int& constraint_count,
107 const std::vector<PathPoint>& left, const std::vector<PathPoint>& right, int num_path_points,
108 double safety_margin) const {
109 // -------- ADD TRACK BOUNDARY CONSTRAINTS --------
110 // For each point, ensure it stays within the left and right boundaries
111 for (int point_idx = 0; point_idx < num_path_points; ++point_idx) {
112 Eigen::Vector2d left_boundary_point(left[point_idx].position.x, left[point_idx].position.y);
113 Eigen::Vector2d right_boundary_point(right[point_idx].position.x, right[point_idx].position.y);
114
115 // Compute lateral direction (perpendicular to track)
116 Eigen::Vector2d lateral_direction = (left_boundary_point - right_boundary_point).normalized();
117
118 // Right boundary constraint: lateral_direction · point + slack >= right_boundary_value
119 constraint_row_indices.push_back(constraint_count);
120 constraint_col_indices.push_back(2 * point_idx);
121 constraint_values.push_back(lateral_direction.x());
122
123 constraint_row_indices.push_back(constraint_count);
124 constraint_col_indices.push_back(2 * point_idx + 1);
125 constraint_values.push_back(lateral_direction.y());
126
127 constraint_row_indices.push_back(constraint_count);
128 constraint_col_indices.push_back(2 * num_path_points + 2 * point_idx);
129 constraint_values.push_back(1.0);
130
131 double right_boundary_constraint = right_boundary_point.dot(lateral_direction) + safety_margin;
132 constraint_lower_bounds.push_back(right_boundary_constraint);
133 constraint_upper_bounds.push_back(OSQP_INFTY);
134 constraint_count++;
135
136 // Left boundary constraint: -lateral_direction · point + slack >= -left_boundary_value
137 constraint_row_indices.push_back(constraint_count);
138 constraint_col_indices.push_back(2 * point_idx);
139 constraint_values.push_back(-lateral_direction.x());
140
141 constraint_row_indices.push_back(constraint_count);
142 constraint_col_indices.push_back(2 * point_idx + 1);
143 constraint_values.push_back(-lateral_direction.y());
144
145 constraint_row_indices.push_back(constraint_count);
146 constraint_col_indices.push_back(2 * num_path_points + 2 * point_idx + 1);
147 constraint_values.push_back(1.0);
148
149 double left_boundary_constraint = -left_boundary_point.dot(lateral_direction) + safety_margin;
150 constraint_lower_bounds.push_back(left_boundary_constraint);
151 constraint_upper_bounds.push_back(OSQP_INFTY);
152 constraint_count++;
153 }
154}

175void PathSmoothing::convert_to_csc_format(const std::vector<OSQPFloat>& values,
176 const std::vector<OSQPInt>& row_indices,
177 const std::vector<OSQPInt>& col_indices,
178 int total_variables, std::vector<OSQPFloat>& csc_x,
179 std::vector<OSQPInt>& csc_i,
180 std::vector<OSQPInt>& csc_p) const {
181 // Convert coordinate format to CSC (Compressed Sparse Column) format
182 std::vector<std::vector<std::pair<OSQPInt, OSQPFloat>>> columns(total_variables);
183 for (size_t entry = 0; entry < values.size(); ++entry) {
184 columns[col_indices[entry]].push_back({row_indices[entry], values[entry]});
185 }
186
187 // Sort each column by row index
188 for (int col = 0; col < total_variables; ++col) {
189 std::sort(columns[col].begin(), columns[col].end(),
190 [](const auto& a, const auto& b) { return a.first < b.first; });
191 }
192
193 // Build CSC arrays
194 csc_x.reserve(values.size());
195 csc_i.reserve(values.size());
196 csc_p.resize(total_variables + 1);
197
198 csc_p[0] = 0;
199 for (int col = 0; col < total_variables; ++col) {
200 for (const auto& [row, value] : columns[col]) {
201 csc_i.push_back(row);
202 csc_x.push_back(value);
203 }
204 csc_p[col + 1] = csc_x.size();
205 }
206}

208std::vector<PathPoint> PathSmoothing::osqp_optimization(const std::vector<PathPoint>& center,
209 const std::vector<PathPoint>& left,
210 const std::vector<PathPoint>& right) const {
211 const int num_path_points = center.size();
212
213 // Check if we have enough points for optimization
214 if (num_path_points < 5) {
215 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"),
216 "Too few points for OSQP optimization (%d points). Minimum is 5.",
217 num_path_points);
218 return center;
219 }
220
221 // -------- COMPUTE SAFETY MARGIN --------
222 const double safety_margin = config_.car_width_ / 2 + config_.safety_margin_;
223
224 // Helper lambda for circular indexing (wraps around the path)
225 auto circular_index = [&](int i) { return (i + num_path_points) % num_path_points; };
226
227 // -------- DEFINE OPTIMIZATION VARIABLES --------
228 // Decision variables: 2 coordinates (x,y) per point + 2 slack variables per point
229 const int num_slack_variables = 2 * num_path_points;
230 const int total_variables = 2 * num_path_points + num_slack_variables;
231
232 // -------- BUILD QUADRATIC OBJECTIVE MATRIX (P) --------
233 // Store quadratic terms in a map for easy accumulation, then convert to sparse format
234 std::map<std::pair<int, int>, double> quadratic_terms;
235
236 // Helper lambda to add coefficients to the upper triangular part of P
237 auto add_quadratic_coefficient = [&](int row_idx, int col_idx, double coefficient) {
238 if (row_idx > col_idx) {
239 std::swap(row_idx, col_idx);
240 }
241 quadratic_terms[{row_idx, col_idx}] += coefficient;
242 };
243
244 add_curvature_terms(num_path_points, circular_index, add_quadratic_coefficient);
245 add_smoothness_terms(num_path_points, circular_index, add_quadratic_coefficient);
246 add_slack_penalty_terms(num_path_points, add_quadratic_coefficient);
247
248 // -------- BUILD LINEAR OBJECTIVE VECTOR (q) --------
249 // All zeros since we only have quadratic terms in the objective
250 std::vector<OSQPFloat> linear_objective(total_variables, 0.0);
251
252 // -------- CONVERT QUADRATIC TERMS TO SPARSE FORMAT --------
253 std::vector<OSQPFloat> P_values;
254 std::vector<OSQPInt> P_row_indices, P_col_indices;
255
256 P_values.reserve(quadratic_terms.size());
257 P_row_indices.reserve(quadratic_terms.size());
258 P_col_indices.reserve(quadratic_terms.size());
259
260 for (const auto& [indices, value] : quadratic_terms) {
261 P_row_indices.push_back(indices.first);
262 P_col_indices.push_back(indices.second);
263 P_values.push_back(value);
264 }
265
266 // -------- BUILD CONSTRAINT MATRIX (A) AND BOUNDS --------
267 std::vector<OSQPFloat> constraint_values;
268 std::vector<OSQPInt> constraint_row_indices, constraint_col_indices;
269 std::vector<OSQPFloat> constraint_lower_bounds, constraint_upper_bounds;
270
271 int constraint_count = 0;
272
273 add_boundary_constraints(constraint_values, constraint_row_indices, constraint_col_indices,
274 constraint_lower_bounds, constraint_upper_bounds, constraint_count, left,
275 right, num_path_points, safety_margin);
276
277 add_slack_nonnegativity_constraints(constraint_values, constraint_row_indices,
278 constraint_col_indices, constraint_lower_bounds,
279 constraint_upper_bounds, constraint_count, num_path_points);
280
281 const int total_constraints = constraint_count;
282
283 // Build CSC arrays for P
284 std::vector<OSQPFloat> P_x;
285 std::vector<OSQPInt> P_i;
286 std::vector<OSQPInt> P_p;
287 convert_to_csc_format(P_values, P_row_indices, P_col_indices, total_variables, P_x, P_i, P_p);
288
289 // Build CSC arrays for A
290 std::vector<OSQPFloat> A_x;
291 std::vector<OSQPInt> A_i;
292 std::vector<OSQPInt> A_p;
293 convert_to_csc_format(constraint_values, constraint_row_indices, constraint_col_indices,
294 total_variables, A_x, A_i, A_p);
295
296 // -------- POPULATE MATRIX STRUCTURES --------
297 OSQPCscMatrix objective_matrix;
298 objective_matrix.m = total_variables;
299 objective_matrix.n = total_variables;
300 objective_matrix.nzmax = P_x.size();
301 objective_matrix.nz = -1;
302 objective_matrix.x = P_x.data();
303 objective_matrix.i = P_i.data();
304 objective_matrix.p = P_p.data();
305
306 OSQPCscMatrix constraint_matrix;
307 constraint_matrix.m = total_constraints;
308 constraint_matrix.n = total_variables;
309 constraint_matrix.nzmax = A_x.size();
310 constraint_matrix.nz = -1;
311 constraint_matrix.x = A_x.data();
312 constraint_matrix.i = A_i.data();
313 constraint_matrix.p = A_p.data();
314
315 // -------- CONFIGURE OSQP SOLVER SETTINGS --------
316 OSQPSettings solver_settings;
317 ::osqp_set_default_settings(&solver_settings);
318 solver_settings.verbose = 1;
319 solver_settings.max_iter = config_.max_iterations_;
320 solver_settings.eps_abs = config_.tolerance_;
321 solver_settings.eps_rel = config_.tolerance_;
322 solver_settings.polishing = 1;
323
324 // -------- SETUP OSQP SOLVER --------
325 OSQPSolver* solver = nullptr;
326
327 OSQPInt setup_status =
328 ::osqp_setup(&solver, &objective_matrix, linear_objective.data(), &constraint_matrix,
329 constraint_lower_bounds.data(), constraint_upper_bounds.data(), total_variables,
330 total_constraints, &solver_settings);
331
332 if (setup_status != 0) {
333 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "OSQP setup failed with status %lld", setup_status);
334 if (solver) {
335 (void)::osqp_cleanup(solver);
336 }
337 return center;
338 }
339
340 // -------- SOLVE THE OPTIMIZATION PROBLEM --------
341 OSQPInt solve_status = ::osqp_solve(solver);
342
343 // -------- EXTRACT OPTIMIZED PATH FROM SOLUTION --------
344 std::vector<PathPoint> optimized_path = center;
345
346 if ((solver->solution != nullptr) && (solver->solution->x != nullptr)) {
347 for (int point_idx = 0; point_idx < num_path_points; ++point_idx) {
348 optimized_path[point_idx].position.x = solver->solution->x[2 * point_idx];
349 optimized_path[point_idx].position.y = solver->solution->x[2 * point_idx + 1];
350 }
351
352 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"),
353 "OSQP optimization completed successfully with %d points (status: %lld)",
354 num_path_points, solve_status);
355 } else {
356 RCLCPP_WARN(rclcpp::get_logger("rclcpp"), "OSQP solution is null, returning original path");
357 }
358
359 // -------- CLEANUP OSQP RESOURCES --------
360 (void)::osqp_cleanup(solver);
361
362 return optimized_path;
363}
