115TEST(NORMAL_VELOCITY_MODEL, CIRCULAR_MOVEMENT_TEST) {
116 Eigen::MatrixXf r_matrix = Eigen::MatrixXf::Zero(6, 6);
117 NormalVelocityModel motion_model = NormalVelocityModel(r_matrix);
118 MotionUpdate prediction_data = {0, 0, 0, 0, 0, rclcpp::Clock().now()};
119 Eigen::VectorXf temp_state;
120 Eigen::VectorXf new_state = Eigen::VectorXf::Zero(10);
121 float radius = static_cast<float>(12.7324);
122 new_state(0) = 0;
123 new_state(1) = -radius; // Radius of the circle
124 Eigen::MatrixXf g_matrix =
125 motion_model.get_motion_to_state_matrix(Eigen::VectorXf::Zero(10), prediction_data, 1.0);
126 Eigen::MatrixXf new_covariance = Eigen::MatrixXf::Ones(10, 10);
127 for (int i = 0; i < 1000; i++) {
128 prediction_data.translational_velocity = 10;
129 prediction_data.rotational_velocity = M_PI / 4;
130 temp_state = new_state;
131 new_state = motion_model.predict_expected_state(new_state, prediction_data, 0.1);
132 g_matrix = motion_model.get_motion_to_state_matrix(new_state, prediction_data, 0.1);
133 new_covariance = g_matrix * new_covariance * g_matrix.transpose();
134 // Max 0.01 percent calculation error
135 EXPECT_LT(new_state(0), radius + 0.1); // x
136 EXPECT_LT(new_state(1), radius + 0.1); // y
137 EXPECT_NEAR(new_state(0) * new_state(0) + new_state(1) * new_state(1), radius * radius,
138 0.0001 * radius * radius); // Radius
139 EXPECT_LT(new_state(2), 2 * M_PI); // theta
140 }
