46void Planning::track_map_callback(const custom_interfaces::msg::ConeArray msg) {
47 if (this->is_predicitve_mission()) {
48 return;
49 }
50
51 Track *track = new Track();
52
53 auto cone_array = msg.cone_array;
54
55 for (auto &cone : cone_array) {
56 RCLCPP_DEBUG(this->get_logger(), "[received] (%f, %f)\t%s", cone.position.x, cone.position.y,
57 cone.color.c_str());
58 track->addCone(cone.position.x, cone.position.y, cone.color);
59 }
60
61 track->validateCones(); // Delete cone outliers
62
63 std::vector<PathPoint *> path = local_path_planner->processNewArray(track); // Calculate Path
64 publish_track_points(path);
65 delete (track);
66
67 RCLCPP_DEBUG(this->get_logger(), "--------------------------------------");

73void Planning::publish_track_points(std::vector<PathPoint *> path) const {

79 point.v = element->getV();
80 message.pathpoint_array.push_back(point);
