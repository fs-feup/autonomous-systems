122void Planning::fetch_discipline() {
123 common_lib::competition_logic::Mission mission_result =
124 common_lib::competition_logic::Mission::NONE;
125
126 if (!param_client_->wait_for_service(std::chrono::milliseconds(100))) {
127 RCLCPP_ERROR(this->get_logger(), "Service /pacsim/pacsim_node/get_parameters not available.");
128 } else {
129 auto request = std::make_shared<rcl_interfaces::srv::GetParameters::Request>();
130 request->names.push_back("discipline");
131
132 param_client_->async_send_request(
133 request, [this](rclcpp::Client<rcl_interfaces::srv::GetParameters>::SharedFuture future) {
134 auto response = future.get();
135 common_lib::competition_logic::Mission mission_result =
136 common_lib::competition_logic::Mission::AUTOCROSS;
137
138 if (!response->values.empty() && response->values[0].type == 4) { // Type 4 = string
139 std::string discipline = response->values[0].string_value;
140 RCLCPP_INFO(this->get_logger(), "Discipline received: %s", discipline.c_str());
141
142 if (discipline == "skidpad") {
143 mission_result = common_lib::competition_logic::Mission::SKIDPAD;
144 } else if (discipline == "acceleration") {
145 mission_result = common_lib::competition_logic::Mission::ACCELERATION;
146 } else if (discipline == "trackdrive") {
147 mission_result = common_lib::competition_logic::Mission::TRACKDRIVE;
148 }
149 } else {
150 RCLCPP_ERROR(this->get_logger(), "Failed to retrieve discipline parameter.");
151 }
152
153 this->mission = mission_result;
154 });
155 }
156
157 this->mission = mission_result;
158}
