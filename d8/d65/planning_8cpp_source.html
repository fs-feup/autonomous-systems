78void Planning::run_planning_algorithms() {
79 RCLCPP_DEBUG(rclcpp::get_logger("planning"), "Running Planning Algorithms");
80 if (this->cone_array_.empty()) {
81 publish_track_points({});
82 return;
83 }
84
85 rclcpp::Time start_time = this->now();
86
87 // Color the cones
88 std::pair<std::vector<Cone>, std::vector<Cone>> colored_cones =
89 cone_coloring_.color_cones(this->cone_array_, this->pose);
90 if (colored_cones.first.size() < 2 || colored_cones.second.size() < 2) {
91 RCLCPP_WARN(rclcpp::get_logger("planning"), "Not enough cones to plan: %d blue, %d yellow",
92 static_cast<int>(colored_cones.first.size()),
93 static_cast<int>(colored_cones.second.size()));
94 publish_track_points({});
95 return;
96 }
97
98 // Outliers dealt by approximating all cones
99 std::pair<std::vector<Cone>, std::vector<Cone>> refined_colored_cones =
100 outliers_.approximate_cones_with_spline(colored_cones);
101 if (refined_colored_cones.first.size() < 2 || refined_colored_cones.second.size() < 2) {
102 RCLCPP_WARN(rclcpp::get_logger("planning"),
103 "Not enough cones to plan after outlier removal: %d blue, %d yellow",
104 static_cast<int>(refined_colored_cones.first.size()),
105 static_cast<int>(refined_colored_cones.second.size()));
106 publish_track_points({});
107 return;
108 }
109 for (auto &cone : colored_cones.first) {
110 cone.color = Color::BLUE;
111 }
112 for (auto &cone : colored_cones.second) {
113 cone.color = Color::YELLOW;
114 }
115
116 // Calculate middle points using triangulations
117 std::vector<PathPoint> triangulations_path =
118 path_calculation_.process_delaunay_triangulations(refined_colored_cones);
119 if (triangulations_path.size() < 2) {
120 RCLCPP_WARN(rclcpp::get_logger("planning"), "Not enough cones to plan after triangulations: %d",
121 static_cast<int>(triangulations_path.size()));
122 publish_track_points({});
123 return;
124 }
125
126 // Smooth the calculated path
127 std::vector<PathPoint> final_path = path_smoothing_.smooth_path(triangulations_path, this->pose);
128
129 if (final_path.size() < 10) {
130 RCLCPP_WARN(rclcpp::get_logger("planning"), "Final path size: %d",
131 static_cast<int>(final_path.size()));
132 }
133 // Velocity Planning
134 // TODO: Remove this when velocity planning is a reality
135 for (auto &path_point : final_path) {
136 path_point.ideal_velocity = desired_velocity_;
137 }
138
139 // Execution Time calculation
140 rclcpp::Time end_time = this->now();
141 std_msgs::msg::Float64 planning_execution_time;
142 planning_execution_time.data = (end_time - start_time).seconds() * 1000;
143 this->_planning_execution_time_publisher_->publish(planning_execution_time);
144
145 publish_track_points(final_path);
146 RCLCPP_DEBUG(this->get_logger(), "Planning will publish %i path points\n",
147 static_cast<int>(final_path.size()));
148
149 if (planning_config_.simulation_.publishing_visualization_msgs_) {
150 publish_visualization_msgs(colored_cones.first, colored_cones.second,
151 refined_colored_cones.first, refined_colored_cones.second,
152 triangulations_path, final_path);
153 }
154}
