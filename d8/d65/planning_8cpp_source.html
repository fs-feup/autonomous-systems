144void Planning::fetch_discipline() {
145 Mission mission_result = Mission::NONE;
146
147 if (!param_client_->wait_for_service(std::chrono::milliseconds(100))) {
148 RCLCPP_ERROR(get_logger(), "Service /pacsim/pacsim_node/get_parameters not available.");
149 } else {
150 auto request = std::make_shared<rcl_interfaces::srv::GetParameters::Request>();
151 request->names.push_back("discipline");
152
153 (void)param_client_->async_send_request(
154 request, [this](rclcpp::Client<rcl_interfaces::srv::GetParameters>::SharedFuture future) {
155 auto response = future.get();
156 Mission mission_result = Mission::AUTOCROSS;
157
158 if (!response->values.empty() && response->values[0].type == 4) {
159 std::string discipline = response->values[0].string_value;
160 RCLCPP_INFO(get_logger(), "Discipline received: %s", discipline.c_str());
161
162 if (discipline == "skidpad") {
163 mission_result = Mission::SKIDPAD;
164 } else if (discipline == "acceleration") {
165 mission_result = Mission::ACCELERATION;
166 } else if (discipline == "trackdrive") {
167 mission_result = Mission::TRACKDRIVE;
168 }else{
169 mission_result = Mission::AUTOCROSS;
170 }
171 } else {
172 RCLCPP_ERROR(get_logger(), "Failed to retrieve discipline parameter.");
173 }
174
175 mission_ = mission_result;
176 });
177 }
178
179 mission_ = mission_result;
180}

226void Planning::run_ebs_test() {
227 calculate_and_smooth_path();
228
229 double distance_from_origin = std::sqrt(pose_.position.x * pose_.position.x +
230 pose_.position.y * pose_.position.y);
231
232 if (distance_from_origin > 90.0) {
233 if (!is_braking_) {
234 is_braking_ = true;
235 brake_time_ = std::chrono::steady_clock::now();
236 }
237
238 for (PathPoint &point : final_path_) {
239 std::chrono::duration<double> time_since_brake_start =
240 std::chrono::steady_clock::now() - brake_time_;
241 point.ideal_velocity = std::max(
242 (desired_velocity_ + (planning_config_.velocity_planning_.braking_acceleration_ *
243 time_since_brake_start.count())),
244 0.0);
245 }
246 } else {
247 for (PathPoint &point : final_path_) {
248 point.ideal_velocity = desired_velocity_;
249 }
250 }

291void Planning::run_planning_algorithms() {
292 RCLCPP_DEBUG(rclcpp::get_logger("planning"), "Running Planning Algorithms");
293
294 if (cone_array_.empty()) {
295 publish_path_points();
296 return;
297 }
298
299 rclcpp::Time start_time = now();
300
301 switch (mission_) {
302 case Mission::NONE:
303 RCLCPP_ERROR(get_logger(), "Mission is NONE, cannot run planning algorithms.");
304 return;
305
306 case Mission::SKIDPAD:
307 final_path_ = skidpad_.skidpad_path(cone_array_, pose_);
308 break;
309
310 case Mission::ACCELERATION:
311 case Mission::EBS_TEST:
312 run_ebs_test();
313 break;
314
315 case Mission::AUTOCROSS:
316 run_autocross();
317 break;
318
319 case Mission::TRACKDRIVE:
320 run_trackdrive();
321 break;
322
323 default:
324 calculate_and_smooth_path();
325 velocity_planning_.set_velocity(final_path_);
326 break;
327 }
328
329 if (final_path_.size() < 10) {
330 RCLCPP_INFO(rclcpp::get_logger("planning"), "Final path size: %d",
331 static_cast<int>(final_path_.size()));
332 }
333
334 publish_execution_time(start_time);
335 publish_path_points();
336
337 RCLCPP_DEBUG(get_logger(), "Planning will publish %i path points\n",
338 static_cast<int>(final_path_.size()));
339
340 if (planning_config_.simulation_.publishing_visualization_msgs_) {
341 publish_visualization_msgs();
342 }
343}
