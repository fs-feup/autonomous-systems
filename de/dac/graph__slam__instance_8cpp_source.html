37 state(0) = current_pose.x();

173void GraphSLAMInstance::process_observations(const ObservationData& observation_data) {
174 Eigen::VectorXd& observations = *(observation_data.observations_);
175 Eigen::VectorXi& associations = *(observation_data.associations_);
176 Eigen::VectorXd& observations_global = *(observation_data.observations_global_);
177 bool new_observation_factors = false;
178 for (unsigned int i = 0; i < observations.size() / 2; i++) {
179 gtsam::Point2 landmark;
180 gtsam::Symbol landmark_symbol;
181 bool landmark_lost_in_optimization =
182 static_cast<int>(this->_landmark_counter_) < (associations(i)) / 2;
183 if (associations(i) == -2) {
184 // No association
185 continue;
186 } else if (associations(i) == -1 || landmark_lost_in_optimization) {
187 // Create new landmark
188 landmark_symbol = gtsam::Symbol('l', ++(this->_landmark_counter_));
189 landmark = gtsam::Point2(observations_global(i * 2), observations_global(i * 2 + 1));
190 this->_graph_values_.insert(landmark_symbol, landmark);
191 } else {
192 // Association to previous landmark
193 landmark_symbol = gtsam::Symbol('l', (associations(i)) / 2 + 1); // Convert to landmark id
194 }
195 new_observation_factors = true;
196 const Eigen::Vector2d observation_cartesian(observations[i * 2], observations[i * 2 + 1]);
197 Eigen::Vector2d observation_cylindrical = common_lib::maths::cartesian_to_cylindrical(
198 observation_cartesian); // Convert to cylindrical coordinates
199 const gtsam::Rot2 observation_rotation(observation_cylindrical(1));
200
201 const gtsam::noiseModel::Diagonal::shared_ptr observation_noise =
202 gtsam::noiseModel::Diagonal::Sigmas(gtsam::Vector2(this->_params_.observation_x_noise_,
203 this->_params_.observation_y_noise_));
204
205 this->_factor_graph_.add(gtsam::BearingRangeFactor<gtsam::Pose2, gtsam::Point2>(
206 gtsam::Symbol('x', this->_pose_counter_), landmark_symbol, observation_rotation,
207 observation_cylindrical(0), observation_noise));
208 }
209 this->_new_pose_node_ = !new_observation_factors;
210 this->_new_observation_factors_ = new_observation_factors;
211}
