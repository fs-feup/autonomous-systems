255 def calculate_theta(self, p1, p2):
256 """
257 Calculate the angle (in radians) between two points relative to the horizontal axis.
258 """
259 dx = p2[0] - p1[0]
260 dy = p2[1] - p1[1]
261 return math.atan2(dy, dx)
