73Eigen::MatrixXd NoRearWSSBicycleModel::jacobian_cg_velocity_to_wheels(
74 const Eigen::Vector3d& cg_velocities) {
75 Eigen::MatrixXd jacobian = Eigen::MatrixXd::Zero(4, 3);
76 double vx = cg_velocities(0);
77 double vy = cg_velocities(1);
78 double omega = cg_velocities(2);
79 double lr = this->car_parameters_
80 .dist_cg_2_rear_axis; // distance from the center of mass to the rear wheels
81 double lf = this->car_parameters_.wheelbase -
82 this->car_parameters_
83 .dist_cg_2_rear_axis; // distance from the center of mass to the front wheels
84 double rear_wheel_velocity = sqrt(pow(vx, 2) + pow(vy - omega * lr, 2));
85 double front_wheel_velocity = sqrt(pow(vx, 2) + pow(vy + omega * lf, 2));
86
87 const double epsilon = 1e-2;
88 if (std::fabs(rear_wheel_velocity) < epsilon || std::fabs(front_wheel_velocity) < epsilon) {
89 jacobian(3, 0) = epsilon;
90 return jacobian;
91 }
92
93 double sign = (vx < 0) ? -1.0 : 1.0;
94 jacobian(0, 0) = std::max(
95 0.0, sign * 60 * vx / (M_PI * this->car_parameters_.wheel_diameter * front_wheel_velocity));
96 jacobian(0, 1) = 0;
97 // std::max(0.0, sign * 60 * (vy + omega * lf) /
98 // (M_PI * this->car_parameters_.wheel_diameter * front_wheel_velocity));
99 jacobian(0, 2) = sign * 60 * lf * (vy + omega * lf) /
100 (M_PI * this->car_parameters_.wheel_diameter * front_wheel_velocity);
101 jacobian(1, 0) = jacobian(0, 0);
102 jacobian(1, 1) = jacobian(0, 1);
103 jacobian(1, 2) = jacobian(0, 2);
104
105 jacobian(3, 0) = sign * this->car_parameters_.gear_ratio * 60 * vx /
106 (M_PI * this->car_parameters_.wheel_diameter * rear_wheel_velocity);
107 jacobian(3, 1) = 0; // sign * this->car_parameters_.gear_ratio * 60 * (vy - omega * lr) /
108 //(M_PI * this->car_parameters_.wheel_diameter * rear_wheel_velocity);
109 jacobian(3, 2) = sign * this->car_parameters_.gear_ratio * 60 * (-lr) * (vy - omega * lr) /
110 (M_PI * this->car_parameters_.wheel_diameter * rear_wheel_velocity);
111 double L = this->car_parameters_.wheelbase;
112 double v = sqrt(pow(vx, 2) + pow(vy, 2));
113 if (std::fabs(v) > epsilon) {
114 jacobian(2, 0) = -((omega * L) / (v * v + (omega * L) * (omega * L))) * (vx / v);
115 jacobian(2, 1) = -((omega * L) / (v * v + (omega * L) * (omega * L))) * (vy / v);
116 jacobian(2, 2) = (L * v) / (v * v + (omega * L) * (omega * L));
117 }
118
119 return jacobian;
120}
