95void SLAMNode::_perception_subscription_callback(
96 const custom_interfaces::msg::PerceptionOutput &msg) {
97 auto const &cone_array = msg.cones.cone_array;
98
99 if (!this->_go_ || this->_mission_ == common_lib::competition_logic::Mission::NONE) {
100 return;
101 }
102
103 rclcpp::Time start_time = this->get_clock()->now();
104 rclcpp::Time current_stamp(msg.header.stamp);
105 if (_last_perception_message_time_.nanoseconds() != 0) {
106 double delta_ms = (current_stamp - _last_perception_message_time_).seconds() * 1000.0;
107
108 std_msgs::msg::Float64 msg;
109 msg.data = delta_ms;
110 _perception_delta_publisher_->publish(msg);
111 }
112
113 // Update last timestamp
114 _last_perception_message_time_ = current_stamp;
115
116 this->_perception_map_.clear();
117 Eigen::Vector3d velocities(this->_vehicle_state_velocities_.velocity_x,
118 this->_vehicle_state_velocities_.velocity_y,
119 this->_vehicle_state_velocities_.rotational_velocity);
120 double perception_exec_time = msg.exec_time;
121 double theta = -velocities(2) * perception_exec_time;
122 double cos_theta = std::cos(theta);
123 double sin_theta = std::sin(theta);
124 for (auto &cone : cone_array) {
125 double x_linear_compensated = cone.position.x - velocities(0) * perception_exec_time;
126 double y_linear_compensated = cone.position.y - velocities(1) * perception_exec_time;
127 double x_compensated = cos_theta * x_linear_compensated - sin_theta * y_linear_compensated;
128 double y_compensated = sin_theta * x_linear_compensated + cos_theta * y_linear_compensated;
129 this->_perception_map_.push_back(common_lib::structures::Cone(
130 x_compensated, y_compensated, cone.color, cone.confidence, msg.header.stamp));
131 }
132
133 if (this->_slam_solver_ == nullptr) {
134 RCLCPP_WARN(this->get_logger(), "ATTR - Slam Solver object is null");
135 return;
136 }
137
138 this->_slam_solver_->add_observations(this->_perception_map_);
139 this->_track_map_ = this->_slam_solver_->get_map_estimate();
140 this->_vehicle_pose_ = this->_slam_solver_->get_pose_estimate();
141 this->_associations_ = this->_slam_solver_->get_associations();
142 this->_observations_global_ = this->_slam_solver_->get_observations_global();
143 this->_map_coordinates_ = this->_slam_solver_->get_map_coordinates();
144
145 if (this->_is_mission_finished()) {
146 this->finish();
147 }
148
149 // this->_publish_covariance(); // TODO: get covariance to work fast
150
151 this->_publish_vehicle_pose();
152 this->_publish_map();
153 this->_publish_lap_counter();
154 this->_publish_associations();
155
156 // Timekeeping
157 rclcpp::Time end_time = this->get_clock()->now();
158 this->_execution_times_->at(1) = (end_time - start_time).seconds() * 1000.0;
159
160 std_msgs::msg::Float64MultiArray execution_time_msg;
161 execution_time_msg.data = *this->_execution_times_;
162 this->_execution_time_publisher_->publish(execution_time_msg);
163}

175 time2 = this->get_clock()->now();
176
177 this->_slam_solver_->add_motion_prior(this->_vehicle_state_velocities_);
178 time3 = this->get_clock()->now();

187

189 this->_execution_times_->at(0) = (end_time - start_time).seconds() * 1000.0;
190 this->_execution_times_->at(11) = (time2 - start_time).seconds() * 1000.0;
191 this->_execution_times_->at(12) = (time3 - time2).seconds() * 1000.0;
192 this->_execution_times_->at(13) = (time4 - time3).seconds() * 1000.0;
193 this->_execution_times_->at(14) = (time5 - time4).seconds() * 1000.0;

333bool SLAMNode::_is_mission_finished() const {
334 if (this->_vehicle_state_velocities_.velocity_x > 0.1 ||
335 this->_vehicle_state_velocities_.velocity_y > 0.1 ||
336 this->_vehicle_state_velocities_.rotational_velocity > 0.05) {
337 return false;
338 }
339
340 if (this->_mission_ == common_lib::competition_logic::Mission::ACCELERATION &&
341 this->_vehicle_pose_.position.x > 75.0) {
342 return true;
343 }
344 if (this->_mission_ == common_lib::competition_logic::Mission::SKIDPAD &&
345 this->_vehicle_pose_.position.x > 22.0) {
346 return true;
347 }
348 if (this->_mission_ == common_lib::competition_logic::Mission::AUTOCROSS &&
349 this->_slam_solver_->get_lap_counter() >= 1) {
350 return true;
351 }
352 if (this->_mission_ == common_lib::competition_logic::Mission::TRACKDRIVE &&
353 this->_slam_solver_->get_lap_counter() >= 10) {
354 return true;
355 }
356 return false;
357}
