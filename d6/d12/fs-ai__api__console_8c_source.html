52int main(int argc, char** argv) {
53 if (argc < 2) {
54 printf("Too few arguments!\r\n");
55 printf("Usage: fs-ai_api_console <can>\r\n");
56 return(1);
57 }
58
59 if(fs_ai_api_init(argv[1],1,0)) { // initialise with Debug flag
60 printf("fs_ai_api_init() failed\r\n");
61 return(1);
62 }
63
64 // spawn thread
65 if(pthread_create(&loop_tid, NULL, &loop_thread, NULL) != 0) {
66 printf("Can't create loop thread...");
67 return(1);
68 }
69
70 printf("\r\n");
71 printf("Ready Player One...\r\n");
72
73 printf("\033[2J"); // clear the screen
74
75 while(1) {
76 // background loop
77 scanf("%s %d", inputs, &data);
78
79 if(0 == strcmp(inputs, "s")) {
80 sprintf(outputs, "Steer: %d \r\n", data);
81 ai2vcu_data.AI2VCU_STEER_ANGLE_REQUEST_deg = (float)data;
82 }
83
84 if(0 == strcmp(inputs, "r")) {
85 sprintf(outputs, "rpm: %d \r\n", data);
86 ai2vcu_data.AI2VCU_AXLE_SPEED_REQUEST_rpm = (float)data;
87 }
88
89 if(0 == strcmp(inputs, "b")) {
90 sprintf(outputs, "Brakes: %d \r\n", data);
91 ai2vcu_data.AI2VCU_BRAKE_PRESS_REQUEST_pct = (float)data;
92 }
93
94 if(0 == strcmp(inputs, "t")) {
95 sprintf(outputs, "Torque: %d \r\n", data);
96 ai2vcu_data.AI2VCU_AXLE_TORQUE_REQUEST_Nm = (float)data;
97 }
98
99 if(0 == strcmp(inputs, "m")) {
100 sprintf(outputs, "Mission: %d \r\n", data);
101 ai2vcu_data.AI2VCU_MISSION_STATUS = data;
102 }
103
104 if(0 == strcmp(inputs, "d")) {
105 sprintf(outputs, "Dir: %d \r\n", data);
106 ai2vcu_data.AI2VCU_DIRECTION_REQUEST = data;
107 }
108
109 if(0 == strcmp(inputs, "e")) {
110 sprintf(outputs, "EStop: %d \r\n", data);
111 ai2vcu_data.AI2VCU_ESTOP_REQUEST = data;
112 }
113
114 if(0 == strcmp(inputs, "l")) {
115 sprintf(outputs, "Loop_us: %d \r\n", data);
116 timing_us = data;
117 }
118 }
119}
