78void InspectionMission::inspection_script(const custom_interfaces::msg::WheelRPM& current_rl_rpm,
79 const custom_interfaces::msg::WheelRPM& current_rr_rpm) {
80 if (!go || mission == common_lib::competition_logic::Mission::NONE) {
81 return;
82 }
83 RCLCPP_DEBUG(this->get_logger(), "Executing Inspection Script.");
84
85 // initialization
86 auto current_time = this->clock.now();
87 auto elapsed_time = (current_time - initial_time).seconds();
88 double average_rpm = (current_rl_rpm.rl_rpm + current_rr_rpm.rr_rpm) / 2.0;
89 double current_velocity = inspection_object.rpm_to_velocity(average_rpm);
90
91 // calculate steering
92 double calculated_steering = mission == common_lib::competition_logic::Mission::INSPECTION
93 ? inspection_object.calculate_steering(elapsed_time)
94 : 0;
95
96 // if the time is over, the car should be stopped
97 if (elapsed_time >= (inspection_object.finish_time)) {
98 inspection_object.current_goal_velocity = 0;
99 }
100
101 // calculate throttle and convert to control command
102 double calculated_throttle = inspection_object.calculate_throttle(current_velocity);
103 RCLCPP_DEBUG(this->get_logger(), "Calculated throttle: %f", calculated_throttle);
104
105 // publish suitable message
106 if (elapsed_time < inspection_object.finish_time ||
107 std::abs(current_velocity) > WHEELS_STOPPED_THRESHOLD) {
108 RCLCPP_DEBUG(this->get_logger(),
109 "Publishing control command. Steering: %f; Torque: %f; Elapsed Time: %f",
110 calculated_steering, calculated_throttle, elapsed_time);
111 publish_controls(inspection_object.throttle_to_adequate_range(calculated_throttle),
112 calculated_steering);
113 } else {
114 // if the mission is over, publish the finish signal
115 RCLCPP_DEBUG(this->get_logger(), "Mission is over. Stopping the car.");
116 this->mission_end_timer = this->create_wall_timer(
117 std::chrono::milliseconds(500), std::bind(&InspectionMission::end_of_mission, this));
118 }
119
120 // update ideal velocity if necessary
121 inspection_object.redefine_goal_velocity(current_velocity);
122}
