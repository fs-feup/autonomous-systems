89void InspectionMission::inspection_script() {
90 if (!_go_ || _mission_ == common_lib::competition_logic::Mission::NONE) {
91 return;
92 }
93 RCLCPP_DEBUG(this->get_logger(), "Executing Inspection Script.");
94
95 if (this->_car_stopped_) {
96 // if the mission is over, publish the finish signal
97 RCLCPP_DEBUG(this->get_logger(), "Mission is over. Stopping the car.");
98 if (!this->_mission_end_timer_started_) {
99 this->_mission_end_timer_ = this->create_wall_timer(
100 std::chrono::milliseconds(500), std::bind(&InspectionMission::end_of_mission, this));
101 this->_mission_end_timer_started_ = true;
102 }
103 return;
104 }
105
106 // initialization
107 auto current_time = this->_clock_.now();
108 auto elapsed_time = (current_time - _initial_time_).seconds();
109 double average_rpm = _motor_rpm_ / 4;
110 RCLCPP_DEBUG(this->get_logger(), "average_rpm %f", average_rpm);
111 double current_velocity = _inspection_object_.rpm_to_velocity(average_rpm);
112 RCLCPP_DEBUG(this->get_logger(), "current_velocity %f", current_velocity);
113
114 // calculate steering
115 double calculated_steering = _mission_ == common_lib::competition_logic::Mission::INSPECTION
116 ? _inspection_object_.calculate_steering(elapsed_time)
117 : 0;
118
119 // if the time is over, the car should be stopped
120 bool steering_straight = calculated_steering < 0.1 && calculated_steering > -0.1;
121 if (elapsed_time >= (_inspection_object_.finish_time_)) {
122 _inspection_object_.current_goal_velocity_ = 0;
123 if (steering_straight) {
124 _inspection_object_.stop_oscilating_ = true;
125 }
126 }
127
128 // calculate throttle and convert to control command
129 double calculated_throttle = _inspection_object_.calculate_throttle(current_velocity);
130
131 if (elapsed_time >= _inspection_object_.finish_time_ &&
132 std::abs(current_velocity) <= WHEELS_STOPPED_THRESHOLD) {
133 calculated_throttle = 0.0;
134 }
135
136 //if (elapsed_time >= _inspection_object_.finish_time_){
138 //}
139
140 if (elapsed_time >= _inspection_object_.finish_time_ && steering_straight) {
141 calculated_steering = 0.0;
142 }
143
144 if (elapsed_time >= _inspection_object_.finish_time_ &&
145 std::abs(current_velocity) <= WHEELS_STOPPED_THRESHOLD && steering_straight) {
146 this->_car_stopped_ = true;
147 }
148
149 // publish suitable message
150 double scaled_throttle = _inspection_object_.throttle_to_adequate_range(calculated_throttle);
151 publish_controls(scaled_throttle,calculated_steering);
152 RCLCPP_DEBUG(this->get_logger(),
153 "Publishing control command. Steering: %lf; Torque after conversion - before: %lf - %lf; Elapsed Time: %lf",
154 calculated_steering, scaled_throttle, calculated_throttle, elapsed_time);
155
156 // update ideal velocity if necessary
157 _inspection_object_.redefine_goal_velocity(current_velocity);
158}
