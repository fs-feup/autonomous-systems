144 "Too few points to calculate spline while executing 'fit_spline'"
145 "Number of cones was %i",
146 static_cast<int>(n));

176 for (size_t i = 0; i < n; i++) {
177 gsl_vector_set(i_values, i, static_cast<double>(i));
178 gsl_vector_set(x_values, i, cone_seq[i].position.x);
179 gsl_vector_set(y_values, i, cone_seq[i].position.y);
180
181 // Weight points based on local spacing (closer points get larger weight)
182 double dist_next = 1.0;
183 if (i + 1 < n) {
184 dist_next = cone_seq[i].position.euclidean_distance(cone_seq[i + 1].position);
185 } else if (i > 0) {
186 dist_next = cone_seq[i].position.euclidean_distance(cone_seq[i - 1].position);
187 }
188 if (dist_next < 1e-6) dist_next = 1e-6;

190 gsl_vector_set(w, i, 1.0 / (dist_next * dist_next));
191 }

196 gsl_bspline_knots_uniform(t_min, t_max, bw);
197 gsl_bspline_knots_uniform(t_min, t_max, cw);

203 gsl_bspline_eval(i, C, cw);
204

206 for (int j = 0; j < static_cast<int>(ncoeffs); j++) {
207 gsl_matrix_set(X, i, j, gsl_vector_get(B, j));
208 gsl_matrix_set(Y, i, j, gsl_vector_get(C, j));
209 }
210 }
211
212 // -------- LEAST SQUARES FIT FOR X AND Y --------
213 gsl_multifit_wlinear(X, w, x_values, c, cov, &chisq, mw);
214 gsl_multifit_wlinear(Y, w, y_values, c2, cov2, &chisq2, mw2);
215
216 // -------- EVALUATE THE SPLINE --------
217 std::vector<T> cone_seq_eval;
218 cone_seq_eval.reserve(n * precision);
219 for (int i = 0; i < static_cast<int>(n); i++) {
220 for (int j = 0; j < precision; j++) {

223 if (t > t_max) t = t_max;

225 gsl_bspline_eval(t, B, bw);
226 gsl_bspline_eval(t, C, cw);

231 gsl_multifit_linear_est(C, c2, cov2, &yi, &err);

234 T new_element;
235 new_element.position.x = xi;
236 new_element.position.y = yi;
237 cone_seq_eval.push_back(new_element);
238 }
239 }
