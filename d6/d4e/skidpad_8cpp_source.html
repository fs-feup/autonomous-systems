19 Eigen::Matrix4f result = Eigen::Matrix4f::Identity();
20
21 // Check if we have enough cones for ICP
22 if (reference_cones_.size() >= static_cast<std::size_t>(config_.minimum_cones_) &&
23 cone_array.size() >= static_cast<std::size_t>(config_.minimum_cones_)) {
24
25 // Convert cone_array (LiDAR) to source cloud
26 pcl::PointCloud<pcl::PointXYZ> cloud_source;
27 cloud_source.reserve(cone_array.size());
28 for (const auto& cone : cone_array) {
29 (void)cloud_source.emplace_back(cone.position.x, cone.position.y, 0.0);
30 }
31
32 // Convert reference cones to target cloud
33 pcl::PointCloud<pcl::PointXYZ> cloud_target;
34 cloud_target.reserve(reference_cones_.size());
35 for (const auto& [x, y] : reference_cones_) {
36 (void)cloud_target.emplace_back(x, y, 0.0);
37 }
38
39 // Run ICP alignment
40 pcl::IterativeClosestPoint<pcl::PointXYZ, pcl::PointXYZ> icp;
41 icp.setInputSource(cloud_source.makeShared());
42 icp.setInputTarget(cloud_target.makeShared());
43 icp.setMaxCorrespondenceDistance(config_.tolerance_);
44 icp.setMaximumIterations(std::numeric_limits<int>::max());
45 icp.setTransformationEpsilon(1e-6);
46 icp.setEuclideanFitnessEpsilon(1e-3);
47
48 pcl::PointCloud<pcl::PointXYZ> Final;
49 icp.align(Final);
50
51 if (icp.hasConverged()) {
52 result = icp.getFinalTransformation();

58 }

63size_t Skidpad::find_closest_path_index(
64 const std::vector<PathPoint>& path,
65 const common_lib::structures::Pose& pose) {
66
67 double min_dist = std::numeric_limits<double>::max();
68 size_t closest_index = 0;
69
70 for (size_t i = 0; i < path.size(); ++i) {
71 double dx = path[i].position.x - pose.position.x;
72 double dy = path[i].position.y - pose.position.y;
73 double dist = std::sqrt(dx * dx + dy * dy);
74
75 if (dist > min_dist) {
76 break; // Stop searching if distance starts increasing
77 }
78
79 if (dist < min_dist) {
80 min_dist = dist;
81 closest_index = i;
82 }
83 }
84
85 return closest_index;
86}
