44void EKFSLAMSolver::add_observations(const std::vector<common_lib::structures::Cone>& cones) {
45 int num_observations = static_cast<int>(cones.size());
46 std::vector<int> matched_landmarks_indices;
47 Eigen::VectorXd matched_observations(0);
48 Eigen::VectorXd new_landmarks(0);
49 Eigen::VectorXd new_confidences(0);
50 Eigen::VectorXd observations(2 * num_observations);
51 Eigen::VectorXd observation_confidences(num_observations);
52 common_lib::conversions::cone_vector_to_eigen(cones, observations, observation_confidences);
53 Eigen::VectorXd global_observations =
54 common_lib::maths::local_to_global_coordinates(this->state_.segment(0, 3), observations);
55 Eigen::VectorXi associations = this->_data_association_->associate(
56 this->state_.segment(3, this->state_.size() - 3), global_observations,
57 this->covariance_.block(3, 3, this->state_.size() - 3, this->state_.size() - 3),
58 observation_confidences);
59 for (int i = 0; i < num_observations; ++i) {
60 if (associations(i) == -2) {
61 continue;
62 } else if (associations(i) == -1) {
63 new_landmarks.conservativeResize(new_landmarks.size() + 2);
64 new_landmarks(new_landmarks.size() - 2) = global_observations(2 * i);
65 new_landmarks(new_landmarks.size() - 1) = global_observations(2 * i + 1);
66 new_confidences.conservativeResize(new_confidences.size() + 1);
67 new_confidences(new_confidences.size() - 1) = observation_confidences(i);
68 } else {
69 matched_landmarks_indices.push_back(associations(i) + 3);
70 matched_observations.conservativeResize(matched_observations.size() + 2);
71 matched_observations(matched_observations.size() - 2) = observations(2 * i);
72 matched_observations(matched_observations.size() - 1) = observations(2 * i + 1);
73 }
74 }
75 Eigen::VectorXd filtered_new_landmarks =
76 this->_landmark_filter_->filter(new_landmarks, new_confidences);
77 this->_landmark_filter_->delete_landmarks(filtered_new_landmarks);
78 this->correct(this->state_, this->covariance_, matched_landmarks_indices, matched_observations);
79 if (this->_mission_ != common_lib::competition_logic::Mission::NONE &&
80 !(this->_params_.using_preloaded_map_ &&
81 (this->_mission_ == common_lib::competition_logic::Mission::SKIDPAD ||
82 this->_mission_ == common_lib::competition_logic::Mission::ACCELERATION)) &&
83 this->lap_counter_ == 0) {
84 this->state_augmentation(this->state_, this->covariance_, filtered_new_landmarks);
85 this->update_process_noise_matrix();
86 }
87}

89void EKFSLAMSolver::predict(Eigen::VectorXd& state, Eigen::MatrixXd& covariance,
90 const Eigen::MatrixXd& process_noise_matrix,
91 const rclcpp::Time last_update,
92 const common_lib::structures::Velocities& velocities) {
93 auto time_interval = velocities.timestamp_.seconds() - last_update.seconds();
94
95 Eigen::Vector3d previous_pose = state.segment(0, 3);
96 Eigen::Vector3d temp_velocities(velocities.velocity_x, velocities.velocity_y,
97 velocities.rotational_velocity);
98
99 Eigen::MatrixXd pose_jacobian =
100 this->_motion_model_->get_jacobian_pose(previous_pose, temp_velocities, time_interval);
101 int desired_jacobian_size = covariance.cols();
102 Eigen::MatrixXd jacobian =
103 Eigen::MatrixXd::Identity(desired_jacobian_size, desired_jacobian_size);
104 jacobian.block(0, 0, 3, 3) = pose_jacobian;
105
106 Eigen::Vector3d next_pose =
107 this->_motion_model_->get_next_pose(previous_pose, temp_velocities, time_interval);
108 state.segment(0, 3) = next_pose;
109 this->pose = next_pose;
110 covariance = jacobian * covariance * jacobian.transpose() + process_noise_matrix;
111}

113void EKFSLAMSolver::correct(Eigen::VectorXd& state, Eigen::MatrixXd& covariance,
114 const std::vector<int>& matched_landmarks_indices,
115 const Eigen::VectorXd& matched_observations) {
116 Eigen::VectorXd predicted_observations =
117 this->observation_model_->observation_model(state, matched_landmarks_indices);
118 Eigen::MatrixXd jacobian =
119 this->observation_model_->observation_model_jacobian(state, matched_landmarks_indices);
120 Eigen::MatrixXd observation_noise_matrix =
121 get_observation_noise_matrix(matched_landmarks_indices.size());
122 Eigen::MatrixXd kalman_gain =
123 covariance * jacobian.transpose() *
124 (jacobian * covariance * jacobian.transpose() + observation_noise_matrix).inverse();
125 state += kalman_gain * (matched_observations - predicted_observations);
126 covariance =
127 (Eigen::MatrixXd::Identity(state.size(), state.size()) - kalman_gain * jacobian) * covariance;
128}

130void EKFSLAMSolver::state_augmentation(Eigen::VectorXd& state, Eigen::MatrixXd& covariance,
131 const Eigen::VectorXd& new_landmarks_coordinates) {
132 // Resize covariance matrix
133 int num_new_entries = static_cast<int>(new_landmarks_coordinates.size());
134 int covariance_size = static_cast<int>(covariance.rows());
135
136 covariance.conservativeResizeLike(
137 Eigen::MatrixXd::Zero(covariance_size + num_new_entries, covariance_size + num_new_entries));
138 for (int i = 0; i < num_new_entries / 2; i++) {
139 covariance(2 * i + covariance_size, 2 * i + covariance_size) =
140 this->_params_.observation_x_noise_;
141 covariance(2 * i + 1 + covariance_size, 2 * i + 1 + covariance_size) =
142 _params_.observation_y_noise_;
143 }
144 // Resize state vector
145 int original_state_size = static_cast<int>(state.size());
146 state.conservativeResizeLike(Eigen::VectorXd::Zero(original_state_size + num_new_entries));
147 state.segment(original_state_size, num_new_entries) = new_landmarks_coordinates;
148}

150void EKFSLAMSolver::load_initial_state(const Eigen::VectorXd& map, const Eigen::VectorXd& pose) {
151 if (map.size() % 2 != 0 || pose.size() != 3) {
152 throw std::runtime_error("Invalid map or pose size");
153 }
154 this->state_ = Eigen::VectorXd::Zero(3 + map.size());
155 this->state_.segment(0, 3) = pose;
156 this->state_.segment(3, map.size()) = map;
157 this->covariance_ = Eigen::MatrixXd::Identity(this->state_.size(), this->state_.size()) *
158 this->_params_.preloaded_map_noise_;
159 this->covariance_.block(0, 0, 3, 3) = Eigen::MatrixXd::Zero(3, 3);
160 this->covariance_(0, 0) = this->slam_parameters_.pose_x_initial_noise_;
161 this->covariance_(1, 1) = this->slam_parameters_.pose_y_initial_noise_;
162 this->covariance_(2, 2) = this->slam_parameters_.pose_theta_initial_noise_;
163 update_process_noise_matrix();
164}
