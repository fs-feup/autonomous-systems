263void Perception::point_cloud_callback(const sensor_msgs::msg::PointCloud2::SharedPtr msg) {
264 rclcpp::Time time = this->now();
265
266 this->lidar_off_timer_ = this->create_wall_timer(
267 std::chrono::milliseconds(2000), std::bind(&Perception::lidar_timer_callback, this));
268
269 // Message Read
270 pcl::PointCloud<pcl::PointXYZI>::Ptr pcl_cloud(new pcl::PointCloud<pcl::PointXYZI>);
271 header = (*msg).header;
272 pcl::fromROSMsg(*msg, *pcl_cloud);
273
274 // Pass-trough Filter (trim Pcl)
275 const SplitParameters split_params = _fov_trim_map_->at(_mission_type_)->fov_trimming(pcl_cloud);
276
277 // Ground Removal
278 pcl::PointCloud<pcl::PointXYZI>::Ptr ground_removed_cloud(new pcl::PointCloud<pcl::PointXYZI>);
279 _ground_removal_->ground_removal(pcl_cloud, ground_removed_cloud, _ground_plane_, split_params);
280
281 this->_deskew_->deskew_point_cloud(ground_removed_cloud, this->_vehicle_velocity_);
282
283 // Debugging utils -> Useful to check the ground removed point cloud
284 sensor_msgs::msg::PointCloud2 ground_removed_msg;
285 pcl::toROSMsg(*ground_removed_cloud, ground_removed_msg);
286 ground_removed_msg.header.frame_id = _vehicle_frame_id_;
287 this->_ground_removed_publisher_->publish(ground_removed_msg);
288
289 // Clustering
290 std::vector<Cluster> clusters;
291 _clustering_->clustering(ground_removed_cloud, &clusters);
292
293 // Z-scores calculation for future validations
294 Cluster::set_z_scores(clusters);
295
296 // Filtering
297 std::vector<Cluster> filtered_clusters;
298
299 for (auto& cluster : clusters) {
300 if (_cone_evaluator_->evaluateCluster(cluster, _ground_plane_)) {
301 filtered_clusters.push_back(cluster);
302 }
303 }
304
305 // Execution Time calculation
306 rclcpp::Time end_time = this->now();
307 std_msgs::msg::Float64 perception_execution_time;
308 double perception_execution_time_seconds = (end_time - time).seconds();
309 perception_execution_time.data = perception_execution_time_seconds * 1000;
310 this->_perception_execution_time_publisher_->publish(perception_execution_time);
311
312 // Logging
313 RCLCPP_DEBUG(this->get_logger(), "---------- Point Cloud Received ----------");
314 RCLCPP_DEBUG(this->get_logger(), "Point Cloud Before Ground Removal: %ld points",
315 pcl_cloud->points.size());
316 RCLCPP_DEBUG(this->get_logger(), "Point Cloud After Ground Removal: %ld points",
317 ground_removed_cloud->points.size());
318 RCLCPP_DEBUG(this->get_logger(), "Point Cloud after Clustering: %ld clusters", clusters.size());
319 RCLCPP_DEBUG(this->get_logger(), "Point Cloud after Validations: %ld clusters",
320 filtered_clusters.size());
321
322 publish_cones(&filtered_clusters, perception_execution_time_seconds);
323}
