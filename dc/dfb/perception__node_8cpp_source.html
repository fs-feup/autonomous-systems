245void Perception::point_cloud_callback(const sensor_msgs::msg::PointCloud2::SharedPtr msg) {
246 this->lidar_off_timer_ = this->create_wall_timer(
247 std::chrono::milliseconds(2000), std::bind(&Perception::lidar_timer_callback, this));
248
249 rclcpp::Time start_time = this->now();
250 rclcpp::Time time1, time2, time3, time4, time5, time6;
251
252 // FOV Trimming
253 auto trimmed_cloud = std::make_shared<sensor_msgs::msg::PointCloud2>();
254 _fov_trim_map_->at(_mission_type_)->fov_trimming(msg, trimmed_cloud);
255
256 time1 = this->now();
257
258 // Ground Removal
259 auto ground_removed_cloud = std::make_shared<sensor_msgs::msg::PointCloud2>();
260 _ground_removal_->ground_removal(trimmed_cloud, ground_removed_cloud, *this->_ground_grid_);
261
262 // Publish ground removed cloud for visualization
263 ground_removed_cloud->header.frame_id = _vehicle_frame_id_;
264 this->_ground_removed_publisher_->publish(*ground_removed_cloud);
265
266 time2 = this->now();
267
268 // Wall Removal
269 auto wall_removed_cloud = std::make_shared<sensor_msgs::msg::PointCloud2>();
270 _wall_removal_->remove_walls(ground_removed_cloud, wall_removed_cloud);
271
272 // Publish wall removed cloud for visualization
273 wall_removed_cloud->header.frame_id = _vehicle_frame_id_;
274 this->_wall_removed_publisher_->publish(*wall_removed_cloud);
275
276 time3 = this->now();
277
278 // Deskewing
279 this->_deskew_->deskew_point_cloud(wall_removed_cloud, this->_vehicle_velocity_);
280
281 time4 = this->now();
282
283 // Clustering
284 std::vector<Cluster> clusters;
285 _clustering_->clustering(wall_removed_cloud, &clusters);
286
287 time5 = this->now();
288
289 // Evaluator
290 std::vector<Cluster> filtered_clusters;
291
292 for (auto& cluster : clusters) {
293 if (_cone_evaluator_->evaluateCluster(cluster, *this->_ground_grid_)) {
294 filtered_clusters.push_back(cluster);
295 }
296 }
297
298 time6 = this->now();
299
300 double perception_execution_time_seconds = (time6 - start_time).seconds();
301 this->_execution_times_->at(0) = perception_execution_time_seconds * 1000.0;
302 this->_execution_times_->at(1) = (time1 - start_time).seconds() * 1000.0;
303 this->_execution_times_->at(2) = (time2 - time1).seconds() * 1000.0;
304 this->_execution_times_->at(3) = (time3 - time2).seconds() * 1000.0;
305 this->_execution_times_->at(4) = (time4 - time3).seconds() * 1000.0;
306 this->_execution_times_->at(5) = (time5 - time4).seconds() * 1000.0;
307 std_msgs::msg::Float64MultiArray exec_time_msg;
308 exec_time_msg.data = *(this->_execution_times_);
309 this->_perception_execution_time_publisher_->publish(exec_time_msg);
310
311 publish_cones(&filtered_clusters, perception_execution_time_seconds, start_time);
312}

314void Perception::publish_cones(std::vector<Cluster>* cones, double exec_time, rclcpp::Time start_time) {
315 auto message = custom_interfaces::msg::PerceptionOutput();
316 std::vector<custom_interfaces::msg::Cone> message_array = {};
317 message.header = ::header;
318 message.header.stamp = start_time;

322 position.y = cones->at(i).get_centroid().y();
323

326 cone_message.color = cones->at(i).get_color();
327 cone_message.is_large = cones->at(i).get_is_large();
328 cone_message.confidence = cones->at(i).get_confidence();
329 message.cones.cone_array.push_back(cone_message);
330 message_array.push_back(cone_message);
331 }
332 message.exec_time = exec_time;
333
334 this->_cones_publisher->publish(message);
335 // TODO: correct frame id to LiDAR instead of vehicle
