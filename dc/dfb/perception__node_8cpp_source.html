103void Perception::point_cloud_callback(const sensor_msgs::msg::PointCloud2::SharedPtr msg) {
104 rclcpp::Time time = this->now();
105
106 // Message Read
107 pcl::PointCloud<pcl::PointXYZI>::Ptr pcl_cloud(new pcl::PointCloud<pcl::PointXYZI>);
108 header = (*msg).header;
109 pcl::fromROSMsg(*msg, *pcl_cloud);
110
111 // Pass-trough Filter
112 fov_trimming(pcl_cloud, this->_pc_max_range_, -_fov_trim_, _fov_trim_, 5);
113
114 // Ground Removal
115 pcl::PointCloud<pcl::PointXYZI>::Ptr ground_removed_cloud(new pcl::PointCloud<pcl::PointXYZI>);
116 _ground_removal_->ground_removal(pcl_cloud, ground_removed_cloud, _ground_plane_);
117
118 // Debugging utils -> Useful to check the ground removed point cloud
119 sensor_msgs::msg::PointCloud2 ground_remved_msg;
120 pcl::toROSMsg(*ground_removed_cloud, ground_remved_msg);
121 ground_remved_msg.header.frame_id = _vehicle_frame_id_;
122 this->_ground_removed_publisher_->publish(ground_remved_msg);
123
124 // Clustering
125 std::vector<Cluster> clusters;
126 _clustering_->clustering(ground_removed_cloud, &clusters);
127
128
129 // Z-scores calculation for future validations
130 Cluster::set_z_scores(clusters);
131
132 // Filtering
133 std::vector<Cluster> filtered_clusters;
134
135 for (auto cluster : clusters) {
136 bool is_valid = true;
137 for (auto validator : _cone_validators_){
138 is_valid = is_valid && validator->coneValidator(&cluster, _ground_plane_);
139
140 // Break the cycle to avoid wasting time on invalid clusters
141 if (!is_valid) break;
142 }
143 if (is_valid) {
144 filtered_clusters.push_back(cluster);
145 }
146 }
147
148 // Execution Time calculation
149 rclcpp::Time end_time = this->now();
150 std_msgs::msg::Float64 perception_execution_time;
151 perception_execution_time.data = (end_time - time).seconds() * 1000;
152 this->_perception_execution_time_publisher_->publish(perception_execution_time);
153
154 // Logging
155 RCLCPP_DEBUG(this->get_logger(), "---------- Point Cloud Received ----------");
156 RCLCPP_DEBUG(this->get_logger(), "Point Cloud Before Ground Removal: %ld points",
157 pcl_cloud->points.size());
158 RCLCPP_DEBUG(this->get_logger(), "Point Cloud After Ground Removal: %ld points",
159 ground_removed_cloud->points.size());
160 RCLCPP_DEBUG(this->get_logger(), "Point Cloud after Clustering: %ld clusters",
161 clusters.size());
162 RCLCPP_DEBUG(this->get_logger(), "Point Cloud after Validations: %ld clusters",
163 filtered_clusters.size());
164
165
166 publish_cones(&filtered_clusters);
167}

192void Perception::fov_trimming(pcl::PointCloud<pcl::PointXYZI>::Ptr cloud, double max_distance,
193 double min_angle, double max_angle, double x_discount) {
194 pcl::PointCloud<pcl::PointXYZI>::Ptr trimmed_cloud(new pcl::PointCloud<pcl::PointXYZI>);
195
196 double center_x = -x_discount; // assuming (0, 0) as the center
197
198 for (const auto& point : cloud->points) {
199 // Calculate distance from the origin (assuming the sensor is at the origin)
200 double distance = std::sqrt((point.x - center_x) * (point.x - center_x) + point.y * point.y);
201
202 // Calculate the angle in the XY plane
203 double angle =
204 std::atan2(point.y, point.x - center_x) * 180 / M_PI; // get angle and convert in to degrees
205
206 if (distance <= 0.5) { // Ignore points from the vehicle
207 continue;
208 }
209
210 // Check if the point is within the specified distance and angle range
211 if (distance <= max_distance && angle >= min_angle && angle <= max_angle) {
212 trimmed_cloud->points.push_back(point);
213 }
214 }
215
216 // Replace the input cloud with the trimmed cloud
217 *cloud = *trimmed_cloud;
218}
