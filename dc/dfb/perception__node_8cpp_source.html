117void Perception::fov_trimming(pcl::PointCloud<pcl::PointXYZI>::Ptr cloud, double max_distance, double min_angle, double max_angle) {
118 pcl::PointCloud<pcl::PointXYZI>::Ptr trimmed_cloud(new pcl::PointCloud<pcl::PointXYZI>);
119
120 for (const auto& point : cloud->points) {
121 // Calculate distance from the origin (assuming the sensor is at the origin)
122 double distance = std::sqrt(point.x * point.x + point.y * point.y);
123
124 // Calculate the angle in the XY plane
125 double angle = std::atan2(point.y, point.x) * 180 / M_PI; // get angle and convert in to degrees
126
127 // Check if the point is within the specified distance and angle range
128 if (distance <= max_distance && angle >= min_angle && angle <= max_angle) {
129 trimmed_cloud->points.push_back(point);
130 }
131 }
132
133 // Replace the input cloud with the trimmed cloud
134 *cloud = *trimmed_cloud;
135}
