84void Planning::run_planning_algorithms() {
85 auto track = new Track(this->outliers_spline_order_, this->outliers_spline_coeffs_ratio_,
86 this->outliers_spline_precision_);
87 auto cone_coloring = std::make_shared<ConeColoring>(this->angle_gain_, this->distance_gain_,
88 this->ncones_gain_, this->angle_exponent_,
89 this->distance_exponent_, this->cost_max_);
90 auto path_smoother = std::make_shared<PathSmoothing>(this->smoothing_spline_order_,
91 this->smoothing_spline_coeffs_ratio_,
92 this->smoothing_spline_precision_);
93
94 if (this->cone_array_.empty()) {
95 publish_track_points({});
96 return;
97 }
98
99 cone_coloring->color_cones(this->cone_array_, this->pose, 5);
100
101 for (auto &cone : cone_coloring->current_left_cones) {
102 track->add_cone_left(cone);
103 }
104 for (auto &cone : cone_coloring->current_right_cones) {
105 track->add_cone_right(cone);
106 }
107
108 track->validateCones(); // Deal with cone outliers
109
110 std::vector<PathPoint *> triangulations_path =
111 local_path_planner->processNewArray(track); // Calculate Path
112
113 auto path = triangulations_path;
114 path_smoother->defaultSmoother(path);
115
116 RCLCPP_DEBUG(this->get_logger(), "Planning published %i path points\n", (int)path.size());
117 publish_track_points(path);
118
119 if (this->publishing_visualization_msgs_) {
120 publish_visualization_msgs(cone_coloring->current_left_cones,
121 cone_coloring->current_right_cones, triangulations_path, path,
122 track->get_left_cones(), track->get_right_cones());
123 }
124}
