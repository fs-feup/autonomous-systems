191std::vector<Cone *> Track::orderCones(std::vector<Cone *> *unord_cone_seq) {
192 // Order cone_array
193 // The algorithm works by iteratively selecting the nearest unvisited
194 // cone to the current cone and updating the traversal direction accordingly.
195 // This approach creates an ordered sequence that efficiently connects nearby
196 // cones.
197 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "START orderCones");
198 std::vector<std::pair<Cone *, bool>> nn_unord_cone_seq;
199 for (int i = 0; i < static_cast<int>(unord_cone_seq->size()); i++)
200 nn_unord_cone_seq.push_back(std::make_pair((*unord_cone_seq)[i], false));
201
202 // Values for first iteration
203 std::vector<Cone *> cone_seq;
204 Cone *cone1 = new Cone(-1, 0, 0);
205 float vx = 1;
206 float vy = 0;
207
208 for (int iter_number = 0; iter_number < static_cast<int>(unord_cone_seq->size()); iter_number++) {
209 float min_dist = MAXFLOAT;
210 int min_index = 0;
211
212 for (int i = 0; i < static_cast<int>(nn_unord_cone_seq.size()); i++) {
213 Cone *cone2 = nn_unord_cone_seq[i].first;
214 // Check only if not visited
215 if (nn_unord_cone_seq[i].second == false ||
216 (iter_number == 0 && vector_direction(cone1, cone2, vx, vy))) {
217 // first iteration we assure the direction is correct to avoid going
218 // backwards
219
220 float new_dist = cone1->getDistanceTo(cone2);
221 if (new_dist < min_dist) {
222 min_dist = new_dist;
223 min_index = i;
224 }
225 }
226 }
227 nn_unord_cone_seq[min_index].second = true; // mark as visited
228
229 // new visited cones is the reference for the next search. Arrays and Cone
230 // updated
231 vx = nn_unord_cone_seq[min_index].first->getX() - cone1->getX();
232 vy = nn_unord_cone_seq[min_index].first->getY() - cone1->getY();
233 cone1 = nn_unord_cone_seq[min_index].first;
234
235 cone_seq.push_back(nn_unord_cone_seq[min_index].first); // add cone to ordered sequence
236 }
237 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "END orderCones");
238 return cone_seq;
239}

241std::vector<Cone *> Track::fitSpline(bool side, int precision, int order, float coeffs_ratio,
242 std::vector<Cone *> cone_seq) {
243 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "START fitSpline with %i cones",
244 static_cast<int>(cone_seq.size()));
245
246 std::unordered_set<Cone *> seen;
247
248 // Use std::remove_if and a lambda function to remove duplicates
249 auto iterator = std::remove_if(cone_seq.begin(), cone_seq.end(), [&seen](const auto &ptr) {
250 return !seen.insert(ptr).second; // insert returns a pair, where .second is false if the
251 // element already existed
252 });
253
254 // Finally, erase the duplicates
255 cone_seq.erase(iterator, cone_seq.end());
256
257 // START ELIMINATING OUTLIERS BASED ON DISTANCE TO SPLINE
258 const int n = cone_seq.size();
259 const int ncoeffs = n / coeffs_ratio; // n > = ncoeffs
260 const int nbreak = ncoeffs - order + 2;
261
262 if (nbreak < 2 || n == 0) {
263 RCLCPP_WARN(rclcpp::get_logger("rclcpp"),
264 "Too few points to calculate spline while executing 'deleteOutliers' on side %i "
265 "Number of cones was %i",
266 side, n);
267 return cone_seq;
268 }
269
270 RCLCPP_DEBUG(
271 rclcpp::get_logger("rclcpp"),
272 "nbreak= %i; n (number of cones) = %i; ncoeffs = %i; left cones: %i; right cones: %i\n",
273 nbreak, n, ncoeffs, static_cast<int>(leftCones.size()), static_cast<int>(rightCones.size()));
274
275 // Initialize vars (pointers)
276 gsl_bspline_workspace *bw, *cw;
277 gsl_vector *B, *C;
278 gsl_vector *c, *c2, *w;
279 gsl_vector *x_values, *y_values, *i_values;
280 gsl_matrix *X, *Y, *cov, *cov2;
281 gsl_multifit_linear_workspace *mw, *mw2;
282 double chisq, chisq2;
283
284 // allocate memory for the actual objects the pointers will point to
285 bw = gsl_bspline_alloc(order, nbreak);
286 cw = gsl_bspline_alloc(order, nbreak);
287 B = gsl_vector_alloc(ncoeffs);
288 C = gsl_vector_alloc(ncoeffs);
289
290 i_values = gsl_vector_alloc(n);
291 x_values = gsl_vector_alloc(n);
292 y_values = gsl_vector_alloc(n);
293
294 X = gsl_matrix_alloc(n, ncoeffs);
295 Y = gsl_matrix_alloc(n, ncoeffs);
296 c = gsl_vector_alloc(ncoeffs);
297 c2 = gsl_vector_alloc(ncoeffs);
298 w = gsl_vector_alloc(n);
299 cov = gsl_matrix_alloc(ncoeffs, ncoeffs);
300 cov2 = gsl_matrix_alloc(ncoeffs, ncoeffs);
301 mw = gsl_multifit_linear_alloc(n, ncoeffs);
302 mw2 = gsl_multifit_linear_alloc(n, ncoeffs);
303
304 // Set spline data
305 for (int i = 0; i < n; i++) {
306 gsl_vector_set(i_values, i, i);
307 gsl_vector_set(x_values, i, cone_seq[i]->getX());
308 gsl_vector_set(y_values, i, cone_seq[i]->getY());
309 // closer cones more important(better stability)
310 gsl_vector_set(w, i, 1.0 / pow(cone_seq[i]->getDistanceTo(cone_seq[(i + 1) % n]), 2));
311 }
312
313 // Set i range within cone set length
314 gsl_bspline_knots_uniform(0, n, bw);
315 gsl_bspline_knots_uniform(0, n, cw);
316
317 /* construct the fit matrix X */
318 for (int i = 0; i < n; i++) {
319 /* compute B_j(xi) for all j */
320 gsl_bspline_eval(i, B, bw);
321 gsl_bspline_eval(i, C, cw);
322
323 /* fill in row i of X */
324 for (int j = 0; j < ncoeffs; j++) {
325 double Bj = gsl_vector_get(B, j);
326 gsl_matrix_set(X, i, j, Bj);
327 double Cj = gsl_vector_get(C, j);
328 gsl_matrix_set(Y, i, j, Cj);
329 }
330 }
331
332 // Fit spline
333 gsl_multifit_wlinear(X, w, x_values, c, cov, &chisq, mw);
334 gsl_multifit_wlinear(Y, w, y_values, c2, cov2, &chisq2, mw2);
335
336 // Log spline coefficients held in gsl_vectors c and c2
337 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Base functions' weights in x: %s\n",
338 gsl_vectorLog(c, ncoeffs).c_str());
339 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "Base functions' weights in y: %s\n",
340 gsl_vectorLog(c2, ncoeffs).c_str());
341
342 // Initialize variables for subsequent spline evaluation
343 double xi, yi, yerr, yerr2;
344 std::vector<double> i_eval, x_eval, y_eval;
345 std::vector<Cone *> cone_seq_eval;
346
347 // Calculate the desired amount of spline points and add them to
348 // "cone_seq_eval"
349 for (int i = 0; i < n; i++) {
350 for (int j = 0; j < precision; j += 1) { // iterate over decimal numbers
351 gsl_bspline_eval(i + static_cast<float>(j) / precision, B, bw);
352 gsl_bspline_eval(i + static_cast<float>(j) / precision, C, cw);
353 gsl_multifit_linear_est(B, c, cov, &xi, &yerr);
354 gsl_multifit_linear_est(C, c2, cov2, &yi, &yerr2);
355 i_eval.push_back(i);
356 x_eval.push_back(xi);
357 y_eval.push_back(yi);
358 int id = static_cast<int>(2 * cone_seq_eval.size()) + 1 - side;
359 Cone *cone_new = new Cone(id, xi, yi);
360 cone_seq_eval.push_back(cone_new);
361 if (j == 0 && i == n - 1) {
362 break; // Decimals can't go over last int
363 }
364 }
365 }
366
367 // Free Memory
368 gsl_bspline_free(bw);
369 gsl_bspline_free(cw);
370 gsl_vector_free(B);
371 gsl_vector_free(C);
372 gsl_vector_free(i_values);
373 gsl_vector_free(x_values);
374 gsl_vector_free(y_values);
375 gsl_matrix_free(X);
376 gsl_matrix_free(Y);
377 gsl_vector_free(c);
378 gsl_vector_free(w);
379 gsl_matrix_free(cov);
380 gsl_multifit_linear_free(mw);
381 gsl_multifit_linear_free(mw2);
382
383 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "END fitSpline with %i points",
384 static_cast<int>(cone_seq_eval.size()));
385
386 // To access spline points uncomment these lines
387 // std::cout << "track spline : " << std::endl;
388 // for (int i = 0; i < static_cast<int>(cone_seq_eval.size()); i++) {
389 // std::cout << "(" << cone_seq_eval[i]->getX() << "," << cone_seq_eval[i]->getY() << "),";
390 // }
391 // std::cout << std::endl;
392
393 return cone_seq_eval;
394}
