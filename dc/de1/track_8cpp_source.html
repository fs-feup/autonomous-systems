37void Track::addCone(float xValue, float yValue, const std::string &color) {
38 if (color == colors::color_names[colors::blue]) {
39 // Blue Cones always have an even number id (2*x)
40 rightCones.push_back(new Cone(this->rightCount * 2, xValue, yValue));
41 rightCount++;
42 } else if (color == colors::color_names[colors::yellow]) {
43 // Yellow Cones always have an odd number id (2*x + 1)
44 leftCones.push_back(new Cone(this->leftCount * 2 + 1, xValue, yValue));
45 leftCount++;
46 } else if (color != colors::color_names[colors::orange] &&
47 color != colors::color_names[colors::large_orange]) {
48 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "Invalid color: %s\n", color.c_str());
49 }
50}

101bool Track::vector_direction(Cone *c1, Cone *c2, float prev_vx, float prev_vy) {
102 float vx = c2->getX() - c1->getX();
103 float vy = c2->getY() - c1->getY();
104 // Cos(angle) higher than 0 yhe vectors are pointing to the same half
105 return (vx * prev_vx + vy * prev_vy) > 0;

117int Track::deleteOutliers(bool side, float distance_threshold, int order, float coeffs_ratio,
118 bool writing) {
119 std::vector<Cone *> &unord_cone_seq = side ? leftCones : rightCones;
120 // if side = 1(left) | = 0(right)
121
122 const int n = unord_cone_seq.size();
123 const int ncoeffs = n / coeffs_ratio; // n > = ncoeffs
124 const int nbreak = ncoeffs - order + 2;
125
126 if (nbreak < 2) {
127 RCLCPP_ERROR(rclcpp::get_logger("rclcpp"), "Too few points to calculate spline\n");
128 return 0;
129 }
130
131 // Initialize vars (pointers)
132 gsl_bspline_workspace *bw, *cw;
133 gsl_vector *B, *C;
134 gsl_vector *c, *c2, *w;
135 gsl_vector *x_values, *y_values, *i_values;
136 gsl_matrix *X, *Y, *cov, *cov2;
137 gsl_multifit_linear_workspace *mw, *mw2;
138 double chisq, chisq2;
139
140 // allocate memory for the actual objects the pointers will point to
141 bw = gsl_bspline_alloc(order, nbreak);
142 cw = gsl_bspline_alloc(order, nbreak);
143 B = gsl_vector_alloc(ncoeffs);
144 C = gsl_vector_alloc(ncoeffs);
145
146 i_values = gsl_vector_alloc(n);
147 x_values = gsl_vector_alloc(n);
148 y_values = gsl_vector_alloc(n);
149
150 X = gsl_matrix_alloc(n, ncoeffs);
151 Y = gsl_matrix_alloc(n, ncoeffs);
152 c = gsl_vector_alloc(ncoeffs);
153 c2 = gsl_vector_alloc(ncoeffs);
154 w = gsl_vector_alloc(n);
155 cov = gsl_matrix_alloc(ncoeffs, ncoeffs);
156 cov2 = gsl_matrix_alloc(ncoeffs, ncoeffs);
157 mw = gsl_multifit_linear_alloc(n, ncoeffs);
158 mw2 = gsl_multifit_linear_alloc(n, ncoeffs);
159
160 // Order cone_array
161 // The algorithm works by iteratively selecting the nearest unvisited
162 // cone to the current cone and updating the traversal direction accordingly.
163 // This approach creates an ordered sequence that efficiently connects nearby
164 // cones.
165 std::vector<std::pair<Cone *, bool>> nn_unord_cone_seq;
166 for (int i = 0; i < static_cast<int>(unord_cone_seq.size()); i++)
167 nn_unord_cone_seq.push_back(std::make_pair(unord_cone_seq[i], false));
168
169 // Values for first iteration
170 std::vector<Cone *> cone_seq;
171 Cone *cone1 = new Cone(-1, 0, 0);
172 float vx = 1;
173 float vy = 0;
174
175 for (int iter_number = 0; iter_number < static_cast<int>(unord_cone_seq.size()); iter_number++) {
176 float min_dist = MAXFLOAT;
177 int min_index = 0;
178
179 for (int i = 0; i < static_cast<int>(nn_unord_cone_seq.size()); i++) {
180 Cone *cone2 = nn_unord_cone_seq[i].first;
181 // Check only if not visited
182 if (nn_unord_cone_seq[i].second == false ||
183 (iter_number == 0 && vector_direction(cone1, cone2, vx, vy))) {
184 // first iteration we assure the direction is correct to avoid going
185 // backwards
186
187 float new_dist = cone1->getDistanceTo(cone2);
188 if (new_dist < min_dist) {
189 min_dist = new_dist;
190 min_index = i;
191 }
192 }
193 }
194 nn_unord_cone_seq[min_index].second = true; // mark as visited
195
196 // new visited cones is the reference for the next search. Arrays and Cone
197 // updated
198 vx = nn_unord_cone_seq[min_index].first->getX() - cone1->getX();
199 vy = nn_unord_cone_seq[min_index].first->getY() - cone1->getY();
200 cone1 = nn_unord_cone_seq[min_index].first;
201
202 cone_seq.push_back(nn_unord_cone_seq[min_index].first); // add cone to ordered sequence
203 }
204
205 // Set spline data
206 for (int i = 0; i < n; i++) {
207 gsl_vector_set(i_values, i, i);
208 gsl_vector_set(x_values, i, cone_seq[i]->getX());
209 gsl_vector_set(y_values, i, cone_seq[i]->getY());
210 // closer cones more important(better stability)
211 gsl_vector_set(w, i, 1.0 / pow(cone_seq[i]->getDistanceTo(cone_seq[(i + 1) % n]), 2));
212 }
213
214 // Set i range within cone set length
215 gsl_bspline_knots_uniform(0, n, bw);
216 gsl_bspline_knots_uniform(0, n, cw);
217
218 /* construct the fit matrix X */
219 for (int i = 0; i < n; i++) {
220 /* compute B_j(xi) for all j */
221 gsl_bspline_eval(i, B, bw);
222 gsl_bspline_eval(i, C, cw);
223
224 /* fill in row i of X */
225 for (int j = 0; j < ncoeffs; j++) {
226 double Bj = gsl_vector_get(B, j);
227 gsl_matrix_set(X, i, j, Bj);
228 double Cj = gsl_vector_get(C, j);
229 gsl_matrix_set(Y, i, j, Cj);
230 }
231 }
232
233 // Fit spline
234 gsl_multifit_wlinear(X, w, x_values, c, cov, &chisq, mw);
235 gsl_multifit_wlinear(Y, w, y_values, c2, cov2, &chisq2, mw2);
236
237 double xi, yi, yerr, yerr2;
238 int divs = 10;
239 std::vector<double> i_eval, x_eval, y_eval;
240 std::vector<std::pair<double, double>> cone_seq_eval; // Eval key_values
241
242 // Output the smoothed curve for more desired points
243 for (int i = 0; i < n; i++) {
244 for (int j = 0; j < divs; j += 1) { // iterate over decimal numbers
245 gsl_bspline_eval(i + static_cast<float>(j) / divs, B, bw);
246 gsl_bspline_eval(i + static_cast<float>(j) / divs, C, cw);
247 gsl_multifit_linear_est(B, c, cov, &xi, &yerr);
248 gsl_multifit_linear_est(C, c2, cov2, &yi, &yerr2);
249 i_eval.push_back(i);
250 x_eval.push_back(xi);
251 y_eval.push_back(yi);
252 if (j == 0) {
253 cone_seq_eval.push_back(std::make_pair(xi, yi));
254 if (i == n - 1) // Decimals can't go over last int
255 break;
256 }
257 }
258 }
259
260 // Delete Outliers
261 int outlierCount = 0;
262
263 for (int i = 0; i < n; i++) {
264 int index = i - outlierCount; // decrease iterator indeleted indexes
265 double dist = sqrt(pow(cone_seq[index]->getX() - cone_seq_eval[i].first, 2) +
266 pow(cone_seq[index]->getY() - cone_seq_eval[i].second, 2));
267 if (dist > distance_threshold) {
268 cone_seq.erase(cone_seq.begin() + index);
269 outlierCount++;
270 }
271 }
272
273 unord_cone_seq = cone_seq;
274
275 // Free Memory
276 gsl_bspline_free(bw);
277 gsl_bspline_free(cw);
278 gsl_vector_free(B);
279 gsl_vector_free(C);
280 gsl_vector_free(i_values);
281 gsl_vector_free(x_values);
282 gsl_vector_free(y_values);
283 gsl_matrix_free(X);
284 gsl_matrix_free(Y);
285 gsl_vector_free(c);
286 gsl_vector_free(w);
287 gsl_matrix_free(cov);
288 gsl_multifit_linear_free(mw);
289 gsl_multifit_linear_free(mw2);
290
291 if (writing) {
292 // Write outputs in files
293 std::string fileSide = side ? "1" : "0";
294
295 std::ofstream splinePathFile = openWriteFile("src/planning/plots/spline" + fileSide + ".txt");
296
297 for (int i = 0; i < static_cast<int>(i_eval.size()); i++)
298 splinePathFile << x_eval[i] << " " << y_eval[i] << "\n";
299 splinePathFile.close();
300
301 std::ofstream outlierPathFile =
302 openWriteFile("src/planning/plots/deletedoutliers" + fileSide + ".txt");
303 for (int i = 0; i < static_cast<int>(cone_seq.size()); i++)
304 outlierPathFile << cone_seq[i]->getX() << " " << cone_seq[i]->getY() << "\n";
305 outlierPathFile.close();
306 }
307
308 return outlierCount;
309}
