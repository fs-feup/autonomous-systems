412 def metrics_to_csv(self, metrics, filename):
413 """!
414 Converts metrics to csv and writes them to a file.
415
416 Args:
417 metrics (list): List of metrics dictionaries.
418 filename (str): Name of the file to write the metrics to.
419 """
420
421 # Add 'time' key to each metric
422 start_time = metrics[0]["timestamp"]
423 for metric in metrics:
424 elapsed_time = (metric["timestamp"] - start_time).total_seconds()
425 metric["time"] = elapsed_time # Add/Update 'time' key with elapsed time
426
427 # Write metrics to csv
428 with open(filename, "w", newline="") as csvfile:
429 fieldnames = metrics[0].keys()
430 writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
431 writer.writeheader()
432 writer.writerows(metrics)

478 def compute_and_publish_state_estimation(
479 self,
480 pose: np.ndarray,
481 groundtruth_pose: np.ndarray,
482 velocities: np.ndarray,
483 groundtruth_velocities: np.ndarray,
484 map: np.ndarray,
485 groundtruth_map: np.ndarray,
486 ) -> None:
487 """!
488 Computes state estimation metrics and publishes them.
489
490 Args:
491 pose (np.ndarray): Vehicle state estimation data. [x,y,theta]
492 groundtruth_pose (np.ndarray): Ground truth vehicle state data. [x,y,theta]
493 veloevaluator-eufs.launch.pyndtruth_velocities (np.ndarray): Ground truth vehicle state velocities. [vx, vy, w]
494 map (np.ndarray): Map data. [[x,y,color,confidence]]
495 groundtruth_map (np.ndarray): Ground truth map data. [[x,y,color,confidence]]
496 """
497 if map.size == 0 or groundtruth_map.size == 0:
498 return
499
500 self.get_logger().debug("Received state estimation")
501
502 # Compute instantaneous vehicle state metrics
503 vehicle_state_error = Float32MultiArray()
504 vehicle_state_error.layout.dim = [MultiArrayDimension()]
505 vehicle_state_error.layout.dim[0].size = 6
506 vehicle_state_error.layout.dim[0].label = (
507 "vehicle state error: [x, y, theta, v, v, w]"
508 )
509 vehicle_state_error.data = [0.0] * 6
510 vehicle_state_error.data[0] = abs(pose[0] - groundtruth_pose[0])
511 vehicle_state_error.data[1] = abs(pose[1] - groundtruth_pose[1])
512 vehicle_state_error.data[2] = abs(pose[2] - groundtruth_pose[2]) % (2 * np.pi)
513 vehicle_state_error.data[3] = abs(
514 velocities[0]
515 - sqrt(
516 pow(groundtruth_velocities[0], 2) + pow(groundtruth_velocities[1], 2)
517 )
518 )
519 vehicle_state_error.data[4] = abs(
520 velocities[1]
521 - sqrt(
522 pow(groundtruth_velocities[0], 2) + pow(groundtruth_velocities[1], 2)
523 )
524 )
525 vehicle_state_error.data[5] = abs(velocities[2] - groundtruth_velocities[2])
526
527 # Compute instantaneous map metrics
528 cone_positions = map[:, :2]
529 groundtruth_cone_positions = groundtruth_map[:, :2]
530 mean_difference = Float32()
531 mean_difference.data = get_average_difference(
532 cone_positions, groundtruth_cone_positions
533 )
534
535 mean_squared_difference = Float32()
536 mean_squared_difference.data = get_mean_squared_difference(
537 cone_positions, groundtruth_cone_positions
538 )
539
540 num_duplicates = Int32()
541 num_duplicates.data = int(get_duplicates(cone_positions, 0.1))
542
543 root_mean_squared_difference = Float32()
544 root_mean_squared_difference.data = sqrt(
545 get_mean_squared_difference(cone_positions, groundtruth_cone_positions)
546 )
547
548 false_positives = Int32()
549 false_positives.data = int(
550 get_false_positives(cone_positions, groundtruth_cone_positions, 2)
551 )
552
553 difference_with_map = Int32()
554 difference_with_map.data = cone_positions.size - groundtruth_cone_positions.size
555
556 self.get_logger().debug(
557 "Computed state estimation metrics:\n \
558 Vehicle state error: {}\n \
559 Mean difference: {}\n \
560 Mean squared difference: {}\n \
561 Root mean squared difference: {}".format(
562 vehicle_state_error,
563 mean_difference,
564 mean_squared_difference,
565 root_mean_squared_difference,
566 )
567 )
568
569 # Publishes computed state estimation metrics
570 self._vehicle_state_difference_.publish(vehicle_state_error)
571 self._map_mean_difference_.publish(mean_difference)
572 self._map_mean_squared_difference_.publish(mean_squared_difference)
573 self._map_root_mean_squared_difference_.publish(root_mean_squared_difference)
574 self._se_number_duplicates.publish(num_duplicates)
575 self._se_false_positives_.publish(false_positives)
576 self._se_difference_with_map_.publish(difference_with_map)
577
578 # Compute map metrics over time
579 self._se_map_sum_error += get_average_difference(
580 cone_positions, groundtruth_cone_positions
581 )
582 self._se_map_squared_sum_error += get_mean_squared_difference(
583 cone_positions, groundtruth_cone_positions
584 )
585 self._se_map_mean_root_squared_sum_error += get_mean_squared_difference(
586 cone_positions, groundtruth_cone_positions
587 ) ** (1 / 2)
588 self._se_count += 1
589 mean_mean_error = Float32()
590 mean_mean_error.data = self._se_map_sum_error / self._se_count
591 mean_mean_squared_error = Float32()
592 mean_mean_squared_error.data = self._se_map_squared_sum_error / self._se_count
593 mean_mean_root_squared_error = Float32()
594 mean_mean_root_squared_error.data = (
595 self._se_map_mean_root_squared_sum_error / self._se_count
596 )
597
598 # Compute vehicle state metrics over time
599 mean_vehicle_state_error = Float32MultiArray()
600 mean_squared_vehicle_state_error = Float32MultiArray()
601 mean_root_squared_vehicle_state_error = Float32MultiArray()
602 mean_vehicle_state_error.layout.dim = [MultiArrayDimension()]
603 mean_squared_vehicle_state_error.layout.dim = [MultiArrayDimension()]
604 mean_root_squared_vehicle_state_error.layout.dim = [MultiArrayDimension()]
605 mean_vehicle_state_error.layout.dim[0].size = 6
606 mean_squared_vehicle_state_error.layout.dim[0].size = 6
607 mean_root_squared_vehicle_state_error.layout.dim[0].size = 6
608 mean_vehicle_state_error.data = [0.0] * 6
609 mean_squared_vehicle_state_error.data = [0.0] * 6
610 mean_root_squared_vehicle_state_error.data = [0.0] * 6
611 for i in range(6):
612 self._sum_vehicle_state_error.data[i] += vehicle_state_error.data[i]
613 self._sum_squared_vehicle_state_error.data[i] += (
614 vehicle_state_error.data[i] ** 2
615 )
616 mean_vehicle_state_error.data[i] = (
617 self._sum_vehicle_state_error.data[i] / self._se_count
618 )
619 mean_squared_vehicle_state_error.data[i] = (
620 self._sum_squared_vehicle_state_error.data[i] / self._se_count
621 )
622 mean_root_squared_vehicle_state_error.data[i] = (
623 sqrt(self._sum_squared_vehicle_state_error.data[i]) / self._se_count
624 )
625
626 # Publish vehicle state errors over time
627 self._state_estimation_mean_vehicle_state_error.publish(
628 mean_vehicle_state_error
629 )
630 self._state_estimation_mean_squared_vehicle_state_error.publish(
631 mean_squared_vehicle_state_error
632 )
633 self._state_estimation_root_mean_squared_vehicle_state_error.publish(
634 mean_root_squared_vehicle_state_error
635 )
636
637 # Publish map metrics over time
638 self._map_mean_mean_error.publish(mean_mean_error)
639 self._map_mean_mean_squared_error.publish(mean_mean_squared_error)
640 self._map_mean_mean_root_squared_error.publish(mean_mean_root_squared_error)
641
642 # For exporting metrics to csv
643 metrics = {
644 "timestamp": datetime.datetime.now(),
645 "vehicle_state_error_x": vehicle_state_error.data[0],
646 "vehicle_state_error_y": vehicle_state_error.data[1],
647 "vehicle_state_error_theta": vehicle_state_error.data[2],
