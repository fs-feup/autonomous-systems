353 def metrics_to_csv(self, metrics, filename):
354 """!
355 Converts metrics to csv and writes them to a file.
356
357 Args:
358 metrics (list): List of metrics dictionaries.
359 filename (str): Name of the file to write the metrics to.
360 """
361
362 # Add 'time' key to each metric
363 start_time = metrics[0]["timestamp"]
364 for metric in metrics:
365 elapsed_time = (metric["timestamp"] - start_time).total_seconds()
366 metric["time"] = elapsed_time # Add/Update 'time' key with elapsed time
