36void PacSimAdapter::timer_callback() {

38 if (tf_buffer_->canTransform("map", "car", tf2::TimePointZero)) {
39 RCLCPP_INFO(this->get_logger(), "timer_callback: canTransform success");
40 custom_interfaces::msg::VehicleState pose;
41 geometry_msgs::msg::TransformStamped t =
42 tf_buffer_->lookupTransform("map", "car", tf2::TimePointZero);
43 pose.header = t.header;
44 tf2::Quaternion q(t.transform.rotation.x, t.transform.rotation.y, t.transform.rotation.z,
45 t.transform.rotation.w);
46 tf2::Matrix3x3 m(q);
47 double roll;
48 double pitch;
49 double yaw;
50 m.getRPY(roll, pitch, yaw);
51 pose.theta = yaw;
52 pose.position.x = t.transform.translation.x;
53 pose.position.y = t.transform.translation.y;
54 pose.linear_velocity = this->last_stored_velocity_;

57 RCLCPP_INFO(get_logger(), "Pose info. Position:%f, %f; Linear velocity %f, Theta %f",
58 pose.position.x, pose.position.y, pose.linear_velocity, pose.theta);
