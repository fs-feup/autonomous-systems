105void NoWSSEKF::correct_motor_and_steering(Eigen::Vector3d& state, Eigen::Matrix3d& covariance,
106 double motor_rpm, double steering_angle) {
107 Eigen::VectorXd predicted_observations = this->s2v_model->cg_velocity_to_wheels(state);
108 Eigen::VectorXd observations = Eigen::VectorXd::Zero(2);
109 observations << steering_angle, motor_rpm;
110 Eigen::VectorXd y = observations - predicted_observations;
111 Eigen::MatrixXd jacobian = this->s2v_model->jacobian_cg_velocity_to_wheels(state);
112 Eigen::MatrixXd kalman_gain = covariance * jacobian.transpose() *
113 (jacobian * covariance * jacobian.transpose() +
114 this->_motor_and_steering_measurement_noise_matrix_)
115 .inverse();
116
117 // DEBUG PRINTS
118 RCLCPP_INFO(rclcpp::get_logger("velocity_estimation"),
119 "correct_motor_and_steering - Predicted observations: %f %f ",
120 predicted_observations(0), predicted_observations(1));
121 RCLCPP_INFO(rclcpp::get_logger("velocity_estimation"),
122 "correct_motor_and_steering - Observations: %f %f", observations(0), observations(1));
123 RCLCPP_INFO_STREAM(rclcpp::get_logger("velocity_estimation"),
124 "correct_motor_and_steering - Covariance: \n"
125 << covariance);
126 RCLCPP_INFO_STREAM(rclcpp::get_logger("velocity_estimation"), "correct_motor_and_steering - y: \n"
127 << y);
128 RCLCPP_INFO_STREAM(rclcpp::get_logger("velocity_estimation"),
129 "correct_motor_and_steering - Jacobian: \n"
130 << jacobian);
131 RCLCPP_INFO_STREAM(rclcpp::get_logger("velocity_estimation"),
132 "correct_motor_and_steering - Kalman gain: \n"
133 << kalman_gain);
134 state += kalman_gain * y;
135 RCLCPP_INFO_STREAM(rclcpp::get_logger("velocity_estimation"),
136 "correct_motor_and_steering - New State: \n"
137 << state);
138 if (this->_has_made_prediction_)
139 covariance = (Eigen::Matrix3d::Identity() - kalman_gain * jacobian) * covariance;
140}
