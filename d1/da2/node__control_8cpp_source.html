121void Control::control_timer_callback() {
122 double throttle_command = this->throttle_command_;
123 double steering_command = this->steering_command_;
124 if (this->params_.test_mode_ == "const_torque") {
125 throttle_command = this->params_.const_torque_value_;
126 } else if (this->params_.test_mode_ == "const_torque_fixed_steering") {
127 throttle_command = this->params_.const_torque_value_;
128 steering_command = 0;
129 } else if (this->params_.test_mode_ == "limited_steering") {
130 steering_command *= this->params_.steering_limiting_factor_;
131 }
132 publish_cmd(throttle_command, steering_command);
133}

136void Control::publish_control(const custom_interfaces::msg::Pose& vehicle_state_msg) {
137 if (!go_signal_) {
138 RCLCPP_INFO(rclcpp::get_logger("control"), "Go Signal Not received");
139
140 return;
141 }
142
143 if (received_path_point_array && !received_vehicle_state) {
144 RCLCPP_DEBUG(rclcpp::get_logger("control"), "First Vehicle State Received");
145
146 received_vehicle_state = true;
147 custom_interfaces::msg::PathPoint initial;
148 initial.x = vehicle_state_msg.x;
149 initial.y = vehicle_state_msg.y;
150 initial.v = this->velocity_;
151
152 // Insert at the beginning of the pathpoint_array_ (true "push_front")
153 pathpoint_array_.insert(pathpoint_array_.begin(), initial);
154 }
155
156 rclcpp::Time start = this->now();
157
158 // update vehicle pose
159 this->point_solver_.update_vehicle_pose(vehicle_state_msg, this->velocity_);
160
161 // find the closest point on the path
162 // print pathpoint array size
163 auto [closest_point, closest_point_id, closest_point_velocity] =
164 this->point_solver_.update_closest_point(pathpoint_array_);
165 if (closest_point_id == -1) {
166 RCLCPP_ERROR(rclcpp::get_logger("control"),
167 "PurePursuit: Failed to update closest point, size of pathpoint_array: %ld",
168 pathpoint_array_.size());
169 }
170
171 // update the Lookahead point
172 auto [lookahead_point, lookahead_velocity, lookahead_error] =
173 this->point_solver_.update_lookahead_point(pathpoint_array_, closest_point_id);
174
175 if (lookahead_error) {
176 RCLCPP_ERROR(rclcpp::get_logger("control"), "PurePursuit: Failed to update lookahed point");
177 }
178
179 // calculate longitudinal control: PI-D
180 double torque = this->long_controller_.update(closest_point_velocity, this->velocity_);
181
182 // calculate Lateral Control: Pure Pursuit
183 double steering_angle = this->lat_controller_.pp_steering_control_law(
184 this->point_solver_.vehicle_pose_.rear_axis_, this->point_solver_.vehicle_pose_.position,
185 lookahead_point, this->point_solver_.dist_cg_2_rear_axis_);
186 RCLCPP_WARN(rclcpp::get_logger("control"), "Steering Angle: %f", steering_angle);
187 // check if steering is Nan
188 if (std::isnan(steering_angle) || std::isnan(torque)) {
189 RCLCPP_ERROR(rclcpp::get_logger("control"), "Steering Angle or Torque is NaN");
190 return;
191 }
192
193 RCLCPP_INFO(rclcpp::get_logger("control"), "Rear Axis: %f",
194 this->point_solver_.vehicle_pose_.rear_axis_);
195 RCLCPP_INFO(rclcpp::get_logger("control"), "CG to rear axis Rear Axis: %f",
196 this->point_solver_.dist_cg_2_rear_axis_);
197 RCLCPP_INFO(rclcpp::get_logger("control"), "Position: (%f, %f)",
198 this->point_solver_.vehicle_pose_.position.x,
199 this->point_solver_.vehicle_pose_.position.y);
200 RCLCPP_INFO(rclcpp::get_logger("control"), "Lookahead Point: (%f, %f)", lookahead_point.x,
201 lookahead_point.y);
202 RCLCPP_INFO(rclcpp::get_logger("control"), "CG to rear axis Rear Axis: %f",
203 this->point_solver_.dist_cg_2_rear_axis_);
204 // check if steering is Nan
205 if (std::isnan(steering_angle) || std::isnan(torque)) {
206 RCLCPP_ERROR(rclcpp::get_logger("control"), "Steering Angle or Torque is NaN");
207 return;
208 }
209
210 double execution_time = (this->now() - start).seconds() * 1000;
211
212 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Current vehicle velocity: %f",
213 this->point_solver_.vehicle_pose_.velocity_);
214 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Rear axis coords: %f, %f",
215 this->point_solver_.vehicle_pose_.rear_axis_.x,
216 this->point_solver_.vehicle_pose_.rear_axis_.y);
217 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Closest Point: %f, %f, ID %d", closest_point.x,
218 closest_point.y, closest_point_id);
219 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Lookahead Point: %f, %f", lookahead_point.x,
220 lookahead_point.y);
221 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Lookahead Velocity: %f", lookahead_velocity);
222 RCLCPP_DEBUG(rclcpp::get_logger("control"), "Torque: %f, Steering Angle: %f", torque,
223 steering_angle);
224
225 // Direct conversion of Pose to VehicleState, still needs a better solution, but works for now to
226 // have control evalution data
227 custom_interfaces::msg::VehicleState vehicle_state;
228 vehicle_state.header = std_msgs::msg::Header();
229 vehicle_state.position.x = this->point_solver_.vehicle_pose_.rear_axis_.x;
230 vehicle_state.position.y = this->point_solver_.vehicle_pose_.rear_axis_.y;
231 vehicle_state.theta = this->point_solver_.vehicle_pose_.orientation;
232 vehicle_state.linear_velocity = this->point_solver_.vehicle_pose_.velocity_;
233 vehicle_state.angular_velocity = 0.0;
234 publish_evaluator_data(lookahead_velocity, lookahead_point, closest_point, vehicle_state,
235 closest_point_velocity, execution_time);
236 publish_visualization_data(lookahead_point, closest_point);
237
238 this->throttle_command_ = torque;
239 this->steering_command_ = steering_angle;
240 // Adapter to communicate with the car
241}
