42Plane ConstrainedRANSAC::calculate_plane(const pcl::PointCloud<pcl::PointXYZI>::Ptr point_cloud,
43 const Plane& target_plane) const {
44 std::random_device rd;
45 std::mt19937 gen(rd());
46
47 Plane best_plane;
48 int best_plane_inliers = 0;
49 double best_plane_max_deviation = 0.0;
50 bool best_plane_found = false;
51 int n_skips = 0;
52
53 for (int i = 0; i < n_tries; ++i) {
54 // Randomly sample 3 different points
55 std::vector<pcl::PointXYZI> sampled_points(3);
56 std::vector<int> indices(3);
57
58 indices = pick_3_random_indices(static_cast<int>(point_cloud->points.size()), gen);
59 sampled_points[0] = point_cloud->points[indices[0]];
60 sampled_points[1] = point_cloud->points[indices[1]];
61 sampled_points[2] = point_cloud->points[indices[2]];
62
63 // Calculate plane from 3 points
64 Plane candidate_plane = fit_plane_to_points(sampled_points);
65
66 // Check angle constraint if target plane is provided
67 if (target_plane.get_a() != 0 || target_plane.get_b() != 0 || target_plane.get_c() != 0) {
68 double angle_diff = calculate_angle_difference(candidate_plane, target_plane);
69
70 // Skip if angle difference is bigger than defined threshold
71 if (angle_diff > plane_angle_diff) {
72 n_skips++;
73 continue;
74 }
75 }
76
77 // Count inliers and calculate max deviation
78 int inliers = 0;
79 double max_deviation = 0.0;
80
81 for (const auto& point : point_cloud->points) {
82 double distance = distance_to_plane(point, candidate_plane);
83 if (distance < epsilon) {
84 inliers++;
85 if (distance > max_deviation) {
86 max_deviation = distance;
87 }
88 }
89 }
90
91 // Update best plane based on inliers count and max deviation
92 if (!best_plane_found || inliers > best_plane_inliers) {
93 best_plane = candidate_plane;
94 best_plane_inliers = inliers;
95 best_plane_max_deviation = max_deviation;
96 best_plane_found = true;
97
98 } else if (inliers == best_plane_inliers && max_deviation < best_plane_max_deviation) {
99 best_plane = candidate_plane;
100 best_plane_inliers = inliers;
101 best_plane_max_deviation = max_deviation;
102 } else {
103 // No update
104 }
105 }
106
107 if (!best_plane_found || n_skips == n_tries) {
108 best_plane = target_plane;
109 }
110 return best_plane;

113Plane ConstrainedRANSAC::fit_plane_to_points(const std::vector<pcl::PointXYZI>& points) const {
114 const auto& p1 = points[0];
115 const auto& p2 = points[1];
116 const auto& p3 = points[2];
117
118 // Calculate two vectors from the points
119 Eigen::Vector3d v1(p2.x - p1.x, p2.y - p1.y, p2.z - p1.z);
120 Eigen::Vector3d v2(p3.x - p1.x, p3.y - p1.y, p3.z - p1.z);
121
122 // Calculate normal vector using cross product
123 Eigen::Vector3d normal = v1.cross(v2);
124 double nrm = normal.norm();
125
126 Plane plane(0.0, 0.0, 0.0, 0.0);
127 // Check for case where points are collinear
128 if (nrm >= 1e-12) {
129 normal /= nrm;
130 plane = Plane(normal.x(), normal.y(), normal.z(),
131 -(normal.x() * p1.x + normal.y() * p1.y + normal.z() * p1.z));
132 }
133
134 return plane;
135}

137double ConstrainedRANSAC::distance_to_plane(const pcl::PointXYZI& point, const Plane& plane) const {
138 // Plane equation: Ax + By + Cz + D = 0
139 double A = plane.get_a();
140 double B = plane.get_b();
141 double C = plane.get_c();
142 double D = plane.get_d();
143
144 double denom = std::sqrt(A * A + B * B + C * C);
145 double result = 0;
146 // Avoid possible division by zero
147 if (denom < 1e-12) {
148 result = std::numeric_limits<double>::infinity();
149 } else {
150 result = std::abs(A * point.x + B * point.y + C * point.z + D) / denom;
151 }
152 return result;
153}

155double ConstrainedRANSAC::calculate_angle_difference(const Plane& plane1,
156 const Plane& plane2) const {
157 Eigen::Vector3d normal1(plane1.get_a(), plane1.get_b(), plane1.get_c());
158 Eigen::Vector3d normal2(plane2.get_a(), plane2.get_b(), plane2.get_c());
159 normal1.normalize();
160 normal2.normalize();
161
162 // Calculate dot product
163 double dot_product = normal1.dot(normal2);
164 dot_product = std::clamp(dot_product, -1.0, 1.0);
165
166 // Calculate angle between the normals
167 double radian_angle = std::acos(std::abs(dot_product));
168 return radian_angle * (180.0 / M_PI); // Convert to degrees
169}

171std::vector<int> ConstrainedRANSAC::pick_3_random_indices(int max_index, std::mt19937& gen) const {
172 std::uniform_int_distribution dis1(0, max_index - 1);
173 int i1 = dis1(gen);
174
175 // Do not choose the last element, because we may need to increment it
176 std::uniform_int_distribution dis2(0, max_index - 2);
177 int i2 = dis2(gen);
178 if (i2 == i1) {
179 ++i2;
180 }
181
182 // Do not choose the last two elements, because we may need to increment it twice
183 std::uniform_int_distribution dis3(0, max_index - 3);
184 int i3 = dis3(gen);
185 if (i3 == i1 || i3 == i2) {
186 ++i3;
187 }
188 if (i3 == i1 || i3 == i2) {
189 ++i3;
190 }
191
192 return {i1, i2, i3};
193}
