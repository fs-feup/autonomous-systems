112def get_key(settings, timeout):
113 """!
114 @brief Get a single key from the user.
115 """
116 tty.setraw(sys.stdin.fileno())
117 rlist, _, _ = select([sys.stdin], [], [], timeout)
118 key = sys.stdin.read(1) if rlist else ""
119 termios.tcsetattr(sys.stdin, termios.TCSADRAIN, settings)
120 return key
