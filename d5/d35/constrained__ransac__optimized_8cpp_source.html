79double ConstrainedRANSACOptimized::calculate_angle_difference(const Plane& plane1,
80 const Plane& plane2) const {
81 Eigen::Vector3d normal1(plane1.get_a(), plane1.get_b(), plane1.get_c());
82 Eigen::Vector3d normal2(plane2.get_a(), plane2.get_b(), plane2.get_c());
83 normal1.normalize();
84 normal2.normalize();
85
86 // Calculate dot product
87 double dot_product = normal1.dot(normal2);
88 dot_product = std::clamp(dot_product, -1.0, 1.0);
89
90 // Calculate angle between the normals
91 double radian_angle = std::acos(std::abs(dot_product));
92 return radian_angle * (180.0 / M_PI); // Convert to degrees
93}

95double ConstrainedRANSACOptimized::distance_to_plane(const pcl::PointXYZI& point,
96 const Plane& plane) const {
97 // Plane equation: Ax + By + Cz + D = 0
98 double A = plane.get_a();
99 double B = plane.get_b();
100 double C = plane.get_c();
101 double D = plane.get_d();
102
103 return std::abs(A * point.x + B * point.y + C * point.z + D) / std::sqrt(A * A + B * B + C * C);
104}
