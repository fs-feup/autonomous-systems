153fs_ai_api_mission_status_e CanInterface::getMissionStatus(const fs_ai_api_vcu2ai_struct data) {
154 switch (data.VCU2AI_AS_STATE) {
155 case fs_ai_api_as_state_e::AS_OFF:
156 // Check whether a mission has been chosen, and acknowledge it if so
157 if (data.VCU2AI_AMI_STATE != fs_ai_api_ami_state_e::AMI_NOT_SELECTED)
158 return fs_ai_api_mission_status_e::MISSION_SELECTED;
159 else
160 return fs_ai_api_mission_status_e::MISSION_NOT_SELECTED;
161 case fs_ai_api_as_state_e::AS_READY:
162 if (driving_flag_) {
163 // our stack is ready to start driving
164 return fs_ai_api_mission_status_e::MISSION_RUNNING;
165 } else {
166 // still not ready to start driving
167 return fs_ai_api_mission_status_e::MISSION_SELECTED;
168 }
169 case fs_ai_api_as_state_e::AS_DRIVING:
170 if (mission_complete_) {
171 // mission has been finished
172 return fs_ai_api_mission_status_e::MISSION_FINISHED;
173 } else {
174 // still doing a mission
175 return fs_ai_api_mission_status_e::MISSION_RUNNING;
176 }
177 case fs_ai_api_as_state_e::AS_FINISHED:
178 return fs_ai_api_mission_status_e::MISSION_FINISHED;
179 default:
180 return fs_ai_api_mission_status_e::MISSION_NOT_SELECTED;
181 }

184void CanInterface::commandCallback(ackermann_msgs::msg::AckermannDriveStamped::SharedPtr msg) {
185 if (driving_flag_) {
186 float acceleration = msg->drive.acceleration;
187 // Divide by two as we have two motors (front and rear) which supply identical torque
188 // 0.5 is an empirically measured constant for the rolling friction
189 torque_ = (TOTAL_MASS_ * WHEEL_RADIUS_ * std::abs(acceleration + 0.5))/ 2.0;
190 if (acceleration > 0.0) {
191 braking_ = 0.0;
192 rpm_request_ = rpm_limit_; // Positive torque if the rpm_request is positive
193 } else if (acceleration == 0.0) {
194 torque_ = 0.0;
195 braking_ = 0.0;
196 } else if (acceleration > engine_threshold_) { // Engine brake if possible
197 braking_ = 0.0;
198 rpm_request_ = 0.0; // A negative torque is achieved by an rpm_request with a value of zero
199 } else { // If we want to decelerate faster than the engine allows, use the mechanical brakes
200 torque_ = 0.0; // Don't engine brake and mechanical brake simultaneously
201 braking_ = (-acceleration / max_dec_) * MAX_BRAKE_; // Braking needs to be positive
202 }
203
204 // Convert from radians to degrees
205 steering_ = static_cast<float>(msg->drive.steering_angle * 180.0 / M_PI);
206 } else {
207 // Vehicle will not move to AS_DRIVING unless these are set to 0
208 steering_ = 0;
209 torque_ = 0;
210 rpm_request_ = 0.0;
211
212 // We don't need braking unless we are driving
213 braking_ = 0;
214 }
215
216 steering_ = checkAndTrunc(steering_, MAX_STEERING_ANGLE_DEG_, "setting steering", false);
217 torque_ = checkAndTrunc(torque_, MAX_TORQUE_, "setting torque");
218 braking_ = checkAndTrunc(braking_, MAX_BRAKE_, "setting brake");
219
220 // Update last message time
221 last_cmd_message_time_ = this->now().seconds();
222}

228void CanInterface::drivingFlagCallback(std_msgs::msg::Bool::SharedPtr msg) {
229 // Driving flag can only be set to true if we're in AS_DRIVING_
230 if (msg->data && as_state_ == fs_ai_api_as_state_e::AS_DRIVING) {
231 driving_flag_ = true;
232 } else if (msg->data) {
233 driving_flag_ = false;
234 RCLCPP_WARN(get_logger(), "Driving flag is true but as_state is %i", as_state_);
235 } else {
236 driving_flag_ = msg->data;
237 }
238
239 if (driving_flag_ && last_cmd_message_time_ == 0.0) {
240 // We start the cmd timeout on the positive edge of `driving_flag_`
241 last_cmd_message_time_ = this->now().seconds();
242 } else if (!driving_flag_) {
243 // Reset last cmd message time back to magic value (0.0) if we stop driving
244 last_cmd_message_time_ = 0.0;
245 }

274eufs_msgs::msg::WheelSpeedsStamped CanInterface::makeWsMessage(const fs_ai_api_vcu2ai_struct data) {
275 auto msg = eufs_msgs::msg::WheelSpeedsStamped();
276 msg.header.stamp = this->get_clock()->now();
277 msg.header.frame_id = "base_footprint";
278
279 float steering_feedback = -data.VCU2AI_STEER_ANGLE_deg; // inverted to match ISO convention
280 float fl_speed = data.VCU2AI_FL_WHEEL_SPEED_rpm;
281 float fr_speed = data.VCU2AI_FR_WHEEL_SPEED_rpm;
282 float rl_speed = data.VCU2AI_RL_WHEEL_SPEED_rpm;
283 float rr_speed = data.VCU2AI_RR_WHEEL_SPEED_rpm;
284
285 // check values
286 msg.speeds.lf_speed = checkAndTrunc(fl_speed, MAX_RPM_, "FL wheelspeed");
287 msg.speeds.rf_speed = checkAndTrunc(fr_speed, MAX_RPM_, "FR wheelspeed");
288 msg.speeds.lb_speed = checkAndTrunc(rl_speed, MAX_RPM_, "RL wheelspeed");
289 msg.speeds.rb_speed = checkAndTrunc(rr_speed, MAX_RPM_, "RR wheelspeed");
290
291 steering_feedback = checkAndTrunc(steering_feedback, MAX_STEERING_ANGLE_DEG_, "steering", false);
292 msg.speeds.steering = (steering_feedback / 180) * M_PI; // convert to radians
293
294 return msg;
295}

297geometry_msgs::msg::TwistWithCovarianceStamped CanInterface::makeTwistMessage(
298 const fs_ai_api_vcu2ai_struct data) {
299 auto msg = geometry_msgs::msg::TwistWithCovarianceStamped();
300 msg.header.stamp = get_clock()->now();
301 msg.header.frame_id = "base_footprint";
302
303 auto wheel_speed = (checkAndTrunc(data.VCU2AI_RL_WHEEL_SPEED_rpm, MAX_RPM_, "RL ws") +
304 checkAndTrunc(data.VCU2AI_RR_WHEEL_SPEED_rpm, MAX_RPM_, "RR ws")) /
305 2;
306
307 msg.twist.twist.linear.x = wheel_speed * M_PI * WHEEL_RADIUS_ / 30;
308
309 auto steering_angle =
310 checkAndTrunc(-data.VCU2AI_STEER_ANGLE_deg, MAX_STEERING_ANGLE_DEG_, "steering", false) /
311 180 * M_PI;
312
313 msg.twist.twist.angular.z = msg.twist.twist.linear.x * sin(steering_angle) / WHEELBASE_;
314
315 msg.twist.covariance = {1e-9, 0, 0, 0, 0, 0,
316 0, 0, 0, 0, 0, 0,
317 0, 0, 0, 0, 0, 0,
318 0, 0, 0, 0, 0, 0,
319 0, 0, 0, 0, 0, 0,
320 0, 0, 0, 0, 0, 1e-9};
321
322 return msg;
323}

477// Checks if value exceeds max allowed value, if it does truncate it and warn
478float CanInterface::checkAndTrunc(const float val, const float max_val, const std::string type,
479 bool trunc_at_zero) {
480 float min_val = trunc_at_zero ? 0 : -max_val;
481 if (val > max_val) {
482 RCLCPP_DEBUG(get_logger(), "%s was %f but max is %f", type.c_str(), val, max_val);
483 return max_val;
