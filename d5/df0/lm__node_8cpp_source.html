32void LMNode::_perception_subscription_callback(const custom_interfaces::msg::ConeArray msg) {
33 auto cone_array = msg.cone_array;
34 if (this->_perception_map == nullptr) {
35 RCLCPP_WARN(this->get_logger(), "SUB - Perception map is null");
36 return;
37 }
38 this->_perception_map->map.clear();
39 RCLCPP_DEBUG(this->get_logger(), "SUB - cones from perception:");
40 RCLCPP_DEBUG(this->get_logger(), "--------------------------------------");
41
42 for (auto &cone : cone_array) {
43 auto position = Position();
44 position.x = cone.position.x;
45 position.y = cone.position.y;
46
47 auto color = colors::color_map.at(cone.color.c_str());
48
49 RCLCPP_DEBUG(this->get_logger(), "(%f, %f)\t%s", position.x, position.y, cone.color.c_str());
50
51 this->_perception_map->map.insert({position, color});
52 }
53 RCLCPP_DEBUG(this->get_logger(), "--------------------------------------");
54
55 if (this->_ekf == nullptr) {
56 RCLCPP_WARN(this->get_logger(), "ATTR - EKF object is null");
57 return;
58 }
59 this->_ekf->correction_step(*(this->_perception_map));
60 this->_ekf->update(this->_vehicle_state, this->_track_map);
61 RCLCPP_DEBUG(this->get_logger(), "EKF - EFK correction Step");
62 this->_publish_localization();
63 this->_publish_map();
64}

66void LMNode::_imu_subscription_callback(double angular_velocity, double acceleration_x,
67 double acceleration_y) {
68 if (this->_use_odometry) {
69 return;
70 }
71 if (this->_motion_update == nullptr) {
72 RCLCPP_WARN(this->get_logger(), "ATTR - Motion update object is null");
73 return;
74 }
75 this->_motion_update->rotational_velocity =
76 angular_velocity * (180 / M_PI); // Angular velocity in radians
77 std::chrono::time_point<std::chrono::high_resolution_clock> now =
78 std::chrono::high_resolution_clock::now();
79 double delta =
80 std::chrono::duration_cast<std::chrono::microseconds>(now - this->_motion_update->last_update)
81 .count();
82 this->_motion_update->last_update = now;
83 this->_motion_update->translational_velocity_y += (acceleration_y * delta) / 1000000;
84 this->_motion_update->translational_velocity_x += (acceleration_x * delta) / 1000000;
85 this->_motion_update->translational_velocity =
86 sqrt(this->_motion_update->translational_velocity_x *
87 this->_motion_update->translational_velocity_x +
88 this->_motion_update->translational_velocity_y *
89 this->_motion_update->translational_velocity_y);
90 RCLCPP_DEBUG(this->get_logger(), "SUB - raw from IMU: ax:%f - ay:%f - w:%f", acceleration_x,
91 acceleration_y, this->_motion_update->rotational_velocity);
92 RCLCPP_DEBUG(this->get_logger(), "SUB - translated from IMU: v:%f - w:%f - vx:%f - vy:%f",
93 this->_motion_update->translational_velocity,
94 this->_motion_update->rotational_velocity,
95 this->_motion_update->translational_velocity_x,
96 this->_motion_update->translational_velocity_y);
97}

99void LMNode::_wheel_speeds_subscription_callback(double lb_speed, double lf_speed, double rb_speed,
100 double rf_speed, double steering_angle) {
101 if (!this->_use_odometry) {
102 return;
103 }
104 RCLCPP_DEBUG(this->get_logger(),
105 "SUB - Raw from wheel speeds: lb:%f - rb:%f - lf:%f - rf:%f - "
106 "steering: %f",
107 lb_speed, rb_speed, lf_speed, rf_speed, steering_angle);
108 MotionUpdate motion_prediction_data = LMNode::odometry_to_velocities_transform(
109 lb_speed, lf_speed, rb_speed, rf_speed, steering_angle);
110 this->_motion_update->translational_velocity = motion_prediction_data.translational_velocity;
111 this->_motion_update->rotational_velocity = motion_prediction_data.rotational_velocity;
112 this->_motion_update->steering_angle = steering_angle;
113 this->_motion_update->last_update = std::chrono::high_resolution_clock::now();
114 RCLCPP_DEBUG(this->get_logger(), "SUB - translated from wheel speeds: v:%f - w:%f",
115 this->_motion_update->translational_velocity,
116 this->_motion_update->rotational_velocity);
117
118 if (this->_ekf == nullptr) {
119 RCLCPP_WARN(this->get_logger(), "PUB - EKF object is null");
120 return;
121 }
122 MotionUpdate temp_update = *(this->_motion_update);
123 this->_ekf->prediction_step(temp_update);
124 this->_ekf->update(this->_vehicle_state, this->_track_map);
125 RCLCPP_DEBUG(this->get_logger(), "EKF - EFK prediction Step");
126 this->_publish_localization();
127 this->_publish_map();
128}

200MotionUpdate LMNode::odometry_to_velocities_transform(double lb_speed,
201 [[maybe_unused]] double lf_speed,
202 double rb_speed,
203 [[maybe_unused]] double rf_speed,
204 double steering_angle) {
205 MotionUpdate motion_prediction_data_transformed;
206 if (steering_angle == 0) { // If no steering angle, moving straight
207 double lb_velocity = get_wheel_velocity_from_rpm(lb_speed, WHEEL_DIAMETER);
208 double rb_velocity = get_wheel_velocity_from_rpm(rb_speed, WHEEL_DIAMETER);
209 // double lf_velocity = get_wheel_velocity_from_rpm(lf_speed,
210 // WHEEL_DIAMETER); // Simulator double rf_velocity =
211 // get_wheel_velocity_from_rpm(rf_speed, WHEEL_DIAMETER); // Are always 0
212 motion_prediction_data_transformed.translational_velocity = (lb_velocity + rb_velocity) / 2;
213 } else if (steering_angle > 0) {
214 double lb_velocity = get_wheel_velocity_from_rpm(lb_speed, WHEEL_DIAMETER);
215 double rear_axis_center_rotation_radius = WHEELBASE / tan(steering_angle);
216 motion_prediction_data_transformed.rotational_velocity =
217 lb_velocity / (rear_axis_center_rotation_radius - (AXIS_LENGTH / 2));
218 motion_prediction_data_transformed.translational_velocity =
219 sqrt(pow(rear_axis_center_rotation_radius, 2) + pow(REAR_AXIS_TO_CAMERA, 2)) *
220 abs(motion_prediction_data_transformed.rotational_velocity);
221 } else {
222 double rb_velocity = get_wheel_velocity_from_rpm(rb_speed, WHEEL_DIAMETER);
223 double rear_axis_center_rotation_radius = WHEELBASE / tan(steering_angle);
224 motion_prediction_data_transformed.rotational_velocity =
225 rb_velocity / (rear_axis_center_rotation_radius + (AXIS_LENGTH / 2));
226 motion_prediction_data_transformed.translational_velocity =
227 sqrt(pow(rear_axis_center_rotation_radius, 2) + pow(REAR_AXIS_TO_CAMERA, 2)) *
228 abs(motion_prediction_data_transformed.rotational_velocity);
229 }
230 return motion_prediction_data_transformed;
231}
