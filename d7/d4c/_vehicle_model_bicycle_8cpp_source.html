145 double processSlipAngleLat(double alpha_input, double Fz)
146 {
147 using namespace VehicleModelConstants;
148
149 double dpi = (p_input - NOMPRES) / NOMPRES;
150 double Fz_0_prime = LFZO * FNOM;
151 double Kya = PKY1 * Fz_0_prime * (1 + PPY1 * dpi) * (1 - PKY3 * abs(y_input))
152 * sin(PKY4 * atan((Fz / Fz_0_prime) / ((PKY2 + PKY5 * y_input * y_input) * (1 + PPY2 * dpi)))) * zeta_3
153 * LKY;
154 double dfz = (Fz - Fz_0_prime) / Fz_0_prime;
155 double Kyg0 = Fz * (PKY6 + PKY7 * dfz) * (1 + PPY5 * dpi) * LKYC;
156 double Vs_y = tan(alpha_input) * abs(V_cx);
157 double Vs_x = -slip_ratio * abs(V_cx);
158 double Vs = sqrt(Vs_x * Vs_x + Vs_y * Vs_y);
159 double V0 = LONGVL;
160 double LMUY_star = LMUY / (1.0 + LMUV * (Vs / V0));
161 double SVyg = Fz * (PVY3 + PVY4 * dfz) * y_input * LKYC * LMUY_star * zeta_2;
162 double SHy
163 = (PHY1 + PHY2 * dfz) + ((Kyg0 * y_input - SVyg) / (Kya + eps_k * sign(Kya))) * zeta_0 + zeta_4 - 1.0;
164 double alpha_y = alpha_input + SHy;
165 double Ey
166 = (PEY1 + PEY2 * dfz) * (1 + PEY5 * y_input * y_input - (PEY3 + PEY4 * y_input) * sign(alpha_y)) * LEY;
167 double Cy = PCY1 * LCY;
168 double mu_y = (PDY1 + PDY2 * dfz) * (1.0 + PPY3 * dpi + PPY4 * dpi * dpi) * (1.0 - PDY3 * y_input * y_input)
169 * LMUY_star;
170 double Dy = mu_y * Fz * zeta_2;
171 double By = Kya / (Cy * Dy + eps_y * sign(Dy));
172 double DVyk = mu_y * Fz * (RVY1 + RVY2 * dfz + RVY3 * y_input) * cos(atan(RVY4 * alpha_input)) * zeta_2;
173 double SVyk = DVyk * sin(RVY5 * atan(RVY6 * slip_ratio)) * LVYKA;
174 double Eyk = REY1 + REY2 * dfz;
175 double SHyk = RHY1 + RHY2 * dfz;
176 double ks = slip_ratio + SHyk;
177 double Byk = (RBY1 + RBY4 * y_input * y_input) * cos(atan(RBY2 * (alpha_input - RBY3))) * LYKA;
178 double Cyk = RCY1;
179 double Gyk_0 = cos(Cyk * atan(Byk * SHyk - Eyk * (Byk * SHyk - atan(Byk * SHyk))));
180 double Gyk = (cos(Cyk * atan(Byk * ks - Eyk * (Byk * ks - atan(Byk * ks)))) / Gyk_0);
181 double SVy = Fz * (PVY1 + PVY2 * dfz) * LVY * LMUY_star * zeta_2 + SVyg;
182 double Fy_0 = Dy * sin(Cy * atan(By * alpha_y - Ey * (By * alpha_y - atan(By * alpha_y)))) + SVy;
183 double Fy = Gyk * Fy_0 + SVyk;
184
185 return Fy;

215 Eigen::Vector3d rRR = Eigen::Vector3d(-lr, -0.5 * sr, 0.0);
216 Eigen::Vector3d rFront = Eigen::Vector3d(lf, 0.0, 0.0);
217 Eigen::Vector3d rRear = Eigen::Vector3d(-lf, 0.0, 0.0);

219 // Calculate the velocities of the wheels
220 Eigen::Vector3d vFL = vCog + omega.cross(rFL);

222 Eigen::Vector3d vRL = vCog + omega.cross(rRL);
223 Eigen::Vector3d vRR = vCog + omega.cross(rRR);
224 Eigen::Vector3d vFront = vCog + omega.cross(rFront);
225 Eigen::Vector3d vRear = vCog + omega.cross(rRear);

227 double rpm2ms = this->wheelRadius * 2.0 * M_PI / 60;
228
229 // Check if the vehicle is at a standstill
230 bool stillstand = (vCog.norm() < 0.1) && (std::abs(this->angularVelocity.z()) < 0.001);
231
232 // Calculate tire side slip angles
233 double eps = 0.00001;
234 double kappaFront = std::atan2(vFront.y(), std::max(std::abs(vFront.x()), eps)) - steeringFront;
235 double kappaRear = std::atan2(vRear.y(), std::max(std::abs(vRear.x()), eps));
236
237 if (stillstand)

239 kappaFront = 0.0;
240 kappaRear = 0.0;

242

244 double Fx_FL = this->gearRatio * this->torques.FL / this->wheelRadius;
245 double Fx_FR = this->gearRatio * this->torques.FR / this->wheelRadius;
246 double Fx_RL = this->gearRatio * this->torques.RL / this->wheelRadius;

252 Fx_RL *= (((this->torques.RL) > 0.5) || (vCog.x() > 0.3)) ? 1.0 : 0.0;
253 Fx_RR *= (((this->torques.RR) > 0.5) || (vCog.x() > 0.3)) ? 1.0 : 0.0;

255 // Calculate lateral forces on the front and rear axles
256 // double Dlat_Front = this->Dlat * Fz_Front;
257 // double Dlat_Rear = this->Dlat * Fz_Rear;
258 double Fy_Front = /* Dlat_Front * */ processSlipAngleLat(kappaFront, Fz_Front/2) * 2;
259 double Fy_Rear = /* Dlat_Rear * */ processSlipAngleLat(kappaRear, Fz_Rear/2) * 2;

262 // Convert wheel speeds to RPM

264 this->wheelspeeds.FR = vFR.x() / rpm2ms;
265 this->wheelspeeds.RL = vRL.x() / rpm2ms;
266 this->wheelspeeds.RR = vRR.x() / rpm2ms;
267
268 // Calculate longitudinal and lateral accelerations
269 double axTires = (std::cos(this->steeringAngles.FL) * Fx_FL + std::cos(this->steeringAngles.FR) * Fx_FR + Fx_RL
270 + Fx_RR - std::sin(steeringFront) * Fy_Front)
271 / m;
272 double axModel = axTires - F_aero_drag / m;
273
274 double ayTires = (std::sin(this->steeringAngles.FL) * Fx_FL + std::sin(this->steeringAngles.FR) * Fx_FR
275 + std::cos(steeringFront) * Fy_Front + Fy_Rear)
276 / m;
277 double ayModel = (ayTires);
278
279 // Calculate the rate of change of yaw rate (angular acceleration)
280 double rdotFx
281 = 0.5 * this->sf * (-Fx_FL * std::cos(this->steeringAngles.FL) + Fx_FR * std::cos(this->steeringAngles.FR))
282 + this->lf * (Fx_FL * std::sin(this->steeringAngles.FL) + Fx_FR * std::sin(this->steeringAngles.FR))
283 + 0.5 * this->sr * (Fx_RR * std::cos(this->steeringAngles.RR) - Fx_RL * std::cos(this->steeringAngles.RL))
284 - this->lr * (Fx_RL * std::sin(this->steeringAngles.RL) + Fx_RR * std::sin(this->steeringAngles.RR));
285 double rdotFy = this->lf * (Fy_Front * std::cos(steeringFront)) - this->lr * (Fy_Rear);
286 double rdot = (1 / Izz * (rdotFx + rdotFy));
287
288 // Return the calculated dynamic states
289 Eigen::Vector3d ret(axModel, ayModel, rdot);
290 return ret;
291 }

293 // Integrate the vehicle state forward in time by dt
294 void forwardIntegrate(double dt, Wheels frictionCoefficients)
295 {
296 // Calculate friction forces
297 Eigen::Vector3d friction(std::min(200.0, 2000.0 * std::abs(this->velocity.x())),
298 std::min(200.0, 2000.0 * std::abs(this->velocity.y())),
299 std::min(200.0, 2000.0 * std::abs(this->velocity.z())));
300 friction[0] = (this->velocity.x() > 0) ? friction.x() : -friction.x();
301 friction[1] = (this->velocity.y() > 0) ? friction.y() : -friction.y();
302 friction[2] = (this->velocity.z() > 0) ? friction.z() : -friction.z();
303
304 // Update position based on velocity and orientation
305 Eigen::AngleAxisd yawAngle(this->orientation.z(), Eigen::Vector3d::UnitZ());
306 this->position += (yawAngle.matrix() * this->velocity) * dt;
307
308 // Set torques to maximum torques
309 this->torques = this->maxTorques;
310
311 // Get dynamic states
312 Eigen::Vector3d xdotdyn = getDynamicStates(dt);
313
314 // Update orientation based on angular velocity
315 this->orientation += Eigen::Vector3d(0.0, 0.0, dt * angularVelocity.z());
316
317 // Update acceleration based on dynamic states and friction
318 this->acceleration = Eigen::Vector3d(xdotdyn[0] - friction.x() / m, xdotdyn[1], 0.0);
319
320 // Update angular velocity and angular acceleration
321 this->angularVelocity = (this->angularVelocity + Eigen::Vector3d(0.0, 0.0, xdotdyn[2] * dt));
322 this->angularAcceleration = Eigen::Vector3d(0.0, 0.0, xdotdyn[2]);
323
324 // Update velocity based on acceleration and angular velocity
325 this->velocity += dt * (this->acceleration - this->angularVelocity.cross(this->velocity));
326
327 // Update wheel orientations based on wheel speeds
328 this->wheelOrientations.FL = std::fmod(
329 this->wheelOrientations.FL + (this->wheelspeeds.FL / (60.0 * this->gearRatio)) * dt * 2.0 * M_PI,
330 2.0 * M_PI);
331 this->wheelOrientations.FR = std::fmod(
332 this->wheelOrientations.FR + (this->wheelspeeds.FR / (60.0 * this->gearRatio)) * dt * 2.0 * M_PI,
333 2.0 * M_PI);
334 this->wheelOrientations.RL = std::fmod(
335 this->wheelOrientations.RL + (this->wheelspeeds.RL / (60.0 * this->gearRatio)) * dt * 2.0 * M_PI,
336 2.0 * M_PI);
337 this->wheelOrientations.RR = std::fmod(
338 this->wheelOrientations.RR + (this->wheelspeeds.RR / (60.0 * this->gearRatio)) * dt * 2.0 * M_PI,
339 2.0 * M_PI);
340 }
