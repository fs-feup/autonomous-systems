129double VehicleModelBicycle::TireModel::calculateLateralForce(double slipAngle,
130 double normalForce) const {
131 if (slipAngle == 0.0) {
132 return 0.0; // No slip angle, no lateral force
133 }
134 double dpi = (p_input - NOMPRES) / NOMPRES;
135 double Fz_0_prime = LFZO * FNOM;
136 double Kya = PKY1 * Fz_0_prime * (1 + PPY1 * dpi) * (1 - PKY3 * abs(y_input)) *
137 sin(PKY4 * atan((normalForce / Fz_0_prime) /
138 ((PKY2 + PKY5 * y_input * y_input) * (1 + PPY2 * dpi)))) *
139 zeta_3 * LKY;
140 double dfz = (normalForce - Fz_0_prime) / Fz_0_prime;
141 double Kyg0 = normalForce * (PKY6 + PKY7 * dfz) * (1 + PPY5 * dpi) * LKYC;
142 double Vs_y = tan(slipAngle) * abs(V_cx);
143 double Vs_x = -slip_ratio * abs(V_cx);
144 double Vs = sqrt(Vs_x * Vs_x + Vs_y * Vs_y);
145 double V0 = LONGVL;
146 double LMUY_star = LMUY / (1.0 + LMUV * (Vs / V0));
147 double SVyg = normalForce * (PVY3 + PVY4 * dfz) * y_input * LKYC * LMUY_star * zeta_2;
148 double SHy = (PHY1 + PHY2 * dfz) +
149 ((Kyg0 * y_input - SVyg) / (Kya + eps_k * sign(Kya))) * zeta_0 + zeta_4 - 1.0;
150 double alpha_y = slipAngle + SHy;
151 double Ey = (PEY1 + PEY2 * dfz) *
152 (1 + PEY5 * y_input * y_input - (PEY3 + PEY4 * y_input) * sign(alpha_y)) * LEY;
153 double Cy = PCY1 * LCY;
154 double mu_y = (PDY1 + PDY2 * dfz) * (1.0 + PPY3 * dpi + PPY4 * dpi * dpi) *
155 (1.0 - PDY3 * y_input * y_input) * LMUY_star;
156 double Dy = mu_y * normalForce * zeta_2;
157 double By = Kya / (Cy * Dy + eps_y * sign(Dy));
158 double DVyk = mu_y * normalForce * (RVY1 + RVY2 * dfz + RVY3 * y_input) *
159 cos(atan(RVY4 * slipAngle)) * zeta_2;
160 double SVyk = DVyk * sin(RVY5 * atan(RVY6 * slip_ratio)) * LVYKA;
161 double Eyk = REY1 + REY2 * dfz;
162 double SHyk = RHY1 + RHY2 * dfz;
163 double ks = slip_ratio + SHyk;
164 double Byk = (RBY1 + RBY4 * y_input * y_input) * cos(atan(RBY2 * (slipAngle - RBY3))) * LYKA;
165 double Cyk = RCY1;
166 double Gyk_0 = cos(Cyk * atan(Byk * SHyk - Eyk * (Byk * SHyk - atan(Byk * SHyk))));
167 double Gyk = (cos(Cyk * atan(Byk * ks - Eyk * (Byk * ks - atan(Byk * ks)))) / Gyk_0);
168 double SVy = normalForce * (PVY1 + PVY2 * dfz) * LVY * LMUY_star * zeta_2 + SVyg;
169 double Fy_0 = Dy * sin(Cy * atan(By * alpha_y - Ey * (By * alpha_y - atan(By * alpha_y)))) + SVy;
170 double Fy = Gyk * Fy_0 + SVyk;
171
172 return Fy;
173}

194double VehicleModelBicycle::PowertrainModel::calculateCurrent(const Wheels& torques,
195 const Wheels& wheelspeeds,
196 double voltage) const {
197 constexpr double powerCoeff = 1.0 / 9.55;
198
199 // Calculate power for each wheel
200 double powerFL = torques.FL * wheelspeeds.FL * powerCoeff;
201 double powerFR = torques.FR * wheelspeeds.FR * powerCoeff;
202 double powerRL = torques.RL * wheelspeeds.RL * powerCoeff;
203 double powerRR = torques.RR * wheelspeeds.RR * powerCoeff;
204
205 // Calculate total power and return current
206 double totalPower = (powerFL + powerFR + powerRL + powerRR) / calculateEfficiency(torques);
207 return (totalPower / voltage);

238void VehicleModelBicycle::calculateNormalForces(double& Fz_Front, double& Fz_Rear) const {
239 double l = lr + lf;
240
241 // Calculate downforce from aerodynamics
242 double downforce = 0.0;
243 double drag = 0.0;
244 aeroModel.calculateForces(velocity.x(), downforce, drag);
245
246 // Add ground effect if requested
247 downforce += powerGroundSetpoint * powertrainModel.powerGroundForce;
248
249 // Calculate normal forces on axles with weight distribution and aero effects
250 Fz_Front = std::max(0.0, (m * GRAVITY + downforce) * 0.5 * lr / l);
251 Fz_Rear = std::max(0.0, (m * GRAVITY + downforce) * 0.5 * lf / l);

255/*
256Currently this function depends on the longitudinal forces applied on the wheels and therefore can only be called after calling calculateLongitudinalForces.
257This can be changed but it would imply calling the calculation of longitudinal forces twice.
258
259Also note that this only accounts for longitudinal weight transfer, because we are using a bycicle model , lateral weight transfer cannot be applied as we only have one wheel in each axle.
260
261To apply lateral weight transfer it would be necessary to account for each of the 4 wheels seperately.
262*/
263void VehicleModelBicycle::calculateWeightTransfer(double& Fz_Front, double& Fz_Rear , double& Fx_FL , double& Fx_FR , double& Fx_RL , double& Fx_RR) const {
264 // Calculate total longitudinal force (traction + braking)
265 double Fx_total = Fx_FL + Fx_FR + Fx_RL + Fx_RR;
266
267
268 double l = lf + lr; // wheelbase
269 double deltaW = (Fx_total * h_cg) / l;
270
271 // Positive Fx_total (acceleration): weight shifts to rear
272 // Negative Fx_total (braking): weight shifts to front
273 Fz_Front -= deltaW;
274 Fz_Rear += deltaW;
275}

278void VehicleModelBicycle::calculateSlipAngles(double& kappaFront, double& kappaRear) const {
279 constexpr double eps = 0.00001; // Prevent division by zero
280
286 bool stillstand = (velocity.norm() < VELOCITY_THRESHOLD) &&
287 (std::abs(angularVelocity.z()) < ANGULAR_VELOCITY_THRESHOLD);
288
289 // Calculate wheel positions relative to CoG
290 Eigen::Vector3d rFront(lf, 0.0, 0.0);
291 Eigen::Vector3d rRear(-lr, 0.0, 0.0);
292
293 // Calculate wheel velocities
294 Eigen::Vector3d vFront = velocity + angularVelocity.cross(rFront);
295 Eigen::Vector3d vRear = velocity + angularVelocity.cross(rRear);
296
297 // Calculate average front steering angle
298 double steeringFront = 0.5 * (steeringAngles.FL + steeringAngles.FR);
299
300 // Calculate slip angles for front and rear
301 kappaFront = std::atan2(vFront.y(), std::max(std::abs(vFront.x()), eps)) - steeringFront;
302 kappaRear = std::atan2(vRear.y(), std::max(std::abs(vRear.x()), eps));
303
304 if (stillstand) {
305 kappaFront = 0.0;
306 kappaRear = 0.0;
307 }
308}

338// New helper function to calculate longitudinal forces
339void VehicleModelBicycle::calculateLongitudinalForces(double& Fx_FL, double& Fx_FR, double& Fx_RL,
340 double& Fx_RR) const {
341 // Calculate powertrain efficiency
342 double powertrainEfficiency = powertrainModel.calculateEfficiency(torques);
343
344 // Calculate longitudinal forces on wheels
345 Fx_FL = 0; // Front wheel drive not enabled
346 Fx_FR = 0; // Front wheel drive not enabled
347 Fx_RL =
348 (powertrainModel.gearRatio * torques.RL / powertrainModel.wheelRadius) * powertrainEfficiency;
349 Fx_RR =
350 (powertrainModel.gearRatio * torques.RR / powertrainModel.wheelRadius) * powertrainEfficiency;
351
352 // Apply forces only when torque is significant or vehicle is moving
353 Fx_FL *= (abs(torques.FL) > TORQUE_THRESHOLD || velocity.x() > VELOCITY_MIN_THRESHOLD) ? 1.0 : 0.0;
354 Fx_FR *= (abs(torques.FR) > TORQUE_THRESHOLD || velocity.x() > VELOCITY_MIN_THRESHOLD) ? 1.0 : 0.0;
355 Fx_RL *= (abs(torques.RL) > TORQUE_THRESHOLD || velocity.x() > VELOCITY_MIN_THRESHOLD) ? 1.0 : 0.0;
356 Fx_RR *= (abs(torques.RR) > TORQUE_THRESHOLD || velocity.x() > VELOCITY_MIN_THRESHOLD) ? 1.0 : 0.0;
357}

360//tested
361Eigen::Vector3d VehicleModelBicycle::calculateAccelerations(double steeringFront, double Fx_FL,
362 double Fx_FR, double Fx_RL,
363 double Fx_RR, double Fy_Front,
364 double Fy_Rear, double drag,
365 const Eigen::Vector3d& friction) const {
366 // Calculate longitudinal acceleration from tire forces
367 double axTires = (std::cos(steeringAngles.FL) * Fx_FL + std::cos(steeringAngles.FR) * Fx_FR +
368 Fx_RL + Fx_RR - std::sin(steeringFront) * Fy_Front) /
369 m;
370
371 // Apply aerodynamic drag and friction to get total acceleration
372 double axModel = axTires + drag / m + friction.x() / m;
373
374 // Calculate lateral acceleration from tire forces
375 double ayTires = (std::sin(steeringAngles.FL) * Fx_FL + std::sin(steeringAngles.FR) * Fx_FR +
376 std::cos(steeringFront) * Fy_Front + Fy_Rear) /
377 m;
378 double ayModel = ayTires + friction.y() / m;
379
380 // Calculate yaw moment contributions
381 double rdotFx
382 = 0.5 * this->sf * (-Fx_FL * std::cos(this->steeringAngles.FL) + Fx_FR * std::cos(this->steeringAngles.FR))
383 + this->lf * (Fx_FL * std::sin(this->steeringAngles.FL) + Fx_FR * std::sin(this->steeringAngles.FR))
384 + 0.5 * this->sr * (Fx_RR * std::cos(this->steeringAngles.RR) - Fx_RL * std::cos(this->steeringAngles.RL))
385 - this->lr * (Fx_RL * std::sin(this->steeringAngles.RL) + Fx_RR * std::sin(this->steeringAngles.RR));
386
387 double rdotFy = lf * (Fy_Front * std::cos(steeringFront)) - lr * (Fy_Rear);
388
389 // Calculate yaw acceleration
390 double rdot = (rdotFx + rdotFy) / Izz;
391
392 return Eigen::Vector3d(axModel, ayModel, rdot);
393}

411Eigen::Vector3d VehicleModelBicycle::getDynamicStates(double dt) {
412 // Calculate wheel positions and velocities
413 double steeringFront;
414 Eigen::Vector3d rFL, rFR, rRL, rRR, rFront, rRear;
415 Eigen::Vector3d vFL, vFR, vRL, vRR;
416 calculateWheelGeometry(steeringFront, rFL, rFR, rRL, rRR, rFront, rRear, vFL, vFR, vRL, vRR);
417
418 // Calculate normal forces on axles
419 double Fz_Front, Fz_Rear;
420 calculateNormalForces(Fz_Front, Fz_Rear);
421
422 // Calculate aero drag
423 double downforce, drag;
424 aeroModel.calculateForces(velocity.x(), downforce, drag);
425
426 // Calculate rolling resistance (applied opposite to velocity direction)
427 double frict = (Fz_Front + Fz_Rear) * ROLLING_RESISTANCE_COEFF;
428 Eigen::Vector3d friction;
429
430 if (velocity.norm() < 0.01) {
431 friction = Eigen::Vector3d::Zero();
432 } else {
433 Eigen::Vector3d velDir = -velocity.normalized();
434 friction = frict * velDir;
435 }
436
437 // Calculate longitudinal forces
438 double Fx_FL, Fx_FR, Fx_RL, Fx_RR;
439 calculateLongitudinalForces(Fx_FL, Fx_FR, Fx_RL, Fx_RR);
440
441 // Calculate weight transfer based on longitudinal forces -> this works because fz front and rear were calculate above
442 calculateWeightTransfer(Fz_Front, Fz_Rear, Fx_FL, Fx_FR, Fx_RL, Fx_RR);
443
444 // Calculate slip angles
445 double kappaFront, kappaRear;
446 calculateSlipAngles(kappaFront, kappaRear);
447
448 // Calculate lateral forces using tire model
449 double Fy_Front = processSlipAngleLat(kappaFront, Fz_Front / 2) * 2;
450 double Fy_Rear = processSlipAngleLat(kappaRear, Fz_Rear / 2) * 2;
451
452 // Update wheel speeds
453 updateWheelSpeeds(vFL, vFR, vRL, vRR);
454
455 RCLCPP_DEBUG(rclcpp::get_logger("a"),
456 "Vx: %f, Vy: %f, W: %f, KappaF: %f, KappaR: %f, Fy_Front: %f, Fy_Rear: %f, drag: %f, friction: %f", velocity.x(),
457 velocity.y(), angularVelocity.z(), kappaFront, kappaRear, Fy_Front, Fy_Rear, drag, friction.x());
458
459 // Calculate accelerations
460 return calculateAccelerations(steeringFront, Fx_FL, Fx_FR, Fx_RL, Fx_RR, Fy_Front, Fy_Rear, drag,
461 friction);
462}

464void VehicleModelBicycle::forwardIntegrate(double dt) {
465
466 // Convert throttle to wheel torques (using PowertrainModel)
467 powertrainModel.calculateWheelTorques(throttleActuation, torques);
468
469 // Calculate dynamic states
470 Eigen::Vector3d dynamicStates = getDynamicStates(dt);
471
472 // Update acceleration from dynamic states
473 acceleration = Eigen::Vector3d(dynamicStates.x(), dynamicStates.y(), 0.0);
474
475 // Update angular velocity and acceleration
476 angularVelocity.z() += dynamicStates.z() * dt;
477 angularAcceleration.z() = dynamicStates.z();
478
479 // Update velocity with acceleration and angular velocity effects
480 velocity += dt * (acceleration - angularVelocity.cross(velocity));
481
482 // Update position based on velocity and orientation
483 Eigen::AngleAxisd yawAngle(orientation.z(), Eigen::Vector3d::UnitZ());
484 position += (yawAngle.matrix() * velocity) * dt;
485
486 // Update orientation based on angular velocity
487 orientation.z() += dt * angularVelocity.z();
488
489 // Update wheel orientations
490 const double wheelRotationFactor = TWO_PI / (60.0 * powertrainModel.gearRatio);
491
492 wheelOrientations.FL =
493 std::fmod(wheelOrientations.FL + wheelspeeds.FL * dt * wheelRotationFactor, TWO_PI);
494 wheelOrientations.FR =
495 std::fmod(wheelOrientations.FR + wheelspeeds.FR * dt * wheelRotationFactor, TWO_PI);
496 wheelOrientations.RL =
497 std::fmod(wheelOrientations.RL + wheelspeeds.RL * dt * wheelRotationFactor, TWO_PI);
498 wheelOrientations.RR =
499 std::fmod(wheelOrientations.RR + wheelspeeds.RR * dt * wheelRotationFactor, TWO_PI);
500}
