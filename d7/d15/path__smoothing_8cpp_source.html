37bool PathSmoothing::vector_direction(PathPoint *c1, PathPoint *c2, float prev_vx, float prev_vy) {
38 float vx = c2->getX() - c1->getX();
39 float vy = c2->getY() - c1->getY();
40 // Cos(angle) higher than 0 yhe vectors are pointing to the same half
41 return (vx * prev_vx + vy * prev_vy) > 0;

44std::vector<PathPoint *> PathSmoothing::orderPath(std::vector<PathPoint *> *unord_path) {
45 // Order Pathpoint_array
46 // The algorithm works by iteratively selecting the nearest unvisited
47 // path_point to the current path_point and updating the traversal direction accordingly.
48 // This approach creates an ordered sequence that efficiently connects nearby
49 // path points.
50 std::vector<std::pair<PathPoint *, bool>> new_unord_path;
51 for (int i = 0; i < static_cast<int>(unord_path->size()); i++)
52 new_unord_path.push_back(std::make_pair((*unord_path)[i], false));
53
54 // Values for first iteration
55 std::vector<PathPoint *> path;
56 PathPoint *path_point1 = new PathPoint(0, 0, 0);
57 float vx = 1;
58 float vy = 0;
59
60 for (int iter_number = 0; iter_number < static_cast<int>(unord_path->size()); iter_number++) {
61 float min_dist = MAXFLOAT;
62 int min_index = 0;
63
64 for (int i = 0; i < static_cast<int>(new_unord_path.size()); i++) {
65 PathPoint *path_point2 = new_unord_path[i].first;
66 // Check only if not visited
67 if (new_unord_path[i].second == false ||
68 (iter_number == 0 && vector_direction(path_point1, path_point2, vx, vy))) {
69 // first iteration we assure the direction is correct to avoid going
70 // backwards
71
72 float new_dist = path_point1->getDistanceTo(path_point2);
73 if (new_dist < min_dist) {
74 min_dist = new_dist;
75 min_index = i;
76 }
77 }
78 }
79 new_unord_path[min_index].second = true; // mark as visited
80
81 // new visited path points is the reference for the next search. Arrays and path point
82 // updated
83 vx = new_unord_path[min_index].first->getX() - path_point1->getX();
84 vy = new_unord_path[min_index].first->getY() - path_point1->getY();
85 path_point1 = new_unord_path[min_index].first;
86
87 path.push_back(new_unord_path[min_index].first); // add path points to ordered sequence
88 }
89 return path;
90}

97std::vector<PathPoint *> PathSmoothing::splineSmoother(int precision, int order, float coeffs_ratio,
98 std::vector<PathPoint *> path) {
99 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "START splineSmoother with %i path points",
100 static_cast<int>(path.size()));
101
102 const int n = path.size();
103 const int ncoeffs = n / coeffs_ratio; // n > = ncoeffs
104 const int nbreak = ncoeffs - order + 2;
105
106 // Initialize vars (pointers)
107 gsl_bspline_workspace *bw, *cw;
108 gsl_vector *B, *C;
109 gsl_vector *c, *c2, *w;
110 gsl_vector *x_values, *y_values, *i_values;
111 gsl_matrix *X, *Y, *cov, *cov2;
112 gsl_multifit_linear_workspace *mw, *mw2;
113 double chisq, chisq2;
114
115 // allocate memory for the actual objects the pointers will point to
116 bw = gsl_bspline_alloc(order, nbreak);
117 cw = gsl_bspline_alloc(order, nbreak);
118 B = gsl_vector_alloc(ncoeffs);
119 C = gsl_vector_alloc(ncoeffs);
120 i_values = gsl_vector_alloc(n);
121 x_values = gsl_vector_alloc(n);
122 y_values = gsl_vector_alloc(n);
123 X = gsl_matrix_alloc(n, ncoeffs);
124 Y = gsl_matrix_alloc(n, ncoeffs);
125 c = gsl_vector_alloc(ncoeffs);
126 c2 = gsl_vector_alloc(ncoeffs);
127 w = gsl_vector_alloc(n);
128 cov = gsl_matrix_alloc(ncoeffs, ncoeffs);
129 cov2 = gsl_matrix_alloc(ncoeffs, ncoeffs);
130 mw = gsl_multifit_linear_alloc(n, ncoeffs);
131 mw2 = gsl_multifit_linear_alloc(n, ncoeffs);
132
133 // Set spline data
134 for (int i = 0; i < n; i++) {
135 gsl_vector_set(i_values, i, i);
136 gsl_vector_set(x_values, i, path[i]->getX());
137 gsl_vector_set(y_values, i, path[i]->getY());
138 // closer cones more important(better stability)
139 gsl_vector_set(w, i, 1.0 / pow(path[i]->getDistanceTo(path[(i + 1) % n]), 2));
140 }
141
142 // Set i range within cone set length
143 gsl_bspline_knots_uniform(0, n, bw);
144 gsl_bspline_knots_uniform(0, n, cw);
145
146 /* construct the fit matrix X */
147 for (int i = 0; i < n; i++) {
148 /* compute B_j(xi) for all j */
149 gsl_bspline_eval(i, B, bw);
150 gsl_bspline_eval(i, C, cw);
151
152 /* fill in row i of X */
153 for (int j = 0; j < ncoeffs; j++) {
154 double Bj = gsl_vector_get(B, j);
155 gsl_matrix_set(X, i, j, Bj);
156 double Cj = gsl_vector_get(C, j);
157 gsl_matrix_set(Y, i, j, Cj);
158 }
159 }
160
161 // Fit spline
162 gsl_multifit_wlinear(X, w, x_values, c, cov, &chisq, mw);
163 gsl_multifit_wlinear(Y, w, y_values, c2, cov2, &chisq2, mw2);
164
165 // Initialize variables for subsequent spline evaluation
166 double xi, yi, yerr, yerr2;
167 std::vector<double> i_eval, x_eval, y_eval;
168 std::vector<PathPoint *> path_eval;
169
170 // Calculate the desired amount of spline points and add them to "path_eval"
171 for (int i = 0; i < n; i++) {
172 for (int j = 0; j < precision; j += 1) { // iterate over decimal numbers
173 gsl_bspline_eval(i + static_cast<float>(j) / precision, B, bw);
174 gsl_bspline_eval(i + static_cast<float>(j) / precision, C, cw);
175 gsl_multifit_linear_est(B, c, cov, &xi, &yerr);
176 gsl_multifit_linear_est(C, c2, cov2, &yi, &yerr2);
177 i_eval.push_back(i);
178 x_eval.push_back(xi);
179 y_eval.push_back(yi);
180 PathPoint *path_point = new PathPoint(xi, yi);
181 path_eval.push_back(path_point);
182 if (j == 0 && i == n - 1) {
183 break; // Decimals can't go over last int
184 }
185 }
186 }
187
188 // Free Memory
189 gsl_bspline_free(bw);
190 gsl_bspline_free(cw);
191 gsl_vector_free(B);
192 gsl_vector_free(C);
193 gsl_vector_free(i_values);
194 gsl_vector_free(x_values);
195 gsl_vector_free(y_values);
196 gsl_matrix_free(X);
197 gsl_matrix_free(Y);
198 gsl_vector_free(c);
199 gsl_vector_free(w);
200 gsl_matrix_free(cov);
201 gsl_multifit_linear_free(mw);
202 gsl_multifit_linear_free(mw2);
203
204 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "END fitSpline with %i points",
205 static_cast<int>(path_eval.size()));
206
207 return path_eval;
208}
