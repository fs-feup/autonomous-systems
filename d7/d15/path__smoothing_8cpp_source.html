40bool PathSmoothing::vector_direction(PathPoint *c1, PathPoint *c2, float prev_vx, float prev_vy) {
41 float vx = c2->getX() - c1->getX();
42 float vy = c2->getY() - c1->getY();
43 // Cos(angle) higher than 0 yhe vectors are pointing to the same half
44 return (vx * prev_vx + vy * prev_vy) > 0;

47std::vector<PathPoint *> PathSmoothing::orderPath(std::vector<PathPoint *> *unord_path) {
48 // Order Pathpoint_array
49 // The algorithm works by iteratively selecting the nearest unvisited
50 // path_point to the current path_point and updating the traversal direction accordingly.
51 // This approach creates an ordered sequence that efficiently connects nearby
52 // path points.
53 std::vector<std::pair<PathPoint *, bool>> new_unord_path;
54 for (int i = 0; i < static_cast<int>(unord_path->size()); i++)
55 new_unord_path.push_back(std::make_pair((*unord_path)[i], false));
56
57 // Values for first iteration
58 std::vector<PathPoint *> ordered_path;
59 auto path_point1 = std::make_shared<PathPoint>(0, 0, 0);
60 float vx = 1;
61 float vy = 0;
62
63 for (int iter_number = 0; iter_number < static_cast<int>(unord_path->size()); iter_number++) {
64 float min_dist = MAXFLOAT;
65 int min_index = 0;
66
67 for (int i = 0; i < static_cast<int>(new_unord_path.size()); i++) {
68 PathPoint *path_point2 = new_unord_path[i].first;
69 // Check only if not visited
70 if (new_unord_path[i].second == false ||
71 (iter_number == 0 && vector_direction(path_point1.get(), path_point2, vx, vy))) {
72 // first iteration we assure the direction is correct to avoid going
73 // backwards
74
75 float new_dist = path_point1->getDistanceTo(path_point2);
76 if (new_dist < min_dist) {
77 min_dist = new_dist;
78 min_index = i;
79 }
80 }
81 }
82 new_unord_path[min_index].second = true; // mark as visited
83
84 // new visited path points is the reference for the next search. Arrays and path point
85 // updated
86 vx = new_unord_path[min_index].first->getX() - path_point1->getX();
87 vy = new_unord_path[min_index].first->getY() - path_point1->getY();
88 path_point1 = std::make_shared<PathPoint>(new_unord_path[min_index].first->getX(),
89 new_unord_path[min_index].first->getY());
90
91 ordered_path.push_back(new_unord_path[min_index].first); // add path points to ordered sequence
92 }
93 return ordered_path;
94}

103std::vector<PathPoint *> PathSmoothing::splineSmoother(int precision, int order, float coeffs_ratio,
104 std::vector<PathPoint *> input_path) {
105 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "START splineSmoother with %i path points",
106 static_cast<int>(input_path.size()));
107
108 // Remove duplicates that otherwise would cause the spline to fail
109 std::unordered_set<PathPoint *> seen;
110 auto iterator = std::remove_if(input_path.begin(), input_path.end(), [&seen](const auto &ptr) {
111 return !seen.insert(ptr).second; // insert returns a pair, where .second is false if the
112 // element already existed
113 });
114 input_path.erase(iterator, input_path.end());
115
116 // Initialize values after removing duplicates
117 const int n = input_path.size();
118 const int ncoeffs = n / coeffs_ratio; // n > = ncoeffs
119 const int nbreak = ncoeffs - order + 2;
120
121 if (nbreak < 2 || input_path.size() < 1) {
122 RCLCPP_WARN(rclcpp::get_logger("rclcpp"),
123 "Too few points to calculate spline. Number of different path points was %i", n);
124 return input_path;
125 }
126
127 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"),
128 "Fitting spline in path smoothing : nbreak= %i; n (number of path points) = %i; "
129 "ncoeffs = %i;\n",
130 nbreak, n, ncoeffs);
131
132 // Initialize vars (pointers)
133 gsl_bspline_workspace *bw, *cw;
134 gsl_vector *B, *C;
135 gsl_vector *c, *c2, *w;
136 gsl_vector *x_values, *y_values, *i_values;
137 gsl_matrix *X, *Y, *cov, *cov2;
138 gsl_multifit_linear_workspace *mw, *mw2;
139 double chisq, chisq2;
140
141 // allocate memory for the actual objects the pointers will point to
142 bw = gsl_bspline_alloc(order, nbreak);
143 cw = gsl_bspline_alloc(order, nbreak);
144 B = gsl_vector_alloc(ncoeffs);
145 C = gsl_vector_alloc(ncoeffs);
146
147 i_values = gsl_vector_alloc(n);
148 x_values = gsl_vector_alloc(n);
149 y_values = gsl_vector_alloc(n);
150
151 X = gsl_matrix_alloc(n, ncoeffs);
152 Y = gsl_matrix_alloc(n, ncoeffs);
153 c = gsl_vector_alloc(ncoeffs);
154 c2 = gsl_vector_alloc(ncoeffs);
155 w = gsl_vector_alloc(n);
156 cov = gsl_matrix_alloc(ncoeffs, ncoeffs);
157 cov2 = gsl_matrix_alloc(ncoeffs, ncoeffs);
158 mw = gsl_multifit_linear_alloc(n, ncoeffs);
159 mw2 = gsl_multifit_linear_alloc(n, ncoeffs);
160
161 // Set spline data
162 for (int i = 0; i < n; i++) {
163 gsl_vector_set(i_values, i, i);
164 gsl_vector_set(x_values, i, input_path[i]->getX());
165 gsl_vector_set(y_values, i, input_path[i]->getY());
166 // closer points more important(better stability)
167 gsl_vector_set(w, i, 1.0 / pow(input_path[i]->getDistanceTo(input_path[(i + 1) % n]), 2));
168 }
169
170 // Set i range within input path length
171 gsl_bspline_knots_uniform(0, n, bw);
172 gsl_bspline_knots_uniform(0, n, cw);
173
174 /* construct the fit matrix X */
175 for (int i = 0; i < n; i++) {
176 /* compute B_j(xi) for all j */
177 gsl_bspline_eval(i, B, bw);
178 gsl_bspline_eval(i, C, cw);
179
180 /* fill in row i of X */
181 for (int j = 0; j < ncoeffs; j++) {
182 double Bj = gsl_vector_get(B, j);
183 gsl_matrix_set(X, i, j, Bj);
184 double Cj = gsl_vector_get(C, j);
185 gsl_matrix_set(Y, i, j, Cj);
186 }
187 }
188
189 // Fit spline
190 gsl_multifit_wlinear(X, w, x_values, c, cov, &chisq, mw);
191 gsl_multifit_wlinear(Y, w, y_values, c2, cov2, &chisq2, mw2);
192
193 // Initialize variables for subsequent spline evaluation
194 double xi, yi, yerr, yerr2;
195 std::vector<double> i_eval, x_eval, y_eval;
196 std::vector<PathPoint *> final_path;
197
198 // Calculate the desired amount of spline points and add them to "final_path"
199 for (int i = 0; i < n; i++) {
200 for (int j = 0; j < precision; j += 1) { // iterate over decimal numbers
201 gsl_bspline_eval((double)i + static_cast<double>(j) / precision, B, bw);
202 gsl_bspline_eval((double)i + static_cast<double>(j) / precision, C, cw);
203 gsl_multifit_linear_est(B, c, cov, &xi, &yerr);
204 gsl_multifit_linear_est(C, c2, cov2, &yi, &yerr2);
205 i_eval.push_back(i);
206 x_eval.push_back(xi);
207 y_eval.push_back(yi);
208 PathPoint *new_path_point = new PathPoint(xi, yi);
209 final_path.push_back(new_path_point);
210 if (j == 0 && i == n - 1) {
211 break; // Decimals can't go over last int
212 }
213 }
214 }
215
216 // Free Memory
217 gsl_bspline_free(bw);
218 gsl_bspline_free(cw);
219 gsl_vector_free(B);
220 gsl_vector_free(C);
221 gsl_vector_free(i_values);
222 gsl_vector_free(x_values);
223 gsl_vector_free(y_values);
224 gsl_matrix_free(X);
225 gsl_matrix_free(Y);
226 gsl_vector_free(c);
227 gsl_vector_free(w);
228 gsl_matrix_free(cov);
229 gsl_multifit_linear_free(mw);
230 gsl_multifit_linear_free(mw2);
231
232 RCLCPP_DEBUG(rclcpp::get_logger("rclcpp"), "END splineSmoother with %i points",
233 static_cast<int>(final_path.size()));
234
235 // To access spline points uncomment these lines
236 // std::cout << "Smoothing spline : " << std::endl;
237 // for (int i = 0; i < static_cast<int>(final_path.size()); i++) {
238 // std::cout << "(" << final_path[i]->getX() << "," << final_path[i]->getY() << "),";
239 // }
240 // std::cout << std::endl;
241
242 return final_path;
243}
