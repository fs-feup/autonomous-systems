125std::vector<PathPoint> processTriangulations(std::string filename) {
126 Track *track = new Track();
127 track->fillTrack(filename);
128
129 LocalPathPlanner *pathplanner = new LocalPathPlanner();
130 std::vector<PathPoint> path;
131 std::ofstream measuresPath = openWriteFile("performance/exec_time/planning/planning_" +
132 get_current_date_time_as_string() + ".csv");
133 double total_time = 0;
134 int no_iters = 100;
135
136 // No_iters repetitions to get average
137 for (int i = 0; i < no_iters; i++) {
138 auto t0 = std::chrono::high_resolution_clock::now();
139
140 std::vector<PathPoint *> pathPointers = pathplanner->processNewArray(track);
141
142 auto t1 = std::chrono::high_resolution_clock::now();
143
144 double elapsed_time_ms = std::chrono::duration<double, std::milli>(t1 - t0).count();
145
146 total_time += elapsed_time_ms;
147
148 for (size_t i = 0; i < pathPointers.size(); i++) path.push_back(*pathPointers[i]);
149 }
150
151 measuresPath << total_time / no_iters << "\n";
152 measuresPath.close();
153
154 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "Average Delaunay Triangulations processed in %f ms.",
155 (total_time / no_iters));
156
157 return path;
158}
