28bool customComparator(const std::pair<double, double> &a, const std::pair<double, double> &b) {
29 if (a.first != b.first) {
30 return a.first < b.first;
31 }
32 return a.second < b.second;
33}

41std::vector<std::pair<double, double>> orderVectorOfPairs(
42 const std::vector<std::pair<double, double>> &vec) {
43 std::vector<std::pair<double, double>> result = vec;
44 std::sort(result.begin(), result.end(), customComparator);

70float round_n(float num, int decimal_places) {
71 num *= pow(10, decimal_places);
72 int intermediate = round(num);
73 num = intermediate / pow(10, decimal_places);
74 return num;

84std::vector<PathPoint> processTriangulations(std::string filename) {
85 Track *track = new Track();
86 track->fillTrack(filename);
87
88 LocalPathPlanner *pathplanner = new LocalPathPlanner();
89 std::vector<PathPoint> path;
90 std::ofstream measuresPath = openWriteFile("performance/exec_time/planning/planning_" +
91 getCurrentDateTimeAsString() + ".csv");
92 double total_time = 0;
93 int no_iters = 100;
94
95 // No_iters repetitions to get average
96 for (int i = 0; i < no_iters; i++) {
97 auto t0 = std::chrono::high_resolution_clock::now();
98
99 std::vector<PathPoint *> pathPointers = pathplanner->processNewArray(track);
100
101 auto t1 = std::chrono::high_resolution_clock::now();
102
103 double elapsed_time_ms = std::chrono::duration<double, std::milli>(t1 - t0).count();
104
105 total_time += elapsed_time_ms;
106
107 for (size_t i = 0; i < pathPointers.size(); i++) path.push_back(*pathPointers[i]);
108 }
109
110 measuresPath << total_time / no_iters << "\n";
111 measuresPath.close();
112
113 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "Average Delaunay Triangulations processed in %f ms.",
114 (total_time / no_iters));
115
116 return path;
117}

126void outlierCalculations(std::string filename, int num_outliers = 0) {
127 Track *track = new Track();
128 track->fillTrack(filename); // fill track with file data
129
130 // Remove outliers no_iters times to get average
131 int no_iters = 100;
132 double total_time = 0;
133
134 std::ofstream measuresPath = openWriteFile(
135 "performance/exec_time/planning/planning_" + getCurrentDateTimeAsString() + ".csv",
136 "Number of Left Cones,Number of Right Cones,Number of Outliers,Outliers Removal Execution "
137 "Time,Triangulations Execution Time");
138
139 for (int i = 0; i < no_iters; i++) {
140 track->reset();
141 track->fillTrack(filename);
142 auto t0 = std::chrono::high_resolution_clock::now();
143 track->validateCones();
144 auto t1 = std::chrono::high_resolution_clock::now();
145 double elapsed_time_iter_ms = std::chrono::duration<double, std::milli>(t1 - t0).count();
146 total_time += elapsed_time_iter_ms;
147 }
148
149 measuresPath << track->getLeftConesSize() << "," << track->getRightConesSize() << ","
150 << num_outliers << "," << total_time / no_iters << ",";
151 measuresPath.close();
152
153 RCLCPP_INFO(rclcpp::get_logger("rclcpp"), "Outliers removed in average %f ms.",
154 (total_time / no_iters));
155}
