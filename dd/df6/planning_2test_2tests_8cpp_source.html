28bool customComparator(const std::pair<double, double> &a,
29 const std::pair<double, double> &b) {
30 if (a.first != b.first) {
31 return a.first < b.first;
32 }
33 return a.second < b.second;
34}

42std::vector<std::pair<double, double>> orderVectorOfPairs(
43 const std::vector<std::pair<double, double>> &vec) {
44 std::vector<std::pair<double, double>> result = vec;
45 std::sort(result.begin(), result.end(), customComparator);

71float round_n(float num, int decimal_places) {

85std::vector<PathPoint> processTriangulations(std::string filename) {
86 Track *track = new Track();
87 track->fillTrack(filename);
88
89 LocalPathPlanner *pathplanner = new LocalPathPlanner();
90 std::vector<PathPoint> path;
91 std::ofstream measuresPath =
92 openWriteFile("performance/exec_time/planning/planning_" +
93 get_current_date_time_as_string() + ".csv");
94 double total_time = 0;
95 int no_iters = 100;
96
97 // No_iters repetitions to get average
98 for (int i = 0; i < no_iters; i++) {
99 auto t0 = std::chrono::high_resolution_clock::now();
100
101 std::vector<PathPoint *> pathPointers = pathplanner->processNewArray(track);
102
103 auto t1 = std::chrono::high_resolution_clock::now();
104
105 double elapsed_time_ms =
106 std::chrono::duration<double, std::milli>(t1 - t0).count();
107
108 total_time += elapsed_time_ms;
109
110 for (size_t i = 0; i < pathPointers.size(); i++)
111 path.push_back(*pathPointers[i]);
112 }
113
114 measuresPath << total_time / no_iters << "\n";
115 measuresPath.close();
116
117 RCLCPP_INFO(rclcpp::get_logger("rclcpp"),
118 "Average Delaunay Triangulations processed in %f ms.",
119 (total_time / no_iters));
120
121 return path;
122}

131void outlierCalculations(std::string filename, int num_outliers = 0) {
132 Track *track = new Track();
133 track->fillTrack(filename); // fill track with file data
134
135 // Remove outliers no_iters times to get average
136 int no_iters = 100;
137 double total_time = 0;
138
139 std::ofstream measuresPath =
140 openWriteFile("performance/exec_time/planning/planning_" +
141 get_current_date_time_as_string() + ".csv",
142 "Number of Left Cones,Number of Right Cones,Number of "
143 "Outliers,Outliers Removal Execution "
144 "Time,Triangulations Execution Time");
145
146 for (int i = 0; i < no_iters; i++) {
147 track->reset();
148 track->fillTrack(filename);
149 auto t0 = std::chrono::high_resolution_clock::now();
150 track->validateCones();
151 auto t1 = std::chrono::high_resolution_clock::now();
152 double elapsed_time_iter_ms =
153 std::chrono::duration<double, std::milli>(t1 - t0).count();
154 total_time += elapsed_time_iter_ms;
155 }
156
157 measuresPath << track->getLeftConesSize() << "," << track->getRightConesSize()
158 << "," << num_outliers << "," << total_time / no_iters << ",";
159 measuresPath.close();
160
161 RCLCPP_INFO(rclcpp::get_logger("rclcpp"),
162 "Outliers removed in average %f ms.", (total_time / no_iters));
163}
