59void RosCan::control_callback(fs_msgs::msg::ControlCommand::SharedPtr controlCmd) {
60 // Check if the steering value is within the limits
61 if (controlCmd->steering < STEERING_LOWER_LIMIT || controlCmd->steering > STEERING_UPPER_LIMIT) {
62 RCLCPP_ERROR(this->get_logger(), "Steering value out of range");
63 return;
64 }
65 // Check if the throttle value is within the limits
66 if (controlCmd->throttle < THROTTLE_LOWER_LIMIT || controlCmd->throttle > THROTTLE_UPPER_LIMIT) {
67 RCLCPP_ERROR(this->get_logger(), "Throttle value out of range");
68 return;
69 }
70 if (currentState == State::AS_Driving) {
71 RCLCPP_DEBUG(this->get_logger(), "State is Driving: Steering: %f, Throttle: %f",
72 controlCmd->steering, controlCmd->throttle);
73 canInitializeLibrary(); // initialize the CAN library again, just in case (could be removed)
74 // Prepare the steering message
75 long steering_id = STEERING_CUBEM_ID;
76 void* steering_requestData = reinterpret_cast<void*>(&controlCmd->steering);
77 unsigned int steering_dlc = 8;
78 unsigned int flag = 0;
79
80 // Write the steering message to the CAN bus
81 stat = canWrite(hnd, steering_id, steering_requestData, steering_dlc, flag);
82 if (stat != canOK) {
83 RCLCPP_ERROR(this->get_logger(), "Failed to write steering to CAN bus");
84 }
85
86 // Prepare the throttle message
87 long throttle_id = BAMO_RECEIVER; // TODO: confirm ID
88 void* throttle_requestData = reinterpret_cast<void*>(&controlCmd->throttle);
89 unsigned int throttle_dlc = 8;

151void RosCan::canSniffer() {
152 long id;
153 unsigned char msg[8];
154 unsigned int dlc;
155 unsigned int flag;
156 unsigned long time;
157
158 stat = canRead(hnd, &id, &msg, &dlc, &flag, &time);
159 while (stat == canOK) {
160 // Convert the CAN message to a ROS message and publish it
161 canInterpreter(id, msg, dlc, flag, time);
162
163 stat = canRead(hnd, &id, &msg, &dlc, &flag, &time);

167void RosCan::canInterpreter(long id, unsigned char msg[8], unsigned int dlc, unsigned int flag,
168 unsigned long time) {
169 switch (id) {
170 case MASTER_STATUS:
171 canInterpreterMasterStatus(msg);
172 break;
173 case IMU_YAW_RATE_ACC_Y_ID:
174 imuYawAccYPublisher(msg);
175 break;
176 case IMU_ROLL_RATE_ACC_X_ID:
177 imuRollAccXPublisher(msg);
178 break;
179 case IMU_PITCH_RATE_ACC_Z_ID:
180 imuPitchAccZPublisher(msg);
181 break;
182 case TEENSY_C1:
183 if (msg[0] == TEENSY_C1_RR_RPM_CODE) rrRPMPublisher(msg);
184 break;
185 case STEERING_CUBEM_ID:
186 steeringAngleCubeMPublisher(msg);
187 break;
188 case STEERING_BOSCH_ID:
189 steeringAngleBoschPublisher(msg);
190 default:
191 break;
192 }
193}

195void RosCan::canInterpreterMasterStatus(unsigned char msg[8]) {
196 switch (msg[0]) {
197 case MASTER_AS_STATE_CODE: {
198 if (msg[1] == 3) // If AS State == Driving
199 this->goSignal = 1;
200 else
201 this->goSignal = 0;
202 opStatusPublisher();
203 break;
204 }
205 case MASTER_AS_MISSION_CODE: {
206 this->asMission = msg[1];
207 opStatusPublisher();
208 break;
209 }
210 case MASTER_RL_RPM_CODE: {
211 rlRPMPublisher(msg);
212 break;
213 }
214 default:
215 break;
216 }

284 // Mask to check the first 3 bits
285 char mask = 0b00000111;
286
287 // Extract the first 3 bits
288 char firstThreeBits = msg[3] & mask;
289
290 if (firstThreeBits == mask) {
