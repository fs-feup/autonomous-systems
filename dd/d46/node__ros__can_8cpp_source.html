159void RosCan::canInterperter(long id, unsigned char msg[8], unsigned int dlc, unsigned int flag,
160 unsigned long time) {
161 switch (id) {
162 case MASTER_STATUS:
163 switch (msg[0]) {
164 case 0x31: // Current AS State
165 asStatePublisher(msg[1]);
166 currentState = static_cast<State>(
167 msg[1]); // TODO: check if static casting is the best solution here
168 break;
169 case 0x32: // Current AS Mission
170 asMissionPublisher(msg[1]);
171 case 0x33: // Left Wheel RPM
172 leftWheelPublisher(msg[1], msg[2]);
173 break;
174 default:
175 break;
176 }
177 break;
178
179 case MASTER_IMU:
180 imuPublisher(msg);
181 break;
182 case TEENSY_C1:
183 if (msg[0] == 0x11) rightWheelPublisher(msg[1], msg[2]);
184 break;
185 case STEERING_ID:
186 steeringAnglePublisher(msg[1], msg[2]);
187 break;
188
189 default:
190 break;
191 }
192}
