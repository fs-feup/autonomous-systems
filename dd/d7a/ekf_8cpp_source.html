12bool ExtendedKalmanFilter::cone_match(const double x_from_state, const double y_from_state,
13 const double x_from_perception,
14 const double y_from_perception,
15 const double distance_to_vehicle) {
16 double delta_x = x_from_state - x_from_perception;
17 double delta_y = y_from_state - y_from_perception;
18 double delta = std::sqrt(std::pow(delta_x, 2) + std::pow(delta_y, 2));
19 auto limit_function = [](double distance) {
20 double curvature = 8.0;
21 double initial_limit = 0.5;
22 return pow(M_E, distance / curvature) - (1 - initial_limit);
23 };
24 return limit_function(distance_to_vehicle) > delta;
25}

58void ExtendedKalmanFilter::prediction_step(const MotionUpdate &motion_update) {
59 std::chrono::time_point<std::chrono::high_resolution_clock> now =
60 std::chrono::high_resolution_clock::now();
61 double delta =
62 std::chrono::duration_cast<std::chrono::microseconds>(now - this->_last_update).count();
63 this->_last_motion_update = motion_update;
64 Eigen::VectorXf tempX = this->X;
65 this->X = this->_motion_model.predict_expected_state(tempX, motion_update, delta / 1000000);
66 Eigen::MatrixXf G =
67 this->_motion_model.get_motion_to_state_matrix(tempX, motion_update, delta / 1000000);
68 Eigen::MatrixXf R = this->_motion_model.get_process_noise_covariance_matrix(
69 this->X.size()); // Process Noise Matrix
70 this->P = G * this->P * G.transpose() + R;
71 this->_last_update = now;
72}

74void ExtendedKalmanFilter::correction_step(const ConeMap &perception_map) {
75 for (auto cone : perception_map.map) {
76 ObservationData observation_data = ObservationData(cone.first.x, cone.first.y, cone.second);
77 int landmark_index = this->discovery(observation_data);
78 if (landmark_index == -1) { // Too far away landmark
79 continue;
80 }
81 Eigen::MatrixXf H = this->_observation_model.get_state_to_observation_matrix(
82 this->X, landmark_index, this->X.size());
83 Eigen::MatrixXf Q =
84 this->_observation_model.get_observation_noise_covariance_matrix(); // Observation Noise
85 // Matrix
86 Eigen::MatrixXf K = this->get_kalman_gain(H, this->P, Q);
87 Eigen::Vector2f z_hat = this->_observation_model.observation_model(this->X, landmark_index);
88 Eigen::Vector2f z = Eigen::Vector2f(observation_data.position.x, observation_data.position.y);
89 this->X = this->X + K * (z - z_hat);
90 this->P = (Eigen::MatrixXf::Identity(this->P.rows(), this->P.cols()) - K * H) * this->P;
91 }
92}

94Eigen::MatrixXf ExtendedKalmanFilter::get_kalman_gain(const Eigen::MatrixXf &H,
95 const Eigen::MatrixXf &P,
96 const Eigen::MatrixXf &Q) {
97 Eigen::MatrixXf S = H * P * H.transpose() + Q;
98 Eigen::MatrixXf K = P * H.transpose() * S.inverse();
99 return K;
100}

102int ExtendedKalmanFilter::discovery(const ObservationData &observation_data) {
103 Eigen::Vector2f landmark_absolute =
104 this->_observation_model.inverse_observation_model(this->X, observation_data);
105 double distance =
106 std::sqrt(pow(observation_data.position.x, 2) + pow(observation_data.position.y, 2));
107 if (distance > ExtendedKalmanFilter::max_landmark_distance) {
108 return -1;
109 }
110 double best_delta = 1000000000.0;
111 int best_index = -1;
112 for (int i = 3; i < this->X.size() - 1; i += 2) {
113 double delta_x = this->X(i) - landmark_absolute(0);
114 double delta_y = this->X(i + 1) - landmark_absolute(1);
115 double delta = std::sqrt(std::pow(delta_x, 2) + std::pow(delta_y, 2));
116 if (delta < best_delta && this->_colors[(i - 3) / 2] == observation_data.color) {
117 best_index = i;
118 best_delta = delta;
119 }
120 }
121 if (best_index != -1) {
122 double score =
123 ExtendedKalmanFilter::cone_match(this->X(best_index), this->X(best_index + 1),
124 landmark_absolute(0), landmark_absolute(1), distance);
125 if (score > 0) {
126 return best_index;
127 }
128 }
129
130 // If not found, add to the map
131 this->X.conservativeResizeLike(Eigen::VectorXf::Zero(this->X.size() + 2));
132 this->X(this->X.size() - 2) = landmark_absolute(0);
133 this->X(this->X.size() - 1) = landmark_absolute(1);
134 this->P.conservativeResizeLike(Eigen::MatrixXf::Zero(this->P.rows() + 2, this->P.cols() + 2));
135 this->_colors.push_back(observation_data.color);
136 return this->X.size() - 2;
137}
