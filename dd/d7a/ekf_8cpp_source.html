23void ExtendedKalmanFilter::prediction_step(const MotionUpdate &motion_update,
24 const std::string &sensor_type) {
25 std::shared_ptr<MotionModel> motion_model = _motion_models_[sensor_type];
26 rclcpp::Time temp_this_last_update = this->_last_update_;
27 this->_last_update_ = motion_update.last_update;
28 if (this->_first_prediction_) { // To set the last update correctly
29 this->_first_prediction_ = false;
30 return;
31 }
32
33 double delta = (this->_last_update_ - temp_this_last_update).seconds();
34 Eigen::VectorXf temp_x_vector = this->_x_vector_;
35 this->_x_vector_ = motion_model->predict_expected_state(temp_x_vector, motion_update, delta);
36
37 Eigen::MatrixXf g_matrix =
38 motion_model->get_motion_to_state_matrix(temp_x_vector, motion_update, delta);
39 Eigen::MatrixXf r_matrix = motion_model->get_process_noise_covariance_matrix(
40 static_cast<unsigned int>(this->_x_vector_.size())); // Process Noise Matrix
41
42 this->_p_matrix_ = g_matrix * Eigen::MatrixXf(this->_p_matrix_) * g_matrix.transpose() + r_matrix;
43}

73 // get expect observation from the state vector with the observation model
74 Eigen::Vector2f z_hat = this->_observation_model_->observation_model(this->_x_vector_, j);
75 // print dbg msg z_hat and cone in the samle line
76 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "Z_HAT X|%f| Y|%f| CONE X|%f| Y|%f|",

79 Eigen::MatrixXf h_matrix = this->_observation_model_->get_state_to_observation_matrix(
80 this->_x_vector_, j, static_cast<unsigned int>(this->_x_vector_.size()));
81 // get the observation noise covariance matrix
82 Eigen::MatrixXf q_matrix =
83 this->_observation_model_->get_observation_noise_covariance_matrix();
84 // calculate innovation cone position - z_hat(predicted cone position)
85 Eigen::Vector2f innovation = Eigen::Vector2f(cone.position.x, cone.position.y) - z_hat;

88 innovation(0), innovation(1));
89 // calculate the S matrix
90 Eigen::MatrixXf s_matrix = h_matrix * this->_p_matrix_ * h_matrix.transpose() + q_matrix;
91 // calculate nis, that means normalized innovation squared
92 float nis = innovation.transpose() * s_matrix.inverse() * innovation;
93 // dbg
94 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "NIS %f", nis);
95 // calculate nd, that means normalized determinant
96 double nd = nis + log(s_matrix.determinant());
97
98 // dbg
99 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "ND %f\n", nd);
100 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "CURRENT CONE: %f %f", cone.position.x,

108 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "ND %f", nd);

111 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "MATCHED CONE: %f %f", cone.position.x,

121 j_best = j;
122 } else if (nis < outer) {
123 outer = nis; // outer is the closest unmatched distance
124 }
125 }
126 if (j_best != 0) {
127 // dbg msg with the match cone position and the best match position
128 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "ADDING TO CORRECT MATCHED CONE %f %f",

131 matched_cone_positions.push_back(Eigen::Vector2f(cone.position.x, cone.position.y));
132 } else if (outer > 20 /* gate2 */) { // TUNE
133 new_features.push_back(Eigen::Vector2f(cone.position.x, cone.position.y));
134 }
135 }
136 // loop through the matched ids and cone positions and make the corrections
137 for (int i = 0; i < matched_ids.size(); i++) {
138 Eigen::Vector2f z_hat =
139 this->_observation_model_->observation_model(this->_x_vector_, matched_ids[i]);

141 z_hat(1));
142 Eigen::MatrixXf h_matrix = this->_observation_model_->get_state_to_observation_matrix(
143 this->_x_vector_, matched_ids[i], static_cast<unsigned int>(this->_x_vector_.size()));
144 Eigen::MatrixXf q_matrix = this->_observation_model_->get_observation_noise_covariance_matrix();

146 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "MATCHED Z: %f %f", z(0), z(1));

148 (z - z_hat)(1));

150 Eigen::MatrixXf kalman_gain_matrix =
151 this->get_kalman_gain(h_matrix, this->_p_matrix_, q_matrix);
152 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "BEFORE CORRECTION %f %f %f %f %f %f",
153 this->_x_vector_(0), this->_x_vector_(1), this->_x_vector_(2), this->_x_vector_(3),
154 this->_x_vector_(4), this->_x_vector_(5));
155 this->_x_vector_ = this->_x_vector_ + kalman_gain_matrix * (z - z_hat);
156 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "AFTER CORRECTION %f %f %f %f %f %f",

163 this->_p_matrix_;
164 }

166 // loop through the new features and add them to the state vector
167 for (int i = 0; i < new_features.size(); i++) {

175 common_lib::competition_logic::Color::BLUE));
176 this->_x_vector_(this->_x_vector_.size() - 2) = landmark_absolute(0);
177 this->_x_vector_(this->_x_vector_.size() - 1) = landmark_absolute(1);

180 Eigen::MatrixXf::Zero(this->_p_matrix_.rows() + 2, this->_p_matrix_.cols() + 2);
181 new_p_matrix.block(0, 0, this->_p_matrix_.rows(), this->_p_matrix_.cols()) = this->_p_matrix_;
182 this->_p_matrix_ = new_p_matrix;
183
184 // Get Jacobians
185 Eigen::MatrixXf Gv = _observation_model_->get_gv(
186 _x_vector_, ObservationData(new_features[i](0), new_features[i](1),
187 common_lib::competition_logic::Color::BLUE));
188
189 Eigen::MatrixXf Gz = _observation_model_->get_gz(
190 _x_vector_, ObservationData(new_features[i](0), new_features[i](1),
191 common_lib::competition_logic::Color::BLUE));
192
193 // Get R (measurement noise covariance)
194 Eigen::Matrix2f R = _observation_model_->get_observation_noise_covariance_matrix();
195
196 Eigen::MatrixXf GvT = Gv.transpose();
197 Eigen::MatrixXf GzT = Gz.transpose();
198
199 Eigen::MatrixXf P_submatrix = this->_p_matrix_.block(0, 0, 3, 3);
200
201 Eigen::MatrixXf new_submatrix = Gv * P_submatrix * GvT + Gz * R * GzT;
202
203 this->_p_matrix_.block(length, length, 2, 2) = new_submatrix;
204
205 P_submatrix = this->_p_matrix_.block(0, 0, 3, 3);
206 new_submatrix = Gv * P_submatrix;
207 this->_p_matrix_.block(length, 0, 2, 3) = new_submatrix;
208
209 P_submatrix = this->_p_matrix_.block(length, 0, 2, 3);
210 new_submatrix = P_submatrix.transpose();
211 this->_p_matrix_.block(0, length, 3, 2) = new_submatrix;
212
213 if (length > 6) {
214 this->_p_matrix_.block(length, 6, 2, length - 6) =
215 Gv * this->_p_matrix_.block(0, 6, 3, length - 6);
216 this->_p_matrix_.block(6, length, length - 6, 2) =
217 (this->_p_matrix_.block(length, 6, 2, length - 6)).transpose();
218 }
219 }
220 RCLCPP_DEBUG(rclcpp::get_logger("ekf_state_est"), "STATE AFTER CORRECT %f %f %f %f %f %f",
221 this->_x_vector_(0), this->_x_vector_(1), this->_x_vector_(2), this->_x_vector_(3),
222 this->_x_vector_(4), this->_x_vector_(5));

224Eigen::MatrixXf ExtendedKalmanFilter::get_kalman_gain(const Eigen::MatrixXf &h_matrix,
225 const Eigen::MatrixXf &p_matrix,
226 const Eigen::MatrixXf &q_matrix) const {
227 Eigen::MatrixXf s_matrix = h_matrix * p_matrix * h_matrix.transpose() + q_matrix;
228
229 if ((s_matrix.array() == 0).all()) {
230 return Eigen::MatrixXf::Zero(p_matrix.rows(), h_matrix.rows());
231 }
232 Eigen::MatrixXf kalman_gain_matrix = p_matrix * h_matrix.transpose() * s_matrix.inverse();
233
234 return kalman_gain_matrix;
235}
