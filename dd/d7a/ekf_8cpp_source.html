25void ExtendedKalmanFilter::prediction_step(const MotionUpdate &motion_update) {
26 double delta = (motion_update.last_update - this->_last_update_).seconds();
27 this->_last_update_ = motion_update.last_update;
28 if (this->_first_prediction_) { // To set the last update correctly
29 this->_first_prediction_ = false;
30 return;
31 }
32 Eigen::VectorXf temp_x_vector = this->_x_vector_;
33 this->_x_vector_ =
34 this->_motion_model_.predict_expected_state(temp_x_vector, motion_update, delta / 1000000);
35 Eigen::MatrixXf g_matrix = this->_motion_model_.get_motion_to_state_matrix(
36 temp_x_vector, motion_update, delta / 1000000);
37 Eigen::MatrixXf r_matrix = this->_motion_model_.get_process_noise_covariance_matrix(
38 static_cast<unsigned int>(this->_x_vector_.size())); // Process Noise Matrix
39 // this->_p_matrix_ = g_matrix * this->_p_matrix_ * g_matrix.transpose() + r_matrix;
40 Eigen::MatrixXf p_matrix_dense =
41 g_matrix * Eigen::MatrixXf(this->_p_matrix_) * g_matrix.transpose() + r_matrix;
42 this->_p_matrix_ = p_matrix_dense.sparseView();
43}

45void ExtendedKalmanFilter::correction_step(
46 const std::vector<common_lib::structures::Cone> &perception_map) {
47 for (const common_lib::structures::Cone &cone : perception_map) {
48 ObservationData observation_data =
49 ObservationData(cone.position.x, cone.position.y, cone.color);
50
51 // Data Association
52 Eigen::Vector2f landmark_absolute =
53 this->_observation_model_.inverse_observation_model(this->_x_vector_, observation_data);
54 int landmark_index = _data_association_model_.match_cone(landmark_absolute, this->_x_vector_);
55 if (landmark_index == -1) { // Too far away landmark
56 continue;
57 } else if (landmark_index == -2) { // Did not match any landmark
58 this->_x_vector_.conservativeResizeLike(Eigen::VectorXf::Zero(this->_x_vector_.size() + 2));
59 this->_x_vector_(this->_x_vector_.size() - 2) = landmark_absolute(0);
60 this->_x_vector_(this->_x_vector_.size() - 1) = landmark_absolute(1);
61 Eigen::SparseMatrix<float> new_p_matrix(this->_p_matrix_.rows() + 2,
62 this->_p_matrix_.cols() + 2);
63 common_lib::maths::copy_eigen_sparse_matrix(this->_p_matrix_, new_p_matrix);
64 this->_p_matrix_.swap(new_p_matrix);
65 landmark_index = static_cast<int>(this->_x_vector_.size() - 2);
66 }
67
68 // Matrix Calculation
69 Eigen::MatrixXf h_matrix = this->_observation_model_.get_state_to_observation_matrix(
70 this->_x_vector_, landmark_index, static_cast<unsigned int>(this->_x_vector_.size()));
71 Eigen::MatrixXf q_matrix =
72 this->_observation_model_.get_observation_noise_covariance_matrix(); // Observation Noise
73 // Matrix
74 Eigen::MatrixXf kalman_gain_matrix =
75 this->get_kalman_gain(h_matrix, this->_p_matrix_, q_matrix);
76
77 // Correction
78 Eigen::Vector2f z_hat = this->_observation_model_.observation_model(
79 this->_x_vector_, landmark_index); // expected observation, previously
80 // calculated landmark is used
81 Eigen::Vector2f z =
82 Eigen::Vector2f(observation_data.position.x,
83 observation_data.position.y); // position of the landmark observed
84 this->_x_vector_ = this->_x_vector_ + kalman_gain_matrix * (z - z_hat);
85 // TODO(PedroRomao3): divide into multiple calculation and measure time.
86 // TODO(PedroRomao3): Diagram.
87 this->_p_matrix_ =
88 (Eigen::MatrixXf::Identity(this->_p_matrix_.rows(), this->_p_matrix_.cols()) -
89 kalman_gain_matrix * h_matrix)
90 .sparseView() *
91 this->_p_matrix_;
92 }
93}

95Eigen::MatrixXf ExtendedKalmanFilter::get_kalman_gain(const Eigen::MatrixXf &h_matrix,
96 const Eigen::MatrixXf &p_matrix,
97 const Eigen::MatrixXf &q_matrix) const {
98 Eigen::MatrixXf s_matrix = h_matrix * p_matrix * h_matrix.transpose() + q_matrix;
99 Eigen::MatrixXf kalman_gain_matrix = p_matrix * h_matrix.transpose() * s_matrix.inverse();
100 return kalman_gain_matrix;
101}
