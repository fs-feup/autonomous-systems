25void ExtendedKalmanFilter::prediction_step(const MotionUpdate &motion_update) {
26 rclcpp::Time temp_this_last_update = this->_last_update_;
27 this->_last_update_ = motion_update.last_update;
28 if (this->_first_prediction_) { // To set the last update correctly
29 this->_first_prediction_ = false;
30 return;
31 }
32 double delta = (this->_last_update_ - temp_this_last_update).seconds();
33 Eigen::VectorXf temp_x_vector = this->_x_vector_;
34 this->_x_vector_ =
35 this->_motion_model_->predict_expected_state(temp_x_vector, motion_update, delta);
36 Eigen::MatrixXf g_matrix =
37 this->_motion_model_->get_motion_to_state_matrix(temp_x_vector, motion_update, delta);
38 Eigen::MatrixXf r_matrix = this->_motion_model_->get_process_noise_covariance_matrix(
39 static_cast<unsigned int>(this->_x_vector_.size())); // Process Noise Matrix
40 // this->_p_matrix_ = g_matrix * this->_p_matrix_ * g_matrix.transpose() + r_matrix;
41 Eigen::MatrixXf p_matrix_dense =
42 g_matrix * Eigen::MatrixXf(this->_p_matrix_) * g_matrix.transpose() + r_matrix;
43 this->_p_matrix_ = p_matrix_dense.sparseView();
44}

46void ExtendedKalmanFilter::correction_step(
47 const std::vector<common_lib::structures::Cone> &perception_map) {
48 for (const common_lib::structures::Cone &cone : perception_map) {
49 ObservationData observation_data =
50 ObservationData(cone.position.x, cone.position.y, cone.color);
51
52 // Data Association
53 Eigen::Vector2f landmark_absolute =
54 this->_observation_model_->inverse_observation_model(this->_x_vector_, observation_data);
55 int landmark_index = _data_association_model_->match_cone(landmark_absolute, this->_x_vector_);
56 if (landmark_index == -1) { // Too far away landmark
57 continue;
58 } else if (landmark_index == -2) { // Did not match any landmark
59 this->_x_vector_.conservativeResizeLike(Eigen::VectorXf::Zero(this->_x_vector_.size() + 2));
60 this->_x_vector_(this->_x_vector_.size() - 2) = landmark_absolute(0);
61 this->_x_vector_(this->_x_vector_.size() - 1) = landmark_absolute(1);
62 Eigen::SparseMatrix<float> new_p_matrix(this->_p_matrix_.rows() + 2,
63 this->_p_matrix_.cols() + 2);
64 common_lib::maths::copy_eigen_sparse_matrix(this->_p_matrix_, new_p_matrix);
65 this->_p_matrix_.swap(new_p_matrix);
66 landmark_index = static_cast<int>(this->_x_vector_.size() - 2);
67 }
68
69 // Matrix Calculation
70 Eigen::MatrixXf h_matrix = this->_observation_model_->get_state_to_observation_matrix(
71 this->_x_vector_, landmark_index, static_cast<unsigned int>(this->_x_vector_.size()));
72 Eigen::MatrixXf q_matrix =
73 this->_observation_model_->get_observation_noise_covariance_matrix(); // Observation Noise
74 // Matrix
75 Eigen::MatrixXf kalman_gain_matrix =
76 this->get_kalman_gain(h_matrix, this->_p_matrix_, q_matrix);
77
78 // Correction
79 Eigen::Vector2f z_hat = this->_observation_model_->observation_model(
80 this->_x_vector_, landmark_index); // expected observation, previously
81 // calculated landmark is used
82 Eigen::Vector2f z =
83 Eigen::Vector2f(observation_data.position.x,
84 observation_data.position.y); // position of the landmark observed
85 this->_x_vector_ = this->_x_vector_ + kalman_gain_matrix * (z - z_hat);
86 // TODO(PedroRomao3): divide into multiple calculation and measure time.
87 // TODO(PedroRomao3): Diagram.
88 this->_p_matrix_ =
89 (Eigen::MatrixXf::Identity(this->_p_matrix_.rows(), this->_p_matrix_.cols()) -
90 kalman_gain_matrix * h_matrix)
91 .sparseView() *
92 this->_p_matrix_;
93 }
94}

96Eigen::MatrixXf ExtendedKalmanFilter::get_kalman_gain(const Eigen::MatrixXf &h_matrix,
97 const Eigen::MatrixXf &p_matrix,
98 const Eigen::MatrixXf &q_matrix) const {
99 Eigen::MatrixXf s_matrix = h_matrix * p_matrix * h_matrix.transpose() + q_matrix;
100 Eigen::MatrixXf kalman_gain_matrix = p_matrix * h_matrix.transpose() * s_matrix.inverse();
101 return kalman_gain_matrix;
102}
